\begin{frame}{Matrix Notation}

\vskip 1.5em

\begin{columns}[T]
\begin{column}{0.18\textwidth}
\input{\CommonPath/tikz/mlp_multi_input_withouty.tex}
\end{column}
\begin{column}{0.27\textwidth}
\structure{Scalar notation:}
\begin{align*}
h_j & = \act\!\left(\theta_{j0} + \textstyle\sum_{i=1}^{d} \theta_{ji}\, x_i\right) \\
j &= 1,\ldots,k
\end{align*}

\vskip 0.5em

\structure{Matrix notation:}
\[ \vh = \act\!\left(\pvec_0 + \boldsymbol{\Theta}\vx\right) \in \R^k \]
\end{column}
\begin{column}{0.55\textwidth}
\structure{Collect weights into matrix:}
\[
\boldsymbol{\Theta} =
\begin{pmatrix}
\theta_{11} & \cdots & \theta_{1d} \\
\vdots & \ddots & \vdots \\
\theta_{k1} & \cdots & \theta_{kd}
\end{pmatrix}
\in \R^{k \times d}, \quad
\pvec_0 =
\begin{pmatrix} \theta_{10} \\ \vdots \\ \theta_{k0} \end{pmatrix}
\in \R^k
\]
\vskip 1.3em
activation $\act$ applied \textbf{elementwise}
\end{column}
\end{columns}

\vskip 1em

\structure{Each column of $\boldsymbol{\Theta}$} = weights from one input
\end{column}
\end{columns}

\begin{frame}{Layer Notation}
 
\vskip 1em

\centering
\structure{Index layers} $l = 0, 1, \ldots, L$

\begin{columns}[c]
\begin{column}{0.05\textwidth}
\end{column}
\begin{column}{0.5\textwidth}
\vskip 1em

\structure{Input layer} $l=0$
\[ \vh^{(0)} = \vx \]
 
\vskip 1em

\structure{Hidden layers} $l = 1, \ldots, L-1$:
\[
\vh^{(l)} = \act\!\left(\thetavec{l}_0 + \thetamat{l} \vh^{(l-1)}\right)
\]

\vskip 1em

\structure{Output layer} $l = L$:
\[
\yhat = \thetavec{L}_0 + \thetamat{L} \vh^{(L-1)}
\]
 
\end{column}
\begin{column}{0.4\textwidth}
\centering
\input{\CommonPath/tikz/mlp_layer_notation.tex}
\vskip 1em
\alert{\textbf{Forward pass:}} evaluate layer by layer, $l = 1 \to L$
\end{column}
\begin{column}{0.1\textwidth}
\end{column}
\end{columns}
 
\end{frame}

\begin{frame}{Deep Network as Function Composition}

\vskip 1.5em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Layer by layer:}
\[
\vx = \vh^{(0)} \xrightarrow{\thetamat{1}} \vh^{(1)} \xrightarrow{\thetamat{2}} \cdots \xrightarrow{\thetamat{L}} \vh^{(L)} = \yhat
\]

\vskip 1em

\structure{As a composed function:}
\[
\yhat = f_\theta(\vx) = \left(h^{(L)}_\theta \circ \cdots \circ h^{(1)}_\theta\right)(\vx)
\]

\vskip 1em

\structure{Written out:}
\[
\yhat = \thetavec{L}_0 + \thetamat{L}\,\act\!\left(\cdots\,\act\!\left(\thetavec{1}_0 + \thetamat{1}\vx\right)\right)
\]

\end{column}
\begin{column}{0.44\textwidth}
\centering
\input{\CommonPath/tikz/mlp_layer_notation.tex}
\vskip 0.5em
\structure{Total number of parameters:}
\[
|\pvec| = \sum_{l=1}^{L} k_l \cdot (k_{l-1} + 1)
\]
\alert{Real networks have millions - finding them is the challenge.}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Batch Formulation}
 
\vskip 1.5em
 
\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Single sample} $\vx \in \R^d$:
\[ \vh^{(l)} = \act\!\left(\thetavec{l}_0 + \thetamat{l} \vh^{(l-1)}\right) \in \R^{k_l} \]
 
\vskip 0.5em
 
\structure{Batch of $N$ samples} - stack as rows:
\[ \mathbf{X} \in \R^{N \times d}, \quad \mathbf{H}^{(l)} \in \R^{N \times k_l} \]
 
\vskip 0.5em
 
\alert{\textbf{Batch forward pass:}}
\[ \mathbf{H}^{(l)} = \act\!\left(\mathbf{1}\thetavec{l}_0{}\T + \mathbf{H}^{(l-1)} \left(\thetamat{l}\right)\T\right) \]
 
\end{column}
\begin{column}{0.44\textwidth}
\structure{PyTorch convention: batch first}
\vskip 1em
\centering

{\small
\texttt{X.shape = (N, d)}\\
\texttt{W.shape = (k, d)}\\
\texttt{H = X @ W.T + b}
}
\vskip 0.5em
\alert{This is what PyTorch computes.}

\section{Loss Functions}
% -------------------------------------------------------
 
\begin{frame}{What is a Loss Function?}
 
\vskip 1em
 
\structure{Goal:} measure how wrong $f_\theta$ is with a single scalar
 
\vskip 1.5em
 
\begin{columns}[t]
\begin{column}{0.45\textwidth}
\structure{Per-sample loss (cost) \ } $\ell(\yhat, y)$:
\begin{itemize}
\item $\ell(\yhat, y) = 0$ when $\yhat = y$ (perfect)
\item $\ell(\yhat, y) > 0$ otherwise
\item Larger when more wrong
\end{itemize}
 
\vskip 1em
 
\structure{Dataset loss \ } $\loss(\pvec)$:
\[ \loss(\pvec) = \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell\!\left(f_\theta(\vx^{(i)}), y^{(i)}\right) \]
Average over all training samples
\end{column}
\begin{column}{0.45\textwidth}
\structure{Choice of $\ell$ depends on the task:}
 
\vskip 0.8em
 
\begin{tabular}{ll}
\structure{Task} & \structure{Loss} \\
\hline\\[-0.8em]
Regression & Mean Squared Error \\[0.3em]
Binary classification & Binary Cross-Entropy \\[0.3em]
Multiclass & Cross-Entropy \\
\end{tabular}
 
\vskip 1.5em
\alert{Loss: model compass - what is ``better''?}
\end{column}
\begin{column}{0.05\textwidth}
\
\end{column}
\end{columns}
 
\end{frame}
 
% -------------------------------------------------------
 
\begin{frame}{Regression: Mean Squared Error}
 
\vskip 1em

\begin{beamercolorbox}[rounded=true, sep=0.8em]{highlight body}
% \centering
\begin{columns}[c]
\begin{column}{0.45\textwidth}
\alert{\textbf{Target \& prediction dim match}}\\[0.5em]
Scalar output: $y \in \R, \, \yhat \in \R$
\end{column}
\begin{column}{0.45\textwidth}
\alert{\textbf{Last layer has no activation}}\\[0.5em]
Raw output: $\yhat = \vh^{(L)} = \thetavec{L}_0 + \thetamat{L}\vh^{(L-1)}$
\end{column}
\end{columns}
\end{beamercolorbox}

\begin{columns}[c]
\begin{column}{0.52\textwidth}
 
\vskip 1em
 
\structure{Mean Squared Error (MSE):}
\[ \ell(\yhat, y) = (\yhat - y)^2, \quad \loss(\pvec) = \frac{1}{\nsamples}\sum_{i=1}^{\nsamples}\!\left(\yhat^{(i)} - y^{(i)}\right)^2 \]
 
 
\structure{Intuition:}
\begin{itemize}
\item + / - values $\to$ absolute or square
\item Large errors quadratically
\item Differentiable
\end{itemize}
\end{column}

\centering

\end{frame}

\section{Gradient Descent}
% -------------------------------------------------------
 
\begin{frame}{The Optimization Problem}
 
\vskip 1em
 
% \begin{columns}[c]
% \begin{column}{0.52\textwidth}
\structure{Goal:} find parameters that \alert{\textbf{minimize loss}}
\[ \pvec^* = \arg\min_{\pvec}\ \loss(\pvec) \]
 
\vskip 0.8em
 
\structure{Why not solve analytically?}
\begin{itemize}
\item $\loss(\pvec)$ is \textbf{nonlinear} in $\pvec$ --- no closed form
\item $\pvec$ has millions of dimensions in practice
\item $\loss(\pvec)$ is \textbf{non-convex} --- many local minima
\end{itemize}
 
\vskip 0.8em
 
\structure{Strategy:} iterative, gradient-based optimization
% \end{column}
% % \begin{column}{0.44\textwidth}
% \centering
% \input{\CommonPath/tikz/loss_surface.tex}
% \end{column}
% \begin{column}{0.04\textwidth}
% \end{column}
% \end{columns}
 
\end{frame}
 
% -------------------------------------------------------
 
\begin{frame}{Gradient Descent}
 
\vskip 2em
 
\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Gradient} $\nabla_{\pvec} \loss(\pvec)$:
\begin{itemize}
\item Points in direction of \textbf{steepest ascent}
\item One partial derivative per parameter: $\frac{\partial \loss}{\partial \theta_j}$
\end{itemize}
 
\vskip 1em
 
\structure{Gradient descent update:}
\[
\pvec \leftarrow \pvec - \eta\, \nabla_{\pvec} \loss(\pvec)
\]
\begin{itemize}
\item Step \textbf{opposite} to gradient $\Rightarrow$ go downhill
\item $\eta > 0$: \textbf{learning rate} --- controls step size
\item Repeat until convergence
\end{itemize}
