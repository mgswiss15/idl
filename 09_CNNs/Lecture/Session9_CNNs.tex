\section{Why Not MLPs for Images?}
% -------------------------------------------------------

\begin{frame}{Images in Computers}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.7\textwidth}

\structure{Image:} 3D tensor of shape $C \times H \times W$
\begin{itemize}
    \item Each entry: pixel intensity, integer 0--255 or float 0--1,
    \item $2^8$ possible values $\sim 8$ bits
    \item \structure{Grayscale} ($C=1$): single channel, 0 = black, 255 = white
    \item \structure{RGB} ($C=3$): red, green, blue --- standard cameras, screens
\end{itemize}

\end{column}
\begin{column}{0.25\textwidth}
\centering
\includegraphics[width=\textwidth]{\CommonPath/Pics/rgb.png}
\end{column}
\end{columns}

\structure{Other formats:}
\begin{itemize}
    \item RGBA ($C=4$): RGB + $\alpha$ transparency
    \item CMYK ($C=4$): print format
    \item Medical/hyperspectral: many channels, often 16-bit depth
\end{itemize}

\vskip 1em

\centering
\alert{DL mostly \textbf{RGB and grayscale} --- other formats typically converted before use}

\end{frame}

\begin{frame}{Images as Data}

\vskip 1em

\structure{Image sizes --- from benchmarks to real world:}

\begin{center}
\begin{tabular}{llll}
\toprule
\textbf{Source} & \textbf{Size} & \textbf{Channels} & \textbf{Pixels} \\
\midrule
MNIST & $28 \times 28$ & 1 & 784 \\
ImageNet & $224 \times 224$ & 3 & 150\,528 \\
Phone camera & $4000 \times 3000$ & 3 & 36\,000\,000 \\
Full HD screen & $1920 \times 1080$ & 3 & 6\,220\,800 \\
Medical (CT slice) & $512 \times 512$ & 1 & 262\,144 \\
\bottomrule
\end{tabular}
\end{center}

\vskip 1em

\structure{MLP approach:} flatten image to vector $\vx \in \mathbb{R}^{H \cdot W \cdot C}$, apply fully connected layers

\vskip 0.5em

ImageNet: first hidden from $\R^{150\,528} \to \R^{1024} \Rightarrow \approx 154$M parameters --- one layer only!

\vskip 2em

\centering
\alert{\large How to handle high-dimensional data efficiently?}

\end{frame}

\begin{frame}{How to Look at Images}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Human vision:}
\begin{itemize}
    \item Looks for local patterns --- edges, corners, textures
    \item Same pattern recognised anywhere in the visual field
    \item Simple features combine into complex ones
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Technical equivalent:}
\begin{itemize}
    \item \structure{Local connectivity} --- connect to small patches only
    \item \structure{Parameter sharing} --- same filter at every location
    \item \structure{Hierarchical features} --- edges $\to$ shapes $\to$ objects
\end{itemize}
\end{column}
\end{columns}

\vskip 2em

\begin{center}
Three key properties: \textbf{\alert{locality}, \alert{translation invariance}, \alert{parameter efficiency}}\\[1em]
These define the \textbf{convolutional layer}
\end{center}

\end{frame}

\begin{frame}{Convolution --- The Operation}

\vskip 1em

\centering

\structure{Single step:} dot product of local patch and filter $\to$ one output value\\[0.5em]
\tikzsetnextfilename{conv2d}
\input{\CommonPath/tikz/conv2d.tex}

\vskip 0.5em

\structure{Full operation:} same filter slides across input --- each position produces one output value\\[0.5em]
\tikzsetnextfilename{conv2d_steps}
\input{\CommonPath/tikz/conv2d_steps.tex}

\vskip 0.5em

\structure{All 3 requirements:} local connectivity, parameter sharing, translation invariance

\end{frame}

\begin{frame}{1D Convolution}

\vskip 2em

\structure{Filter $\kernel \in \mathbb{R}^k$} slides over signal $\vx \in \mathbb{R}^n$ --- at each position $i$, dot product with local patch

\begin{columns}[c]
\begin{column}{0.4\textwidth}
\centering
\tikzsetnextfilename{conv1d}
\input{\CommonPath/tikz/conv1d.tex}
\end{column}
\begin{column}{0.55\textwidth}

\[
    (\vx * \kernel)[i] = \sum_{j=0}^{k-1} \vx[i+j]\; \kernel[j]
\]

\begin{itemize}
    \item $i$ --- output position; $j$ --- index within filter
    \item Dot product of filter with local patch
\end{itemize}

\vskip 1em

\structure{Example:} at position $i=1$:
\[
    1{\cdot}1 + 3{\cdot}(-1) + 4{\cdot}2 = 6
\]

\end{column}
\end{columns}

\end{frame}

\begin{frame}{2D Convolution}

\vskip 2em

\structure{Extend to 2D:} filter $\kernel \in \mathbb{R}^{k \times k}$ slides over image $\mX \in \mathbb{R}^{H \times W}$:
\[
    (\mX * \kernel)[i,j] = \sum_{p=0}^{k-1}\sum_{q=0}^{k-1} \mX[i+p,\, j+q]\; \kernel[p,q]
\]

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\centering
\tikzsetnextfilename{conv2d}
\input{\CommonPath/tikz/conv2d.tex}
\end{column}
\begin{column}{0.5\textwidth}


\begin{itemize}
    \item $(i,j)$ --- output position
    \item $(p,q)$ --- index within filter
    \item Two sums: over rows and columns of filter
\end{itemize}
\end{column}
\end{columns}

\vskip 1em

\centering
\structure{Output size:} $(H - k + 1) \times (W - k + 1)$


\end{frame}

\begin{frame}{Padding}

\vskip 1em

\structure{Problem:} convolution shrinks spatial dimensions --- each layer loses $k-1$ pixels

\vskip 1em
\centering
\resizebox{0.6\textwidth}{!}{%
    \tikzsetnextfilename{conv2d_steps}
    \input{\CommonPath/tikz/conv2d_steps.tex}%
}

\vskip 1em


\begin{columns}[t]
\begin{column}{0.5\textwidth}
\tikzsetnextfilename{conv2d_padding}
\input{\CommonPath/tikz/conv2d_padding.tex}
\end{column}
\begin{column}{0.5\textwidth}
\structure{Valid padding} ($p=0$)
\begin{itemize}
    \item No zeros added
    \item Output shrinks: $(H-k+1) \times (W-k+1)$
    \item Spatial information lost at borders
\end{itemize}
\structure{Same padding} ($p=(k-1)/2$)
\begin{itemize}
    \item Zeros added around border
    \item Output same size as input
    \item Standard choice in most architectures
\end{itemize}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Stride}

\vskip 1em

\structure{Stride $s$:} filter $\kernel$ moves $s$ pixels at each step instead of 1

\vskip 1em

\tikzsetnextfilename{conv2d_stride}
\input{\CommonPath/tikz/conv2d_stride.tex}

\vspace{-2.5cm}\hspace{8cm}%
\begin{minipage}{0.4\textwidth}
\structure{Stride 1:} $\kernel$ moves one step a time\\[0.3em]
\structure{Stride 2:} $\kernel$ skips one - output halved\\[0.5em]
\end{minipage}

\vspace{0.5cm}%
\[
   \hspace{6cm} \text{\alert{\textbf{General output size:}}} \quad \left\lfloor \frac{H + 2p - k}{s} \right\rfloor + 1
\]

\end{frame}

\begin{frame}{Multiple Input Channels}

\vskip 1em

\structure{RGB input:} $C_{in}$ channels --- filter must have same depth $C_{in}$

\vskip 1em

\structure{Operation:} filter has $C_{in}$ slices --- each slice convolves with one input channel

\[
    (\mX * \kernel)[i,j] = \sum_{c=0}^{C_{in}-1}\underbrace{\sum_{p=0}^{k-1}\sum_{q=0}^{k-1}
    \mX[c,\, i+p,\, j+q]\; \kernel[c,\, p,\, q]}_{\text{conv with channel } c}
\]

\begin{itemize}
    \item Inner double sum: convolution of filter slice $c$ with input channel $c$
    \item Outer sum over $c$: results summed across all channels
    \item One filter $\to$ one output feature map
\end{itemize}

\begin{center}
\tikzsetnextfilename{conv2d_channels}
\input{\CommonPath/tikz/conv2d_channels.tex}
\end{center}

\end{frame}

\begin{frame}{Multiple Output Channels}

\vskip 1em

\structure{$C_{out}$ filters:} each filter independently applied to the full input volume

\vskip 0.5em

\[
    \vh_f[i,j] = \sum_{c=0}^{C_{in}-1}\sum_{p=0}^{k-1}\sum_{q=0}^{k-1}
    \mX[c,\, i+p,\, j+q]\; \kernel_f[c,\, p,\, q]
    \quad f = 0,\ldots,C_{out}-1
\]

\begin{center}
\tikzsetnextfilename{conv2d_filters}
\input{\CommonPath/tikz/conv2d_filters.tex}
\end{center}

\begin{itemize}
    \item Each filter $\kernel_f$ produces one feature map $\vh_f$ independently
    \item Output tensor: $C_{out} \times H' \times W'$
    \item Parameters: $C_{out} \times C_{in} \times k \times k + C_{out}$ (with bias)
\end{itemize}

\end{frame}

\begin{frame}{Receptive Field}

\vskip 2em

\structure{Receptive field:} region of the input that influences a given neuron's output

\begin{center}
\tikzsetnextfilename{receptive_field}
\input{\CommonPath/tikz/receptive_field.tex}
\end{center}

\vskip 0.5em

\begin{itemize}
    \item Each $3\times3$ conv layer adds 2 pixels to the receptive field in each dimension
    \item $L$ layers of $3\times3$ filters: receptive field $= (2L+1)\times(2L+1)$
    \item Deep networks see large regions despite small local filters
    \item Early layers: local features (edges) --- later layers: global patterns (objects)
\end{itemize}

\end{frame}

\end{center}

\begin{frame}{Pooling}

\vskip 2em

\structure{Pooling:} spatial downsampling --- reduces feature map size

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Max pooling:}
\[
    y[i,j] = \max_{p,q \in \mathcal{R}_{ij}} x[p,q]
\]
\begin{itemize}
    \item Takes maximum over local region
    \item Retains strongest activation
    \item Most commonly used
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Average pooling:}
\[
    y[i,j] = \frac{1}{|\mathcal{R}_{ij}|}\sum_{p,q \in \mathcal{R}_{ij}} x[p,q]
\]
\begin{itemize}
    \item Takes average over local region
    \item Smoother downsampling
    \item Less common in hidden layers
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1.5em

\structure{Benefits:}
\begin{itemize}
    \item Reduces spatial dimensions --- fewer parameters in subsequent layers
    \item Local translation invariance --- small shifts in feature position do not affect output
\end{itemize}

\end{frame}
