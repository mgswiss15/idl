\section{From Gradients to Learning}
% -------------------------------------------------------

\begin{frame}{Where We Are}

\vskip 2em

\structure{So far we have:}
\begin{itemize}
    \item \alert{Network} $f_{\pvec} : \inspace \to \outspace$ - parameterized function
    \item \alert{Loss} $\loss(\pvec; \dataset)$ - measures how wrong $f_{\pvec}$ is
    \item \alert{Backpropagation} to compute $\nabla_{\pvec} \loss(\pvec)$ exactly and efficiently
\end{itemize}

\vskip 1.5em

\structure{Open question:}

\vskip 0.5em

\begin{center}
{\large How do we \textbf{use} those gradients to find good parameters $\pvec$?}
\end{center}

\vskip 1.5em

\structure{This lecture:} algorithms for minimizing $\loss(\pvec)$

\end{frame}

% -------------------------------------------------------

\begin{frame}{Optimization Problem}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.6\textwidth}

\structure{Goal:} find parameters minimizing empirical loss

\[
    \pvec^* = \arg\min_{\pvec}\; \loss(\pvec)
    = \arg\min_{\pvec}\; \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
\]

\vskip 1em

\structure{Why is this hard?}
\begin{itemize}
    \item $\pvec$ can have millions of dimensions
    \item $\loss$ is \textbf{non-convex} - no closed-form solution
    \item Computing $\loss(\pvec)$ over all data - expensive
\end{itemize}

\end{column}

\section{Stochastic and Mini-Batch Gradient Descent}
% -------------------------------------------------------

\begin{frame}{Cost of Full-Batch GD}
 
\vskip 2em
 
\structure{Full loss is an average over all samples:}
\[
    \loss(\pvec;\, \dataset) = \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
\]
 
\structure{Problem:} $\nsamples$ can be millions - one evaluation requires full pass through data
 
\vskip 1em
 
\structure{Key insight:} random subset $\mathcal{B} \subset \{1,\ldots,\nsamples\}$ (\textbf{mini-batch}) gives unbiased loss estimate:
\[
    \hat{\loss}(\pvec;\, \mathcal{B}) = \frac{1}{|\mathcal{B}|} \sum_{i \in \mathcal{B}} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
    \qquad \Rightarrow \qquad \mathbb{E}_{\mathcal{B}}\bigl[\hat{\loss}(\pvec;\,\mathcal{B})\bigr] = \loss(\pvec;\,\dataset)
\]
 
\structure{Gradient follows for free} (gradient is linear):
\[
    \mathbb{E}_{\mathcal{B}}\bigl[\nabla_{\pvec}\, \hat{\loss}(\pvec;\,\mathcal{B})\bigr] = \nabla_{\pvec}\, \loss(\pvec;\,\dataset)
\]
 
\end{frame}

% -------------------------------------------------------

\begin{frame}{Batch, Mini-Batch, and SGD}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Batch GD} \\[0.5em]
$|\mathcal{B}| = \nsamples$
\begin{itemize}
    \item Exact gradient
    \item Stable updates
    \item Very slow per epoch
    \item Converges smoothly
\end{itemize}
\end{column}
\begin{column}{0.32\textwidth}
\alert{\textbf{Mini-Batch GD (SGD)}} \\[0.5em]
$|\mathcal{B}| = B$ (e.g.\ 32--256)
\begin{itemize}
    \item Noisy estimate
    \item Fast updates
    \item Can escape minima
    \item GPU-friendly
\end{itemize}
\end{column}
\begin{column}{0.32\textwidth}
\structure{SGD} \\[0.5em]
$|\mathcal{B}| = 1$
\begin{itemize}
    \item Very noisy
    \item Cheapest per step
    \item High variance
    \item Theoretically solid
\end{itemize}
\end{column}
\end{columns}

\vskip 3em

\structure{Noise as feature:} stochastic updates help jump over sharp minima and saddle points 

\end{frame}

% -------------------------------------------------------

\begin{frame}{Training Loop}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.56\textwidth}
\begin{algorithm}[H]
\DontPrintSemicolon
\For{epoch $= 1, \ldots, T$}{
    \For{each mini-batch $\mathcal{B} \subset \dataset$}{
        \alert{forward}: $f_{\pvec}(\vx^{(i)})$ for $i \in \mathcal{B}$ \;
        \alert{loss}: $\hat{\loss} = \frac{1}{|\mathcal{B}|}\sum_{i\in\mathcal{B}} \ell_i$ \;
        \alert{backprop}: $\nabla_{\pvec} \hat{\loss}$ \;
        \alert{update}: $\pvec \leftarrow \pvec - \eta\,\nabla_{\pvec} \hat{\loss}$ \;
    }
}
\end{algorithm}
\end{column}
\begin{column}{0.40\textwidth}
\vskip 0.5em
\structure{Key concepts:}
\begin{itemize}
    \item \textbf{Epoch:} full pass through $\dataset$
    \item \textbf{Iteration:} mini-batch update
    \item \textbf{Shuffle} each epoch - break correlations
\end{itemize}
\end{column}
\end{columns}

\end{frame}

\section{Input Normalization and Weight Initialization}
% -------------------------------------------------------

\begin{frame}{Why Feature Scale Matters}
 
% \vskip 1em
\[
    \loss(\theta_1, \theta_2; \dataset) = \frac{1}{\nsamples}\sum_{i=1}^{\nsamples}
    \bigl(\theta_1 x_1^{(i)} + \theta_2 x_2^{(i)} - y^{(i)}\bigr)^2
\]
 
Say $x_1 \in [1, 4]$ (avg final MAI grade), $x_2 \in [0, 1500]$ (total days in MAI), $y$ income
 
\vskip 1em

\begin{columns}[c]
\begin{column}{0.65\textwidth}
 
\structure{Gradients:}
\[
    \frac{\partial \loss}{\partial \theta_1} = \frac{2}{\nsamples}\sum_i r^{(i)}\, x_1^{(i)}, \qquad
    \frac{\partial \loss}{\partial \theta_2} = \frac{2}{\nsamples}\sum_i r^{(i)}\, x_2^{(i)}
\]
 
where $r^{(i)} = \theta_1 x_1^{(i)} + \theta_2 x_2^{(i)} - y^{(i)}$

\[x_2 \gg x_1 \;\Rightarrow\; \dfrac{\partial \loss}{\partial \theta_2} \gg \dfrac{\partial \loss}{\partial \theta_1}\]
 
\end{column}
\begin{column}{0.25\textwidth}
\centering
\input{\CommonPath/tikz/normalization_ellipse.tex}
\end{column}
\begin{column}{0.02\textwidth}
\end{column}
\end{columns}

\vskip 1em

\centering
\alert{GD oscillates in $\theta_2$ direction, crawls in $\theta_1$ direction}

\end{frame}
 
% -------------------------------------------------------
 
\begin{frame}{Input Normalization}
 
\vskip 2em

\centering 
\structure{Solution:} rescale features so all dimensions have similar influence on loss
 
\vskip 1em

\begin{columns}[c]
\begin{column}{0.65\textwidth}
 
\structure{Z-score normalization (feature-wise):}
\[
    \tilde{x}_j = \frac{x_j - \mu_j}{\sigma_j}, \quad
    \mu_j = \frac{1}{\nsamples}\sum_i x_j^{(i)}, \quad
    \sigma_j = \sqrt{\frac{1}{\nsamples}\sum_i (x_j^{(i)} - \mu_j)^2}
\]
 
\vskip 1em
 
\begin{itemize}
    \item $\mu_j,\, \sigma_j$ computed on \textbf{training set only}
    \item Same transform applied to validation and test sets
    \item Elongated contours $\to$ circular $\Rightarrow$ GD converges faster
\end{itemize}
 
\end{column}
\begin{column}{0.25\textwidth}
\centering
\input{\CommonPath/tikz/normalization_circle.tex}
\end{column}
\end{columns}
 
\end{frame}

\begin{frame}{Initialization: The Symmetry Problem}
 
\vskip 2em
 
\structure{What if we fix weights, e.g. start at zero?}
 
\vskip 0.5em
 
All neurons in a layer compute identical pre-activations:
\[
    z_j^{(l)} = \sum_i \theta_{ji}^{(l)}\, h_i^{(l-1)} + \theta_{j0}^{(l)} = c
    \quad \forall\, j
\]
 
\vskip 0.5em
 
\begin{itemize}
    \item All neurons produce same output $\Rightarrow$ same gradient $\Rightarrow$ same update
    \item Symmetry never broken - network equivalent to single neuron per layer
    \item Width is wasted
\end{itemize}
 
\vskip 1em
 
\structure{Fix:} initialize weights \textbf{randomly} to break symmetry
 
\vskip 0.5em
 
But random from where? Scale matters enormously.
 
\end{frame}
 
% -------------------------------------------------------
 
\begin{frame}{Initialization: Vanishing and Exploding Signals}
 
\vskip 1.5em
 
\structure{Simple model:} $L$-layer network, linear activations, weights drawn i.i.d.\ with $\mathbb{E}[\theta]=0$, $\mathrm{Var}[\theta] = \sigma^2$, $n_{\text{in}}$ inputs per layer
 
\vskip 0.5em
 
Variance of pre-activation after one layer (assuming independent inputs):
\[
    \mathrm{Var}[z^{(l)}] = n_{\text{in}}\,\sigma^2\cdot \mathrm{Var}[h^{(l-1)}]
\]
 
After $L$ layers:
\[
    \mathrm{Var}[z^{(L)}] = (n_{\text{in}}\,\sigma^2)^L \cdot \mathrm{Var}[x]
\]
 
\vskip 0.8em
 
\begin{columns}[t]
\begin{column}{0.45\textwidth}
\structure{$n_{\text{in}}\,\sigma^2 < 1$:} signal \alert{vanishes}
\begin{itemize}
    \item Activations $\to 0$
    \item Gradients $\to 0$, early layers stop learning
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\structure{$n_{\text{in}}\,\sigma^2 > 1$:} signal \alert{explodes}
\begin{itemize}
    \item Activations $\to \infty$
    \item Gradients $\to \infty$, training diverges
\end{itemize}
\end{column}
\end{columns}
 
\vskip 0.8em

\centering
\structure{Key idea:} want $n_{\text{in}}\,\sigma^2 = 1$ $\Rightarrow$ set $\sigma^2 = 1/n_{\text{in}}$
 
\end{frame}
 
% -------------------------------------------------------
 
\begin{frame}{Xavier and He Initialization}
 
\vskip 2em
 
\structure{Xavier / Glorot} (Glorot \& Bengio, 2010) --- for \textbf{tanh / sigmoid}:
 
\vskip 0.3em
 
Balance variance preservation in both forward and backward pass $\Rightarrow$ compromise:
\[
    \sigma^2 = \frac{2}{n_{\text{in}} + n_{\text{out}}}, \qquad
    \theta^{(l)}_{ji} \sim \mathcal{U}\!\left[-\frac{\sqrt{6}}{\sqrt{n_{\text{in}}+n_{\text{out}}}},\, \frac{\sqrt{6}}{\sqrt{n_{\text{in}}+n_{\text{out}}}}\right]
\]
 
\vskip 1em
 
\structure{He / Kaiming} (He et al., 2015) --- for \textbf{ReLU}:
 
\vskip 0.3em
 
ReLU zeroes half the neurons $\Rightarrow$ effective variance halved $\Rightarrow$ need factor of 2:
\[
    \sigma^2 = \frac{2}{n_{\text{in}}}, \qquad
    \theta^{(l)}_{ji} \sim \mathcal{N}\!\left(0,\, \frac{2}{n_{\text{in}}}\right)
\]
 
 
\end{frame}

\section{Momentum and Adaptive Methods}
% -------------------------------------------------------

\begin{frame}{Limitations of Vanilla GD}

\vskip 2em

\structure{Even with good initialization and normalization - vanilla SGD struggles:}

\vskip 0.5em

\begin{itemize}
    \item \textbf{Same $\eta$ for all parameters} - not all directions need same step size
    \item \textbf{Oscillations in narrow valleys} - gradients point across valley, not toward minimum
    \item \textbf{Slow progress along flat directions}
    \item \textbf{Noisy mini-batch gradients} - updates jump around
\end{itemize}

\vskip 1.5em

\structure{Three ideas to fix this:}
\begin{enumerate}
    \item \textbf{Momentum} - smooth out noisy gradients
    \item \textbf{RMSProp} - adapt step size per parameter
    \item \textbf{Adam} - combine both
\end{enumerate}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Momentum}

\vskip 2em

\centering
\structure{Idea:} accumulate velocity in direction of consistent gradients

\vskip 0.5em

\begin{columns}[c]
\begin{column}{0.6\textwidth}

\[
    \vm_t = \beta\, \vm_{t-1} + (1-\beta)\, \nabla_{\pvec} \loss_t
\]
\[
    \pvec \leftarrow \pvec - \eta\, \vm_t
\]

\begin{itemize}
    \item $\beta \in [0,1)$: momentum coeff, typically 0.9
    \item $\vm_t$: \textbf{exponential moving avg} of gradients
    \item Consistent directions $\Rightarrow$ accumulate, accelerate
    \item Inconsistent directions $\Rightarrow$ cancel, dampen oscillations
\end{itemize}


\end{column}
\begin{column}{0.35\textwidth}
\centering
\input{\CommonPath/tikz/momentum_vs_sgd.tex}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{RMSProp}

\vskip 2em

\structure{Idea:} adapt learning rate \textbf{per parameter} based on recent gradient magnitudes

\[
    \vv_t = \gamma\, \vv_{t-1} + (1-\gamma)\, \bigl(\nabla_{\pvec} \loss_t\bigr)^2
\]
\[
    \pvec \leftarrow \pvec - \frac{\eta}{\sqrt{\vv_t + \epsilon}}\, \nabla_{\pvec} \loss_t
\]

\begin{itemize}
    \item $\vv_t$: exp.\ moving average of \textbf{squared} gradients, per parameter
    \item Large gradients $\Rightarrow$ \textbf{smaller} step
    \item Small gradients $\Rightarrow$ \textbf{larger} step
    \item $\epsilon > 0$ (e.g.\ $10^{-8}$) for numerical stability
\end{itemize}

\vskip 1em

\structure{Effect:} automatically rescales each dimension - good for ill-conditioned landscapes

\end{frame}

% -------------------------------------------------------

\begin{frame}{Adam}

\vskip 2em

\structure{Adam (Adaptive Moment Estimation):} combines momentum + RMSProp

\vskip 0.5em

\[
    \vm_t = \beta_1\, \vm_{t-1} + (1-\beta_1)\, \nabla_{\pvec} \loss_t
    \qquad \text{(first moment - mean)}
\]
\[
    \vv_t = \beta_2\, \vv_{t-1} + (1-\beta_2)\, \bigl(\nabla_{\pvec} \loss_t\bigr)^2
    \qquad \text{(second moment - variance)}
\]

\structure{Bias correction} (important early in training when $\vm_0 = \vv_0 = 0$):
\[
    \hat{\vm}_t = \frac{\vm_t}{1-\beta_1^t}, \qquad \hat{\vv}_t = \frac{\vv_t}{1-\beta_2^t}
\]

\structure{Update:}
\[
    \pvec \leftarrow \pvec - \frac{\eta}{\sqrt{\hat{\vv}_t}+\epsilon}\,\hat{\vm}_t
\]

\structure{Defaults:} $\beta_1 = 0.9$, $\beta_2 = 0.999$, $\epsilon = 10^{-8}$


\end{frame}

\section{Learning Rate Schedules}
% -------------------------------------------------------

\begin{frame}{Why Schedule the Learning Rate?}

\vskip 2em

\structure{Fixed learning rate - compromise:}
\begin{itemize}
    \item Large $\eta$ early - fast progress toward good region
    \item Small $\eta$ late - fine-grained convergence near minimum
    \item Single fixed $\eta$ cannot do both well
\end{itemize}

\vskip 1.5em

\structure{Solution:} vary $\eta$ over course of training

\vskip 1em

\structure{Three common schedules:}
\begin{enumerate}
    \item \textbf{Step decay} - reduce by factor every $k$ epochs
    \item \textbf{Cosine annealing} - smooth decay following cosine curve
    \item \textbf{Warmup} - start small, increase, then decay
\end{enumerate}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Learning Rate Schedules}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.54\textwidth}

\structure{Step decay:}
\[
    \eta_t = \eta_0 \cdot \gamma^{\lfloor t / k \rfloor}
\]
Simple, interpretable. Abrupt drops - can be suboptimal.

\vskip 1em

\structure{Cosine annealing:}
\[
    \eta_t = \eta_{\min} + \tfrac{1}{2}(\eta_0 - \eta_{\min})\Bigl(1 + \cos\tfrac{\pi t}{T}\Bigr)
\]
Smooth - widely used.

\vskip 1em

\structure{Linear warmup + cosine decay:}
\begin{itemize}
    \item Ramp $\eta$ linearly for first $t_w$ steps
    \item Then cosine decay to $\eta_{\min}$
    \item Standard for Transformers and large-scale training
\end{itemize}

\end{column}

\end{columns}
