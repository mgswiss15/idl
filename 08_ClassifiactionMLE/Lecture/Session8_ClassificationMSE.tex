\section{Classification Problem}
% -

\begin{frame}{Classification Problem}

\vskip 2em

\structure{Classification:} predict discrete category $y \in \{1, \ldots, K\}$
\begin{itemize}
    \item Output: image class, spam or not, medical diagnosis
    \item Output space fundamentally different from regression - no notion of distance between classes
\end{itemize}

\vskip 2em

\structure{Natural objective:} accuracy - fraction of correctly classified examples

\[
    \text{acc} = \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \mathbf{1}[\hat{y}^{(i)} = y^{(i)}]
\]

\end{frame}

\begin{frame}{From Accuracy to Loss}

\vskip 2em

\structure{Problem:} accuracy not differentiable - cannot use as training loss

\[
    \mathbf{1}[\hat{y} = y] = \begin{cases} 1 & \hat{y} = y \\ 0 & \hat{y} \neq y \end{cases}
    \quad \Rightarrow \quad \frac{\partial}{\partial \pvec}\mathbf{1}[\hat{y} = y] = 0 \text{ almost everywhere}
\]

\vskip 1em

\structure{Solution:} differentiable proxy loss $\to$ \alert{optimise loss in training, evaluate accuracy at test}

\vskip 1em

\structure{Requirements} for proxy:
\begin{itemize}
    \item Differentiable - gradient descent can optimise
    \item Correlates with accuracy - improving loss should improve accuracy
    \item Computatble from NN outputs
\end{itemize}

\vskip 1em

\structure{Problem:} raw network output $f_{\pvec}(\vx) \in \mathbb{R}$ cannot be interpreted as class directly

\vskip 1em

\begin{center}
\textbf{\alert{Idea:} convert output to probability distribution over classes}
\end{center}

\end{frame}


\begin{frame}{What We Need}

\vskip 2em

\structure{Requirements} for valid probability distribution over $K$ classes:

\[
    p_k \geq 0 \quad \forall\, k \qquad \text{and} \qquad \sum_{k=1}^{K} p_k = 1
\]

\vskip 1em

\structure{Raw network outputs} (logits) \alert{$\vz = f_{\pvec}(\vx) \in \mathbb{R}^K$}

\vskip 1em

\structure{Solution:} transformation that maps logits to valid probability distribution
\begin{itemize}
    \item \structure{Binary classification} ($K = 2$): \textbf{sigmoid} $\to$ next section
    \item \structure{Multiclass classification} ($K > 2$): \textbf{softmax} $\to$ following section
\end{itemize}

\vskip 1em

\structure{Loss function:} for probability outputs? not MSE, hm ... 

\end{frame}

\begin{frame}{}

{\Large \structure{Binary Classification}}

\begin{frame}{Sigmoid}

\vskip 2em

\structure{Binary classification:} $y \in \{0, 1\}$ - network outputs single logit $z = f_{\pvec}(\vx) \in \mathbb{R}$

\vskip 1em

\structure{Sigmoid:} squashes logit to $(0, 1)$ - interpreted as probability of class 1

\[
    \hat{p} = \sigma(z) = \frac{1}{1 + e^{-z}}
\]

\vskip 1em

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\begin{itemize}
    \item $\sigma(z) \in (0, 1)$ for all $z \in \mathbb{R}$
    \item $\sigma(0) = 0.5$ - symmetric around origin
    \item $\sigma(z) \to 1$ as $z \to \infty$
    \item $\sigma(z) \to 0$ as $z \to -\infty$
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\centering
\input{\CommonPath/tikz/sigmoid.tex}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Binary Cross-Entropy}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.55\textwidth}
\structure{Loss:} penalise confident wrong predictions
\[
    \ell_{BCE}(\hat{p}, y) = -y \log \hat{p} - (1-y) \log(1 - \hat{p})
\]
  \begin{columns}[c] 
  \begin{column}{0.5\textwidth}
  \structure{$y = 1$:} loss $= -\log \hat{p}$
  \begin{itemize}
      \item $\hat{p} \to 1$ : loss $\to 0$ \checkmark
      \item $\hat{p} \to 0$ : loss $\to \infty$ \alert{\texttimes}
  \end{itemize}
  \end{column}
  \begin{column}{0.5\textwidth}
  \structure{$y = 0$:} loss $= -\log(1-\hat{p})$
  \begin{itemize}
      \item $\hat{p} \to 0$ : loss $\to 0$ \checkmark
      \item $\hat{p} \to 1$ : loss $\to \infty$ \alert{\texttimes}
  \end{itemize}
  \end{column}
\end{columns}
\end{column}
\begin{column}{0.38\textwidth}

\,

\centering
\input{\CommonPath/tikz/bce_loss.tex}
\end{column}
\end{columns}

\vskip 1.5em

\structure{Training loss:}
\[
    \loss_{BCE}(\pvec) = -\frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \left[ y^{(i)} \log \hat{p}^{(i)} + (1 - y^{(i)}) \log(1 - \hat{p}^{(i)}) \right]
\]

\end{frame}

\section{Multiclass Classification}
% -
\begin{frame}{}
\vskip 8em
\begin{center}
{\Large \structure{Multiclass Classification}}
\end{center}
\end{frame}

\begin{frame}{Softmax}

\vskip 2em

\structure{Multiclass classification:} $y \in \{1, \ldots, K\}$ - network outputs $K$ logits $\vz \in \mathbb{R}^K$

\vskip 1em

\structure{Softmax:} maps logits to valid probability distribution
\begin{columns}[c]
\begin{column}{0.4\textwidth}
\[
    \hat{p}_k = \frac{e^{z_k}}{\sum_{j=1}^{K} e^{z_j}}, \qquad k = 1, \ldots, K
\]
\end{column}
\begin{column}{0.6\textwidth}
\begin{itemize}
    \item $\hat{p}_k > 0$ for all $k$ - all probabilities positive
    \item $\sum_{k=1}^{K} \hat{p}_k = 1$ - valid distribution
    \item Larger logit $\Rightarrow$ larger probability
\end{itemize}
\end{column}
\end{columns}

\vskip 2em

\structure{Connection to sigmoid:} softmax with $K=2$ reduces to sigmoid

\[
    \hat{p}_1 = \frac{e^{z_1}}{e^{z_1} + e^{z_2}} = \frac{1}{1 + e^{-(z_1 - z_2)}} = \sigma(z_1 - z_2)
\]

\vskip 0.5em

Single logit $z = z_1 - z_2$ - one degree of freedom is redundant

\end{frame}

\begin{frame}{Categorical Cross-Entropy}

\vskip 2em

\structure{Labels:} one-hot vector $\vy \in \{0,1\}^K$ - $y_k = 1$ for correct class, 0 otherwise

\vskip 1em

\structure{Loss:} picks out log probability of correct class

\[
    \ell_{CE}(\hat{\vp}, \vy) = -\sum_{k=1}^{K} y_k \log \hat{p}_k = -\log \hat{p}_{y}
\]

\vskip 1em

\begin{itemize}
    \item Correct class predicted confidently $\hat{p}_y \to 1$ : loss $\to 0$ \checkmark
    \item Correct class predicted with low confidence $\hat{p}_y \to 0$ : loss $\to \infty$ \alert{\texttimes}
\end{itemize}

\vskip 1em

\structure{Training loss:}
\[
    \loss_{CE}(\pvec) = -\frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \log \hat{p}_{y^{(i)}}^{(i)}
\]

\end{frame}

\section{Maximum Likelihood}
% -
\begin{frame}{}
\vskip 8em
\begin{center}
{\Large \structure{Maximum Likelihood}}
\end{center}
\end{frame}

\begin{frame}{Maximum Likelihood Principle}

\vskip 1em

\structure{Setup:} model defines conditional distribution $p(y \mid \vx;\, \pvec)$ over outputs

\vskip 1em

\structure{Likelihood:} probability of observed data as a \textbf{function of parameters} - data $\dataset$ is fixed

\vskip 0.5em

\structure{Assumption:} examples i.i.d.\ - joint likelihood factorises as product over examples

\[
    L(\pvec) = p(\dataset;\, \pvec) = \prod_{i=1}^{\nsamples} p(y^{(i)} \mid \vx^{(i)};\, \pvec)
\]

\structure{MLE:} find parameters that make observed data most probable

\[
    \pvec^* = \arg\max_{\pvec}\; L(\pvec)
\]

\vskip 0.5em

\structure{Log-likelihood:} sum instead of product - same maximiser, more convenient

\[
    \log L(\pvec) = \sum_{i=1}^{\nsamples} \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)
\]

\end{frame}

\begin{frame}{From MLE to Loss Minimisation}

\vskip 2em

\structure{Maximise} log-likelihood $\equiv$ \structure{minimise} negative log-likelihood (NLL)

\[
    \pvec^* = \arg\max_{\pvec} \log L(\pvec) = \arg\max_{\pvec} \sum_{i=1}^N \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)
    = \arg\min_{\pvec} \underbrace{-\frac{1}{\nsamples}\sum_{i=1}^N \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)}_{\loss(\pvec)}
\]

\vskip 0.5em

\structure{Key insight:} choice of $p(y \mid \vx;\, \pvec)$ determines the loss function

\vskip 1em

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Distributional assumption} & \textbf{Loss function} \\
\midrule
Gaussian & MSE \\
Bernoulli & Binary cross-entropy \\
Categorical & Categorical cross-entropy \\
\bottomrule
\end{tabular}
\end{center}

\end{frame}

\begin{frame}{Gaussian Assumption $\to$ MSE}

\[ \text{\structure{Assume: }} \quad p(y \mid \vx;\, \pvec) = \mathcal{N}(y;\, f_{\pvec}(\vx),\, \sigma^2) = \frac{1}{\sqrt{2\pi\sigma^2}} \exp\left(-\frac{(y - f_{\pvec}(\vx))^2}{2\sigma^2}\right) \]

\vskip 1em

\[
  \text{\structure{Log-likelihood: }} \quad \log p(y \mid \vx;\, \pvec) = -\frac{(y - f_{\pvec}(\vx))^2}{2\sigma^2} - \frac{1}{2}\log(2\pi\sigma^2)
\]

\vskip 1em

\[
\text{\structure{NLL: }} \quad
    -\frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)
    = \frac{1}{2\sigma^2} \underbrace{\frac{1}{\nsamples}\sum_{i=1}^{\nsamples}(y^{(i)} - f_{\pvec}(\vx^{(i)}))^2}_{\text{MSE}} + \text{const}
\]

\vskip 0.5em

Minimising NLL under Gaussian assumption $\equiv$ minimising \alert{MSE} - \structure{MSE not arbitrary!}

\centering
\vskip 0.5em
\colorbox{thwsorange!10}{$\displaystyle\arg\min_{\pvec} NLL_{\mathcal{N}}(\pvec) = \arg \min_{\pvec} MSE(\pvec)$}

\end{frame}

\begin{frame}{Bernoulli Assumption $\to$ Binary Cross-Entropy}

\[
    \text{\structure{Assume: }} \quad p(y \mid \vx;\, \pvec) = \text{Bernoulli}(y;\, \hat{p}) = \hat{p}^{\,y}(1-\hat{p})^{1-y}, \quad \hat{p} = \sigma(f_{\pvec}(\vx))
\]

\vskip 1em

\[
    \text{\structure{Log-likelihood: }} \quad \log p(y \mid \vx;\, \pvec) = y \log \hat{p} + (1-y)\log(1-\hat{p})
\]

\vskip 1em

\[
    \text{\structure{NLL: }} \quad
    -\frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)
    = \underbrace{-\frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \left[y^{(i)} \log \hat{p}^{(i)} + (1-y^{(i)})\log(1-\hat{p}^{(i)})\right]}_{\text{binary cross-entropy } \loss_{BCE}(\pvec)}
\]

\vskip 1em

Minimising NLL under Bernoulli assumption $\equiv$ minimising \alert{binary cross-entropy}

\centering
\vskip 0.5em
\colorbox{thwsorange!10}{$\displaystyle\arg\min_{\pvec} NLL_{Be}(\pvec) = \arg \min_{\pvec} \loss_{BCE}(\pvec)$}


\end{frame}

\begin{frame}{Categorical Assumption $\to$ Categorical Cross-Entropy}

\[
    \text{\structure{Assume: }} \quad p(y \mid \vx;\, \pvec) = \text{Categorical}(y;\, \hat{\vp}) = \prod_{k=1}^{K} \hat{p}_k^{\,y_k} = \hat{p}_y, \quad \hat{\vp} = \text{softmax}(f_{\pvec}(\vx))
\]

\vskip 1em

\[
    \text{\structure{Log-likelihood: }} \quad \log p(y \mid \vx;\, \pvec) = \sum_{k=1}^{K} y_k \log \hat{p}_k = \log \hat{p}_y
\]

\vskip 1em

\[
    \text{\structure{NLL: }} \quad
    -\frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \log p(y^{(i)} \mid \vx^{(i)};\, \pvec)
    = \underbrace{-\frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \log \hat{p}^{(i)}_{y^{(i)}}}_{\loss_{CE}(\pvec)}
\]

\vskip 1em

Minimising NLL under categorical assumption $\equiv$ minimising \alert{categorical cross-entropy}

\centering
\vskip 0.5em
\colorbox{thwsorange!10}{$\displaystyle\arg\min_{\pvec} NLL_{Cat}(\pvec) = \arg \min_{\pvec} \loss_{CE}(\pvec)$}


\end{frame}

\section{Information Theory}
% -
\begin{frame}{}
\vskip 8em
\begin{center}
{\Large \structure{Information Theory}}
\end{center}
\end{frame}

\begin{frame}{Why Information Theory?}

\vskip 1em

\structure{So far:} cross-entropy - mathematical result of MLE derivation

\vskip 1em

\structure{Open question:} what does cross-entropy \textit{mean}?

\vskip 1.5em

\structure{Answer - information theory:}
\begin{itemize}
    \item Cross-entropy: fundamental measure of how well distribution $q$ approximates distribution $p$
    \item Training minimises gap between model distribution $q = \hat{\vp}$ and true distribution $p$
\end{itemize}

\vskip 1.5em

\structure{We will build up:}
\begin{itemize}
    \item \structure{Information content} - how surprising is an event?
    \item \structure{Entropy} - how uncertain is a distribution?
    \item \structure{Cross-entropy and KL divergence} - how far apart are two distributions?
\end{itemize}

\vskip 1em

\structure{Payoff:} MLE, cross-entropy loss, and KL divergence - three views of same objective

\end{frame}

\begin{frame}{Information Content}

\vskip 1.5em

\structure{Intuition:} how surprised are we when event $k$ occurs?
\begin{itemize}
    \item Rare event - very surprising - highly informative
    \item Common event - expected - low information
\end{itemize}

\vskip 1em

\structure{Information content:} \structure{$I(k) = -\log_2 p_k$}  - optimal number of bits to encode event $k$ 

\begin{center}
\begin{tabular}{lll}
\toprule
$p_k$ & $I(k)$ & Interpretation \\
\midrule
$1$ & $0$ & certain event - no bits needed \\
$0.5$ & $1$ bit & coin flip \\
$0.01$ & $\approx 6.6$ bits & rare event - very surprising - long encoding\\
$\to 0$ & $\to \infty$ & impossible event becoming true - impossible to encode\\
\bottomrule
\end{tabular}
\end{center}

\vskip 0.5em

\structure{Why $-\log$?} More probable events - shorter codes; info of independent events - sum of info; monotonic and smooth (Shannon, 1948).

\end{frame}

\begin{frame}{Entropy}

\vskip 2em

\structure{Entropy:} expected information content = minimum average code length for samples from distribution $p$

\[
    H(p) = \mathbb{E}_{k \sim p}[I(k)] = -\sum_{k} p_k \log p_k
\]

\vskip 1em

\structure{Measures uncertainty:}
\begin{itemize}
    \item High entropy - high uncertainty, long average code - distribution spread out
    \item Low entropy - low uncertainty, short average code - distribution concentrated
    \item $H(p) = 0$ - deterministic distribution, no bits needed
    \item $H(p)$ maximised by uniform distribution - maximum uncertainty
\end{itemize}



\end{frame}

\begin{frame}{Binary Entropy and Coding}

\vskip 2em

\structure{Binary distribution:} $p = (p, 1-p)$

\[
    H(p) = -p\log_2 p - (1-p)\log_2(1-p)
\]

\vskip 1em

\begin{columns}[c]
\begin{column}{0.55\textwidth}
\begin{itemize}
    \item \structure{Fair coin} $p = 0.5$: $H = 1$ bit - optimal code 1 bit per outcome; matches practice
    \item \structure{Biased coin} $p = 0.9$: $H \approx 0.47$ bits - in theory we can do better than 1 bit per symbol
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\centering
\input{\CommonPath/tikz/binary_entropy.tex}
\end{column}
\end{columns}

\structure{Practical limit:} single outcomes cannot be encoded in less than 1 bit. Bound $H(p) < 1$ only achievable asymptotically by encoding longer sequences together

\end{frame}

\begin{frame}{Cross-Entropy}

\vskip 1em

\structure{Setup:} true distribution $p$, model distribution $\hat{\vp}$

\vskip 1em

\structure{Cross-entropy:} average code length when using $\hat{\vp}$ to encode events from $p$

\[
    H(p, \hat{\vp}) = -\sum_{k} p_k \log \hat{p}_k
\]

\begin{itemize}
    \item $H(p, \hat{\vp}) \geq H(p)$ - using wrong distribution always costs extra bits
    \item $H(p, \hat{\vp}) = H(p)$ iff $p = \hat{\vp}$ - no extra cost when model is perfect
\end{itemize}

\vskip 1em

\structure{Connection to classification loss:}
\begin{itemize}
    \item True labels $\Rightarrow$ true distribution $p\sim$ one-hot $\vy \in \{0,1\}^K$
    \item Model predictions $\Rightarrow$ estimated distribution $\hat{\vp}_{\pvec} = \text{softmax}(f_{\pvec}(\vx))$
\end{itemize}

\centering
\vskip 0.5em
\colorbox{thwsorange!10}{$\displaystyle\loss_{CE}(\pvec) = NLL_{CE}(\pvec) = H(p, \hat{\vp}_{\pvec})$}

\end{frame}

\begin{frame}{KL Divergence}

\vskip 1em

\structure{KL divergence:} extra cost of using $\hat{\vp}$ instead of $p$

\[
    D_{KL}(p \| \hat{\vp}) = H(p, \hat{\vp}) - H(p) = \sum_{k} p_k \log \frac{p_k}{\hat{p}_k} = \underbrace{\sum_{k} p_k \log p_k}_{- H(p)} - \underbrace{\sum_{k} p_k \log \hat{p}_k}_{H(p, \hat{\vp})}
\]

\begin{itemize}
    \item $D_{KL}(p \| \hat{\vp}) \geq 0$ - always non-negative
    \item $D_{KL}(p \| \hat{\vp}) = 0$ iff $p = \hat{\vp}$
    \item Not symmetric: $D_{KL}(p \| \hat{\vp}) \neq D_{KL}(\hat{\vp} \| p)$ in general
\end{itemize}

\vskip 1em

\structure{Connection to cross-entropy:} $H(p)$ is fixed - labels do not depend on $\pvec$

\centering
\vskip 0.5em

\colorbox{thwsorange!10}{$\displaystyle\arg\min_{\pvec}\; D_{KL}(p \| \hat{\vp}) \equiv \arg\min_{\pvec}\; H(p, \hat{\vp})$}

\vskip 0.5em

Min KL divergence btw true and predicted distribution $\equiv$ minimising cross-entropy

\vskip 0.5em

(Holds equivalently for any other distribution family.)

\end{frame}

\begin{frame}{Three Views, One Objective}

\vskip 2em

\structure{Three perspectives - same optimisation problem:}

\vskip 1.5em

\begin{center}
\begin{tabular}{llll}
\toprule
\textbf{Perspective} & \textbf{Gaussian} & \textbf{Bernoulli} & \textbf{Categorical} \\
\midrule
\structure{Loss} & MSE & binary CE & categorical CE \\[0.5em]
\structure{MLE} & maximise $\log \mathcal{N}$ & maximise $\log \text{Ber}$ & maximise $\log \text{Cat}$ \\[0.5em]
\structure{Info theory} & $\min D_{KL}$ & $\min D_{KL}$ & $\min D_{KL}$ \\
\bottomrule
\end{tabular}
\end{center}

\vskip 2em

\alert{\large Choice of loss not arbitrary $\sim$ assumption about data-generating process!}

\end{frame}

\begin{frame}{}
\vskip 8em
\begin{center}
{\Large \structure{Practical Implementation}}
\end{center}
\end{frame}

\begin{frame}{PyTorch Loss Functions}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.5\textwidth}
\structure{Binary classification:}
\begin{itemize}
    \item \alert{\texttt{nn.BCEWithLogitsLoss}} - takes logits
    \item \texttt{nn.BCELoss} - takes probabilities
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\structure{Multiclass classification:}
\begin{itemize}
    \item \alert{\texttt{nn.CrossEntropyLoss}} - takes logits
    \item \texttt{nn.NLLLoss} - takes log-probabilities
\end{itemize}
\end{column}
\end{columns}

\vskip 1em

\structure{Why logits?}
\begin{itemize}
    \item Numerically stable - PyTorch applies log-sum-exp trick internally
    \item Fewer operations - no need to apply sigmoid/softmax before loss
    \item Less risk of errors - sigmoid then BCELoss is a common bug source
\end{itemize}

\vskip 1em

\structure{Note:} \texttt{nn.CrossEntropyLoss} expects:
\begin{itemize}
    \item Integer class indices of shape \texttt{(N,)} - standard usage
    \item Float probability distributions of shape \texttt{(N, K)} - for soft targets (see later)
\end{itemize}
One-hot integer vectors are \alert{not supported}

\end{frame}

\begin{frame}{Numerical Stability}

\vskip 1em

\structure{Cross-entropy loss requires $\log \hat{p}_k$:}

\[
    \loss = -\frac{1}{\nsamples}\sum_i \log \hat{p}_{y^{(i)}}^{(i)}
    \qquad \Rightarrow \qquad
    \log \hat{p}_k = \log \frac{e^{z_k}}{\sum_j e^{z_j}} = z_k - \log \sum_j e^{z_j}
\]

\structure{Problem:} \texttt{log-softmax} numerically unstable

\vskip 0.5em

Large logits $\Rightarrow$ $e^{z_k}$ overflows to \texttt{inf} \quad Small logits $\Rightarrow$ $\log \sum_j e^{z_j}$ underflows to \texttt{-inf}

\vskip 1em

\structure{Log-sum-exp trick:} subtract maximum logit before exponentiating

\[
    \log \sum_j e^{z_j} = c + \log \sum_j e^{z_j - c}, \qquad c = \max_j z_j
\]

\begin{itemize}
    \item Largest term is always $e^0 = 1$ - no overflow
    \item Other terms $\leq 1$ - no underflow
\end{itemize}

\vskip 0.5em

\structure{\texttt{nn.CrossEntropyLoss} and \texttt{nn.BCEWithLogitsLoss}} apply this internally!
\end{frame}

\begin{frame}{Predictions and Accuracy}

\vskip 2em

\structure{From logits to predicted class:}

\begin{itemize}
    \item Softmax is monotone - highest logit = highest probability class
    \item No need to apply softmax / sigmoid
    \item \structure{Multiclass:} $\hat{y} = \arg\max_k z_k$
    \item \structure{Binary:} decision boundary at logit $= 0$ - predict class 1 if $z > 0$
\end{itemize}

\vskip 1.5em

\structure{Accuracy:} fraction of correctly classified examples

\[
    \text{acc} = \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \mathbf{1}[\hat{y}^{(i)} = y^{(i)}]
\]



\end{frame}

\begin{frame}{Overconfidence}

\vskip 2em

\structure{Problem:} cross-entropy with hard labels always rewards higher confidence

\[
    \ell_{CE}(\hat{\vp}, \vy) = -\sum_{k=1}^{K} y_k \log \hat{p}_k = -\log \hat{p}_{y}
    \qquad \text{minimised as } \hat{p}_{y} \to 1
\]

\begin{itemize}
    \item Rewarded for confidence - never penalised for overconfidence
    \item Learns to saturate softmax - assigns high probability even when uncertain
    \item Out-of-distribution input: may assign 99\% confidence to wrong class
\end{itemize}

\vskip 1.5em

\structure{Information theory view:} entropy $H(\hat{\vp})$ too low - model is too certain

\vskip 1em

\alert{\textbf{Predicted probabilities not indicative for true prediction (un)certainty!}}

\end{frame}

\begin{frame}{Label Smoothing}

\vskip 2em

\structure{Fix:} replace hard one-hot targets $\vy$ with soft targets $\vy^{\text{smooth}}$

\[
    \ell_{CE}(\hat{\vp}, \vy^{\text{smooth}}) = -\sum_{k=1}^{K} y_k^{\text{smooth}} \log \hat{p}_k, \quad \text{ where } y_k^{\text{smooth}} = \begin{cases} 1 - \varepsilon & k = y \\ \varepsilon / (K-1) & k \neq y \end{cases}
\]

\begin{itemize}
    \item Model can never be 100\% confident - softmax cannot saturate
    \item Increases entropy of predictions - more calibrated uncertainty
    \item Typical value: $\varepsilon = 0.1$
\end{itemize}

\vskip 2em

\structure{Standard in modern architectures:} Inception, Vision Transformers, modern ResNets - consistently effective with small $\varepsilon$

\end{frame}

\begin{frame}{Calibration}

\vskip 2em

\structure{Well-calibrated model:} predicted probability $=$ empirical accuracy

\begin{center}
predicted confidence 80\% $\Rightarrow$ correct 80\% of the time
\end{center}

\vskip 1em

\begin{columns}[c]

  \item get actual accuracy of samples in bin

\begin{frame}{Class Imbalance}

\vskip 2em

\structure{Problem:} training set has far more examples of some classes than others

\begin{itemize}
    \item Cross-entropy dominated by frequent classes
    \item Model learns to ignore rare classes - predicting majority class is cheap
    \item Common in practice: medical diagnosis, fraud detection, rare events
\end{itemize}

\vskip 1.5em

\structure{Fixes:}
\begin{itemize}
    \item \structure{Weighted loss} - upweight rare classes: $\loss = -\frac{1}{\nsamples}\sum_i w_{y^{(i)}} \log \hat{p}_{y^{(i)}}^{(i)}$
    \item \structure{Oversampling} - duplicate rare class examples
    \item \structure{Undersampling} - remove frequent class examples
\end{itemize}

\vskip 1em

\alert{Always check class distribution before training}

\end{frame}

\begin{frame}{Weak Labels}

\vskip 2em

\structure{Labels are often imperfect in practice:}

\vskip 1em

\begin{itemize}
    \item \structure{Noisy labels} - annotators make mistakes; loss computed against wrong targets
    \item \structure{Partial labels} - only some examples labelled; missing labels treated as negatives
    \item \structure{Ambiguous labels} - multiple valid answers; hard labels force arbitrary choice
\end{itemize}

\vskip 1.5em

\alert{\textbf{Consequence: model optimises toward label, not the truth!}}

\vskip 1em

\structure{Mitigations:}
\begin{itemize}
    \item Label cleaning - detect and correct noisy labels
    \item Label smoothing - reduces impact of individual wrong labels
    \item Multiple annotators - aggregate labels to reduce noise
\end{itemize}

\vskip 1em

\alert{Label quality matters as much as model quality}

\end{frame}
