\begin{frame}{Supervised Learning}

\vskip 2em

\structure{Setup:} data pairs $\dataset = \{(\vx^{(i)}, y^{(i)})\}_{i=1}^{\nsamples}$ with $\vx \in \inspace$, $y \in \outspace$

\vskip 0.8em

\structure{Assumption:} exists true relationship $f^*: \inspace \to \outspace, \ y = f^*(\vx)$, \alert{but $f^*$ is never observed!}

\vskip 0.8em

\structure{Goal:} learn $f_ \theta: \inspace \to \outspace$ from $\dataset$ such that $f_ \theta \approx f^*$
\alert{$\Rightarrow$ predict on \textbf{unseen} inputs $\vx \notin \dataset$}

\vskip 2em

\centering
\input{\CommonPath/tikz/supervised_learning.tex}

\vskip 0.8em

\structure{Notation:} $f_ \theta(\vx) = f(\vx; \theta) = \yhat$

\section{Rectified Linear Unit}

\begin{frame}{Activation function: Rectified Linear Unit (ReLU)}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\structure{Rectified Linear Unit}
\[ \sigma(x) = \text{ReLU}(x) = \max(0, x) = \begin{cases} x & x > 0 \\ 0 & x \leq 0 \end{cases} \]
\end{column}
\begin{column}{0.5\textwidth}
\input{\CommonPath/tikz/relu.tex}
\end{column}
\end{columns}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\structure{ReLU over linear:} $\yhat = f_ \theta(x) = \sigma(\theta_1 x + \theta_0)$
\vskip 0.8em
\structure{Effect:} one bend at $x = -\theta_0 / \theta_1$
\vskip 0.8em
\structure{Limitation:} only one breakpoint - not expressive enough
\vskip 0.8em
\structure{Solution:} combine multiple ReLUs
\end{column}
\begin{column}{0.5\textwidth}
\input{\CommonPath/tikz/regression_1d_relu_simple.tex}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Combining Multiple ReLUs}
\vskip 0.5em
\begin{columns}[c]
\begin{column}{0.6\textwidth}

\structure{Combine:} weighted sum of ReLU functions
\[ \yhat = f_\theta(x) = \theta_0 + \theta_1 h_1 + \theta_2 h_2 \]
\[h_1 = \sigma(\theta_{11} \, x + \theta_{10}) \quad h_2 = \sigma(\theta_{21} \, x + \theta_{20})\]

\vskip 0.8em
\structure{Effect:} one bend per ReLU term
\[ \yhat = f_\theta(x) = \theta_0 + \theta_1 h_1 + \theta_2 h_2 + \theta_3 h_3\]
\[h_3 = \sigma(\theta_{31} \, x + \theta_{30})\]
\vskip 0.8em
\structure{Key result:} more breakpoints $\Rightarrow$ more expressive $f_\theta$
\[ \yhat = f_\theta(x) = \theta_0 + \sum_j \theta_j \, \sigma\!\left(\theta_{j1} \, x + \theta_{j0}\right) \]

\end{column}
\begin{column}{0.4\textwidth}
\input{\CommonPath/tikz/relu_combi_simple_plot.tex}
\input{\CommonPath/tikz/relu_combi_double_plot.tex}
\end{column}

\end{column}

\section{Universal Approximation}

\begin{frame}{Universal Approximation}

\vskip 1em

\begin{alertblock}{Universal Approximation Theorem}
For any continuous function $g: \inspace \to \R$ and any $\varepsilon > 0$, there exists a 1-hidden-layer neural network $f_\theta$ such that:
\[ \sup_{\vx \in \inspace} |f_\theta(\vx) - g(\vx)| < \varepsilon \]
\end{alertblock}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{What it tells us}
\begin{itemize}
\item MLPs are universal approximators
\item Enough neurons can fit any continuous function
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{What it doesn't tell us}
\begin{itemize}
\item How many neurons are needed
\item How to find the parameters $\pvec$
\item Whether the model will generalize
\end{itemize}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{Binary Classification}

\vskip 0.5em

\begin{columns}[c]
    \begin{column}{0.5\textwidth}
        \structure{Linear classifier:}
        \[ \yhat = \mathds{1}\left[\theta_0 + \sum_i \theta_i x_i \geq 0\right] \]
        \structure{Decision boundary:}\\[0.3em]
Hyperplane $\theta_0 + \pvec\T\vx = 0$

        \vskip 0.8em

        \structure{Limitation:} only works for \textbf{linearly separable} problems

        \vskip 0.8em

        \structure{Solution:} replace linear with neural network
        \[ \yhat = \mathds{1}\left[f_\theta(\vx) \geq 0\right] \]
        \structure{Decision boundary:}\\[0.3em]
        Nonlinear surface $f_\theta(\vx) = 0$
    \end{column}
