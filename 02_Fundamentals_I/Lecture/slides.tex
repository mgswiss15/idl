\begin{frame}{Supervised Learning}

\vskip 2em

\structure{Setup:} data pairs $\dataset = \{(\vx^{(i)}, y^{(i)})\}_{i=1}^{\nsamples}$ with $\vx \in \inspace$, $y \in \outspace$

\vskip 0.8em

\structure{Assumption:} exists true relationship $f^*: \inspace \to \outspace, \ y = f^*(\vx)$, \alert{but $f^*$ is never observed!}

\vskip 0.8em

\structure{Goal:} learn $f_ \theta: \inspace \to \outspace$ from $\dataset$ such that $f_ \theta \approx f^*$
\alert{$\Rightarrow$ predict on \textbf{unseen} inputs $\vx \notin \dataset$}

\vskip 2em

\centering
\input{\CommonPath/tikz/supervised_learning.tex}

\vskip 0.8em

\structure{Notation:} $f_ \theta(\vx) = f(\vx;  \theta) = \yhat$

\end{frame}

\begin{frame}{This is a neural network}
\begin{columns}
    \begin{column}{0.55\textwidth}
        \begin{itemize}
            \item Each hidden neuron computes:
            \[ z_j^{(1)} = w_j x + b_j, \quad a_j^{(1)} = \text{ReLU}(z_j^{(1)}) \]
            \item Output layer combines:
            \[ f(x;  \theta) = \sum_j v_j a_j^{(1)} + c \]
            \vskip 0.5em
            \item<2-> In matrix form:
            \begin{align*}
                \vz^{(1)} &= \mW^{(1)}\vx + \vb^{(1)} \\
                \va^{(1)} &= \text{ReLU}(\vz^{(1)}) \\
                f(\vx;  \theta) &= \mW^{(2)}\va^{(1)} + \vb^{(2)}
            \end{align*}
            \item<2-> Parameters: $ \theta = \{\mW^{(1)}, \vb^{(1)}, \mW^{(2)}, \vb^{(2)}\}$
        \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
        \centering
        \texttt{[mlp\_1hidden.tex]}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Universal approximation}
    \begin{block}{Universal Approximation Theorem}
        Let $\act: \R \to \R$ be a non-constant, bounded, continuous activation function.
        Then for any continuous function $g: [0,1]^d \to \R$ and any $\varepsilon > 0$,
        there exists a 1-hidden-layer neural network $f(\cdot\,;  \theta)$ such that:
        \[ \sup_{\vx \in [0,1]^d} |f(\vx;  \theta) - g(\vx)| < \varepsilon \]
    \end{block}
    \vskip 1em
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \structure{What it tells us}
            \begin{itemize}
                \item MLPs are universal approximators
                \item Enough neurons can fit any continuous function
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \structure{What it doesn't tell us}
            \begin{itemize}
                \item How many neurons are needed
                \item How to find the parameters $ \theta$
                \item Whether the model will generalize
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Activation functions}
\begin{columns}
    \begin{column}{0.48\textwidth}
        \structure{What makes a good activation?}
        \begin{itemize}
            \item Non-linear — otherwise depth buys nothing
            \item Differentiable — needed for backpropagation
            \item Computationally cheap
        \end{itemize}
        \vskip 1em
        \structure{Common choices}
        \begin{itemize}
            \item \textbf{Sigmoid}: $\sigma(z) = \frac{1}{1+e^{-z}} \in (0,1)$
            \item \textbf{Tanh}: $\tanh(z) \in (-1, 1)$
            \item \textbf{ReLU}: $\max(0, z)$ — sparse, fast, but dying neurons
        \end{itemize}
        \vskip 0.5em
        \structure{Dying ReLU problem}
        \begin{itemize}
            \item Neurons with $z < 0$ output 0 and have zero gradient
            \item Can become permanently inactive during training
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \centering
        \texttt{[activation\_functions.tex]}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Notation summary}
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{cll}
    \hline
    \textbf{Symbol} & \textbf{Meaning} & \textbf{Dimensions} \\
    \hline
    $\vx$ & input & $\R^{d}$ \\
    $y$ & target output & $\R$ or $\{0,1\}$ \\
    $ \theta$ & all parameters & — \\
    $\mW^{(l)}$ & weight matrix, layer $l$ & $\R^{d_{l} \times d_{l-1}}$ \\
    $\vb^{(l)}$ & bias vector, layer $l$ & $\R^{d_{l}}$ \\
    $\vz^{(l)}$ & pre-activation, layer $l$ & $\R^{d_{l}}$ \\
    $\va^{(l)}$ & post-activation, layer $l$ & $\R^{d_{l}}$ \\
    $\act$ & activation function & $\R \to \R$ \\
    $L$ & number of hidden layers & — \\
    \hline
\end{tabular}
\end{center}
\vskip 0.5em
\begin{center}
    \structure{Forward pass:} \quad
    $\vz^{(l)} = \mW^{(l)}\va^{(l-1)} + \vb^{(l)}$ \quad
    $\va^{(l)} = \act(\vz^{(l)})$ \quad
    $\va^{(0)} = \vx$
\end{center}
