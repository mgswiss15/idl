\begin{frame}{Linear classifier}
\begin{columns}
    \begin{column}{0.55\textwidth}
        \begin{itemize}
            \item Line in 2D: 
            \begin{align*}
                x_2 & = m x_1 + c \\
                0 & = m x_1 + c - x_2
            \end{align*}
            \item<2-> Separating line: 
            \begin{align*}
                w_1 x_1 + w_2 x_2 + b & = 0\\
                \vw\T\vx + b & = 0
            \end{align*}
            \item<3-> Decision rule:
            \[ \hat{y} = \begin{cases} 1 & \vw\T\vx + b \geq 0 \\ 0 & \text{otherwise} \end{cases} \]
        \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
        \centering
        \input{\CommonPath/tikz/classification_2d_math.tex}

        \structure{Artificial neuron $f: \R^d \to \R$}
        \begin{itemize}

        \end{itemize}
