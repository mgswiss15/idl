\midrule

\vskip 1.5em

\structure{Modern workflow:} 
\begin{enumerate}
  \item download pretrained from torchvision / timm / HuggingFace 
  \item adapt to task
\end{enumerate}

\vskip 0.5em

Training from scratch rare - large organisations with compute and data budget

\end{frame}

% -------------------

\begin{frame}{Why Pretrained Models Work}

\vskip 2em

\structure{Key insight:} CNN layers learn features of increasing specificity

\vskip 2em

\begin{columns}[t]
\begin{column}{0.5\textwidth}
\structure{Early layers - general:}
\begin{itemize}
    \item Edges, colours, orientations
    \item Useful across almost all visual tasks
    \item Transfer well to new domains
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\structure{Later layers - specific:}
\begin{itemize}
    \item Object parts, textures, patterns
    \item Specific to source task
    \item Need adaptation for new task
\end{itemize}
\end{column}
\end{columns}

\vskip 2em

\structure{Pretrained:} ImageNet (1.2M images), LAION-5B (5.8B image-text pairs), proprietary

\vskip 2em

\structure{Works because:} low-level visual features universal - edges exist in all images

\end{frame}

% -------------------

\begin{frame}{Feature Extraction vs Fine-tuning}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.5\textwidth}
\structure{Feature extraction:}
\begin{itemize}
    \item Freeze all pretrained layers
    \item Replace and train FC head only
\end{itemize}

\vskip 0.5em
\alert{+} Fast, few trainable parameters\\
\alert{+} Works with small datasets\\
\alert{+} No risk of catastrophic forgetting\\
\alert{-} Limited adaptation to new domain
\end{column}
\begin{column}{0.5\textwidth}
\structure{Fine-tuning:}
\begin{itemize}
    \item Unfreeze some or all layers
    \item Train with small learning rate
\end{itemize}

\vskip 0.5em
\alert{+} Better performance\\
\alert{+} Adapts to new domain\\
\alert{-} Needs more data\\
\alert{-} Risk of catastrophic forgetting
\end{column}
\end{columns}

\vskip 2em

\structure{Catastrophic forgetting:} large learning rate overwrites pretrained weights - use $10\times$--$100\times$ smaller learning rate than original training

\end{frame}

% -------------------

\begin{frame}{When to Use Which}

\vskip 2em

\begin{center}
\begin{tabular}{lll}
\toprule
 & \textbf{Similar domain} & \textbf{Different domain} \\
\midrule
\structure{Small dataset} &
    Feature extraction &
    Feature extraction \\
 & (freeze all, train head) & (try different layer depths) \\[0.8em]
\structure{Large dataset} &
    Fine-tune all layers &
    Fine-tune all layers \\
 & (small learning rate) & (more epochs, careful tuning) \\
\bottomrule
\end{tabular}
\end{center}

\vskip 1.5em

\structure{Rule of thumb:}
\begin{itemize}
    \item Less data $\to$ freeze more layers
    \item More similar to source domain $\to$ reuse more layers
    \item Always start with feature extraction, add fine-tuning if needed
\end{itemize}

\end{frame}

% -------------------

\begin{frame}{Practical Tuning}

\vskip 1em

\structure{Freezing strategy:}
\begin{itemize}
    \item Start fully frozen - train head only
    \item Unfreeze progressively from top (later layers first)
    \item Earlier layers = more general = freeze longer
\end{itemize}

\vskip 1em

\structure{Learning rates:}
\begin{itemize}
    \item New head: standard learning rate (e.g. \texttt{1e-3})
    \item Fine-tuned layers: much smaller (e.g. \texttt{1e-5})
    \item \textbf{Discriminative learning rates} - different rates per layer group
    \item Warmup - start small, increase gradually $\to$ new head stabilization
\end{itemize}

\vskip 1em

\structure{Other considerations:}
\begin{itemize}
    \item More aggressive data augmentation when dataset is small
    \item Early stopping essential - fine-tuning overfits quickly
\end{itemize}

\end{frame}

\begin{frame}{Adapting Pretrained Model}

\vskip 1em

\structure{Replace classifier head:}

\vskip 0.5em

{\scriptsize\ttfamily
model = resnet50(weights=ResNet50\_Weights.DEFAULT)\\[0.5em]
\# freeze all layers\\
for param in model.parameters():\\
\hspace{1.5em}param.requires\_grad = False\\[0.5em]
\# replace FC head\\
num\_features = model.fc.in\_features\\
model.fc = nn.Linear(num\_features, num\_classes)\\[0.5em]
\# only head parameters are trainable\\
optimizer = torch.optim.Adam(model.fc.parameters(), lr=1e-3)\\
}

\vskip 1em

\structure{For fine-tuning:} unfreeze layers selectively

\vskip 0.5em

{\scriptsize\ttfamily
\# unfreeze last ResNet block\\
for param in model.layer4.parameters():\\
\hspace{1.5em}param.requires\_grad = True\\[0.5em]

\# fine-tune with discriminative learning rate\\
optimizer = torch.optim.Adam([\\
\hspace{1.5em}\{'params': model.layer4.parameters(), 'lr': 1e-5\},\\
\hspace{1.5em}\{'params': model.fc.parameters(),     'lr': 1e-3\},\\
])\\
}

\vspace{-5cm}\hspace{9.5cm}%
\begin{minipage}{0.3\textwidth}
\alert{\textbf{All basics hold:}}\\
{\scriptsize\ttfamily
model.train()\\
model.eval()\\
optimizer.zero\_grad()\\
}
\end{minipage}

\end{frame}

\begin{frame}{Object Detection}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.25\textwidth}
\includegraphics[width=\textwidth]{\CommonPath/Pics/coco2.png}
\end{column}
\begin{column}{0.75\textwidth}
\structure{Output:} bounding boxes + class labels + confidence scores per object instance

\vskip 0.5em

\structure{Annotation:} bounding box around every object instance
\end{column}
\end{columns}

\vskip 2em

\structure{Two main strategies:}
\begin{itemize}
    \item \structure{Two-stage} (\href{https://arxiv.org/abs/1506.01497}{Faster R-CNN}):  region proposal network $\to$ classify and refine - higher accuracy, slower
    \item \structure{One-stage} (\href{https://arxiv.org/abs/1506.02640}{YOLO}, \href{https://arxiv.org/abs/1512.02325}{SSD}): directly predict boxes and classes in single pass - faster, real-time capable
\end{itemize}

\vskip 1em

\structure{Evaluation:} IoU and mAP - see next slide

\end{frame}

\begin{frame}{Evaluation - IoU and mAP}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.4\textwidth}
\tikzsetnextfilename{iou}
\input{\CommonPath/tikz/iou.tex}
\end{column}
\begin{column}{0.6\textwidth}
\structure{IoU (Intersection over Union):}
\begin{itemize}
    \item Overlap between predicted and ground truth box
    \item Prediction correct if IoU $>$ threshold (e.g. 0.5)
    \item  $\text{IoU} = \dfrac{|\text{intersection}|}{|\text{union}|}  \in [0, 1]$
\end{itemize}

\end{column}
\end{columns}

\vskip 2em

\structure{Average Precision (AP):}
\begin{itemize}
    \item Vary confidence threshold - plot precision-recall curve
    \item AP $=$ area under that curve per class
    \item mAP $=$ mean AP across all classes
\end{itemize}

\end{frame}

\begin{frame}{Semantic Segmentation}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.25\textwidth}
\includegraphics[width=\textwidth]{\CommonPath/Pics/coco3.png}
\end{column}
\begin{column}{0.75\textwidth}
\structure{Output:} pixel-wise class label map - same spatial size as input

\vskip 0.5em

\structure{Annotation:} careful pixel-level boundary tracing
\end{column}
\end{columns}

\vskip 2em

\structure{Main strategy:} encoder-decoder architecture
\begin{itemize}
    \item \structure{Encoder:} CNN backbone reduces resolution, extracts features
    \item \structure{Decoder:} upsamples back to input resolution
    \item \structure{Skip connections:} recover spatial detail lost in encoding
\end{itemize}

\vskip 0.5em

Key methods: \href{https://arxiv.org/abs/1505.04597}{U-Net}, \href{https://arxiv.org/abs/1706.05587}{DeepLab v3}, \href{https://arxiv.org/abs/2105.15203}{SegFormer}

\vskip 1em

\structure{Evaluation:} mean Intersection over Union (mIoU) - mean IoU across all classes

\end{frame}

\begin{frame}{Instance Segmentation}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.25\textwidth}
\includegraphics[width=\textwidth]{\CommonPath/Pics/coco4.png}
\end{column}
\begin{column}{0.75\textwidth}
\structure{Output:} per-instance binary mask + class label - distinguishes individual objects of same class

\vskip 0.5em

\structure{Annotation:} per-instance pixel masks
\end{column}
\end{columns}

\vskip 2em

\structure{Main strategy:} detect then segment
\begin{itemize}
    \item \href{https://arxiv.org/abs/1703.06870}{Mask R-CNN}: extends Faster R-CNN with mask prediction head - binary mask per detected box
    \item \href{https://arxiv.org/abs/2304.02643}{SAM} (Segment Anything): foundation model, prompt-based - any object, zero-shot
\end{itemize}

\vskip 0.5em

\structure{Panoptic segmentation:} unified - semantic for background, instance for objects - modern standard

\vskip 1em

\structure{Evaluation:} mAP on masks rather than boxes

\end{frame}

\begin{itemize}

    \item Semi-supervised and weakly supervised methods - learn from cheap labels
    \item Synthetic data - generate annotated data automatically
\end{itemize}
