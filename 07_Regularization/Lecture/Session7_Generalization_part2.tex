\begin{frame}{Training Stability}

\vskip 2em

\structure{Regularisation:} can use large, overparameterised networks without overfitting :)

\vskip 1em

\structure{New problem:} deep NNs training instabilities - more layers amplify small issues :(

\vskip 1em

\begin{itemize}
    \item \structure{Internal covariate shift} - activation distributions shift as parameters update
    \item \structure{Vanishing/exploding grads} - signals decay or amplify exponentially through layers
    \item \structure{Dying ReLU} - neurons get stuck outputting zero permanently
\end{itemize}

\vskip 2em

\alert{Not overfitting problems - regularisation does not help here}

\end{frame}

\begin{frame}{Internal Covariate Shift}

\vskip 2em

\structure{Problem:} as weights update, the input distribution to each layer changes continuously

\vskip 1em

\begin{itemize}
    \item Layer $l$ learns to process inputs with a certain distribution
    \item Layer $l-1$ updates its weights - distribution shifts
    \item Layer $l$ must constantly re-adapt - a moving target
    \item Effect amplifies with depth - later layers more instability
\end{itemize}

\vskip 1em

\structure{Consequences:}
\begin{itemize}
    \item Slower convergence
    \item Need for lower learning rates
    \item High sensitivity to initialisation
\end{itemize}

\vskip 1em

\structure{Solution:} explicitly normalise activations at each layer - \structure{Batch Normalisation}

\end{frame}

\begin{frame}{Batch Normalisation}

\vskip 2em

\structure{Idea:} normalise activations within each mini-batch -> distributions stable across layers

\vskip 0.5em

\[
    \hat{h}_j = \frac{h_j - \mu_{\mathcal{B}}}{\sqrt{\sigma^2_{\mathcal{B}} + \epsilon}}
    \qquad
    \tilde{h}_j = \gamma \hat{h}_j + \beta
\]

$\mu_{\mathcal{B}}, \sigma^2_{\mathcal{B}}$ - mini-batch mean and variance \quad $\gamma, \beta$ - learnable scale and shift

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{\texttt{model.train():}}
\begin{itemize}
    \item Normalise using mini-batch stats
    \item Accumulate running mean and var
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{\texttt{model.eval():}}
\begin{itemize}
    \item Normalise using running stats
    \item Deterministic - no batch dependence
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1em

\structure{Benefits:} stable training, allows higher learning rates, less sensitive to initialisation

\vskip 0.5em

\structure{Side effect:} mini-batch noise introduces mild regularisation - but not main motivation

\end{frame}

\begin{frame}{Vanishing and Exploding Gradients}

\vskip 2em

\structure{Backprop multiplies Jacobians backwards} - effect compounds, earliest layers affected most

\[
    \frac{\partial \loss}{\partial \pvec^{(1)}} =
    \frac{\partial \loss}{\partial \vh^{(L)}}
    \cdot \frac{\partial \vh^{(L)}}{\partial \vh^{(L-1)}}
    \cdots
    \frac{\partial \vh^{(2)}}{\partial \vh^{(1)}}
    \cdot \frac{\partial \vh^{(1)}}{\partial \pvec^{(1)}}
\]


\vskip 2em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Vanishing} (factors $< 1$)
\begin{itemize}
    \item Gradient $\to 0$ at earliest layers
    \item Early layers stop learning
    \item Deep NNs effectively shallow
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Exploding} (factors $> 1$)
\begin{itemize}
    \item Gradient $\to \infty$ at earliest layers
    \item Single large step destroys weights
    \item Training diverges
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Solutions:} Xavier/He init (Session 5); batch norm; gradient clipping (RNN, Transformers)

\end{frame}

\begin{frame}{Dying ReLU}

\vskip 2em

\alert{\textbf{Alarm:}} ReLU has zero gradient for all negative pre-activations - \alert{\textbf{half the input space}}

\[
    \frac{\partial \, \text{ReLU}(z)}{\partial z} = \begin{cases} 1 & z > 0 \\ 0 & z \leq 0 \end{cases}
\]


\vskip 2em

\structure{Relief:} neuron dies if pre-activation negative for \alert{\textbf{all}} training examples \alert{\textbf{simultaneously}}
\begin{itemize}
    \item Pre-activations vary across examples - some positive, some negative
    \item Average gradient is nonzero - diversity in data saves us
    \item \textbf{ReLU remains solid baseline}
\end{itemize}

\end{frame}

\begin{frame}{Embracing Randomness}

\vskip 2em

\structure{Randomness pervasive in deep learning training - not a nuisance but a feature}

\vskip 1em

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Source} & \textbf{Where} & \textbf{Effect} \\
\midrule
Weight initialisation & start of training & which minimum is found \\
Data shuffling & each epoch & different batches, prevents order memorisation \\
Mini-batch SGD & every update step & implicit regularisation, flat minima \\
Dropout & forward pass & ensemble effect, prevents co-adaptation \\
Data augmentation & data loading & invariance, population coverage \\
\bottomrule
\end{tabular}
\end{center}

\vskip 1.5em

\begin{center}
Embracing randomness rather than eliminating it is part of what makes deep networks generalise well
\end{center}

\end{frame}

\begin{frame}{Summary}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Generalisation:}
\begin{itemize}
    \item True goal - low loss on population
    \item Generalisation gap - monitor via validation loss
    \item Deep NNs generalise surprisingly well - flat minima, double descent
\end{itemize}

\vskip 1em

\structure{Regularisation:}
\begin{itemize}
    \item Weight decay - explicit penalty
    \item Dropout - random subnetworks
    \item Early stopping - cut training time
    \item Data augmentation - most powerful
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Training stability:}
\begin{itemize}
    \item Batch norm - stable activation distribs
    \item Gradient clipping - stop exploding grads
    \item LeakyReLU / ELU - avoid dying ReLU
\end{itemize}

\vskip 1em

\structure{In PyTorch:}
\begin{itemize}
    \item \texttt{nn.Dropout(p)}
    \item \texttt{nn.BatchNorm1d}
    \item \texttt{weight\_decay} in \texttt{torch.optim}
    \item \texttt{model.train()} / \texttt{model.eval()}
\end{itemize}
\end{column}
\end{columns}
