\section{Generalization}
% -------------------------------------------------------

\begin{frame}{Where We Are}

\vskip 2em

\structure{So far we have:}
\begin{itemize}
    \item \alert{Network} $f_{\pvec} : \inspace \to \outspace$ - parameterized function
    \item \alert{Loss} $\loss(\pvec;\, \dataset)$ - measures fit on training data
    \item \alert{Optimization} - algorithms to minimize $\loss(\pvec;\, \dataset)$
\end{itemize}

\vskip 1.5em

\structure{Open question:}

\vskip 0.5em

\begin{center}
{\large Does minimizing training loss give us a \textbf{good model}?}
\end{center}

\vskip 1.5em

\structure{This lecture:} what good actually means, and how to get there

\end{frame}

% -------------------------------------------------------

\begin{frame}{True Goal}

\vskip 2em

\structure{What we optimize:}
\[
    \pvec^* = \arg\min_{\pvec}\; \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
\]

\vskip 0.5em

\structure{What we actually want:} low loss on \textbf{new, unseen examples from the population}
\[
    \loss_{\text{pop}}(\pvec) = \mathbb{E}_{(\vx, y) \sim p_{\text{data}}}\bigl[\ell(f_{\pvec}(\vx),\, y)\bigr]
\]

\vskip 0.5em

\structure{Generalization gap:}
\[
    \underbrace{\loss_{\text{pop}}(\pvec)}_{\text{true goal}} - \underbrace{\loss_{\text{train}}(\pvec)}_{\text{what we minimize}} \geq 0
\]

\vskip 0.5em

The \textbf{test set} is a finite sample we create to \textit{estimate} $\loss_{\text{pop}}$ --- it is a measurement tool, not the objective itself

\end{frame}

% -------------------------------------------------------

\begin{frame}{Train / Validation / Test Split}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Training set}\\[0.5em]
Fit model parameters
\[
    \min_{\pvec}\; \loss(\pvec;\, \dataset_{\text{train}})
\]
\end{column}
\begin{column}{0.32\textwidth}
\structure{Validation set}\\[0.5em]
Select model, tune hyperparameters
\[
    \loss(\pvec;\, \dataset_{\text{val}})
\]
Touched many times during development
\end{column}
\begin{column}{0.32\textwidth}
\structure{Test set}\\[0.5em]
Final honest evaluation
\[
    \loss(\pvec;\, \dataset_{\text{test}})
\]
\alert{Touched once, at the end}
\end{column}
\end{columns}

\vskip 2em

\structure{Common mistake:} tuning on the test set - reported performance is optimistic

\vskip 0.5em

\structure{Typical splits:} 80/10/10 or 70/15/15 - depends on dataset size

\end{frame}

% -------------------------------------------------------

\begin{frame}{Overfitting and Underfitting}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.55\textwidth}

\structure{Underfitting:} model too simple
\begin{itemize}
    \item High train loss, high test loss
    \item Small generalisation gap
\end{itemize}

\vskip 1em

\structure{Overfitting:} model memorises training data
\begin{itemize}
    \item Low train loss, high test loss
    \item \alert{Large generalisation gap}
\end{itemize}

\vskip 1em

\structure{Signal:} monitor the generalisation gap --- train loss alone tells us nothing about generalisation

\end{column}
\begin{column}{0.4\textwidth}
\centering
\input{\CommonPath/tikz/overfit_underfit.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{What To Do}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Underfitting} {\small (high train loss, small gap)}
\begin{itemize}
    \item Increase model capacity
    \item Train longer
    \item Tune learning rate and optimiser
    \item Reduce regularisation if applied
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Overfitting} {\small (low train loss, large gap)}
\begin{itemize}
    \item More data --- always most effective
    \item \alert{Regularisation} --- subject of this lecture
    \item Reduce model capacity
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\centering
\structure{This lecture:} deep networks have enormous capacity yet often generalise well --- we study why, and how to ensure it

\end{frame}

% -------------------------------------------------------
\section{The Generalization Paradox}
% -------------------------------------------------------

\begin{frame}{Bias-Variance Tradeoff}

\vskip 2em

\structure{Classical view:} test error decomposes into two competing terms

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Bias} --- error from wrong assumptions
\begin{itemize}
    \item Model too simple to capture the data
    \item High train loss, high test loss
    \item Underfitting
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Variance} --- sensitivity to training set
\begin{itemize}
    \item Model fits noise in training data
    \item Low train loss, high test loss
    \item Overfitting
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1.5em

\structure{Classical prescription:} choose model complexity to balance the two

\vskip 0.5em

\begin{center}
More parameters $\Rightarrow$ lower bias, higher variance $\Rightarrow$ \alert{overfitting}
\end{center}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Limits of Classical View}

\vskip 2em

\structure{Classical prediction:} more parameters than training examples $\Rightarrow$ catastrophic overfitting

\vskip 1em

\structure{Reality in deep learning:}
\begin{itemize}
    \item Deep networks have huge numbers of parameters --- far more than needed to memorise training data
    \item They routinely drive training loss close to zero
    \item Classical theory: a model that nearly memorises training data cannot generalise
    \item In practice: deep networks still generalise well
\end{itemize}

\vskip 1.5em

\begin{center}
{\large Why do deep networks generalise despite being massively overparameterised?}
\end{center}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Many Good Minima}

\vskip 2em

\structure{Classical concern:} gradient descent finds a local minimum --- but which one?

\vskip 1em

\structure{Empirical observation in deep networks:}
\begin{itemize}
    \item Overparameterised networks have a vast number of local minima
    \item Most local minima have very similar loss values
    \item Networks trained from different initialisations reach different minima --- yet generalise similarly
\end{itemize}

\vskip 1.5em

\structure{Why:} the network has far more parameters than needed to fit the data --- many equivalent solutions exist

\vskip 1em

\begin{center}
Which minimum we find matters much less than classical theory suggests
\end{center}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Flat vs Sharp Minima}

\vskip 2em

Not all minima generalise equally well

\vskip 1em

\begin{columns}[c]
\begin{column}{0.28\textwidth}
\structure{Sharp minimum:}
\begin{itemize}
    \item Loss changes rapidly
    \item Small distribution shift $\Rightarrow$ loss jumps
    \item Poor generalisation
\end{itemize}
\end{column}
\begin{column}{0.28\textwidth}
\structure{Flat minimum:}
\begin{itemize}
    \item Loss changes slowly
    \item Small distribution shift $\Rightarrow$ loss stays low
    \item Good generalisation
\end{itemize}
\end{column}
\begin{column}{0.38\textwidth}
\centering
\input{\CommonPath/tikz/flat_sharp_minima.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1.5em

\structure{SGD noise naturally favours flat minima} - implicit regularisation from mini-batching (Session 5)

\end{frame}

% -------------------------------------------------------

\begin{frame}{Double Descent}

\vskip 2em

\structure{Classical bias-variance curve:} test error has one optimal complexity --- U-shaped curve

\vskip 0.5em

\structure{Double descent:} beyond the interpolation threshold, test error decreases again

\vskip 1em

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\centering
\input{\CommonPath/tikz/double_descent.tex}
\end{column}
\begin{column}{0.46\textwidth}
\begin{itemize}
    \item \structure{Underfitting:} too little capacity
    \item \structure{Interpolation threshold:} maximally rigid --- worst generalisation
    \item \structure{Overparameterised:} smoother interpolation --- generalisation recovers
\end{itemize}

\vskip 1em

\alert{Deep networks operate in the overparameterised regime}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

\end{center}

\begin{frame}{Weight Decay}

\vskip 2em

\structure{Explicit regularisation:} add penalty term to the loss

\[
    \tilde{\loss}(\pvec) = \loss(\pvec) + \frac{\lambda}{2}\|\pvec\|^2
\]

Large weights $\Rightarrow$ model relies heavily on specific features $\Rightarrow$ high variance

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{L2 regularisation} (weight decay)
\begin{itemize}
    \item Penalises $\|\pvec\|^2$ --- shrinks all weights
    \item Smooth, differentiable
    \item \texttt{weight\_decay} in \texttt{torch.optim}
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{L1 regularisation}
\begin{itemize}
    \item Penalises $\|\pvec\|_1$ --- promotes sparsity
    \item Some weights go exactly to zero
    \item Useful for feature selection
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1em

\structure{Hyperparameter:} $\lambda > 0$ controls strength --- tuned on validation set

\vskip 0.5em

\structure{In deep learning:} useful baseline, freely combined with other techniques --- but less central than in classical ML

\end{frame}

% -------------------------------------------------------

\begin{frame}{Dropout}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.55\textwidth}

\structure{During training:} randomly zero each activation with probability $p$
\begin{itemize}
    \item Each forward pass uses a different subnetwork
    \item Neurons cannot rely on specific others always being present
    \item Prevents co-adaptation of neurons
\end{itemize}

\vskip 1em

\structure{During evaluation:} all neurons active
\begin{itemize}
    \item Outputs scaled by $(1-p)$ to compensate
    \item Equivalently: weights multiplied by $(1-p)$ at test time
\end{itemize}

\vskip 1em

\structure{Typical values:} $p = 0.1$--$0.5$ --- tuned on validation set

\end{column}
\begin{column}{0.4\textwidth}
\centering
\input{\CommonPath/tikz/dropout.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Dropout --- Ensemble Interpretation}

\vskip 2em

\structure{Why does dropout work?}

\vskip 1em

With $n$ neurons, dropout implicitly trains $2^n$ different subnetworks --- one per possible dropout mask

\vskip 1em

\begin{itemize}
    \item Each mini-batch trains a different subnetwork
    \item Subnetworks share weights --- efficient training
    \item At test time: full network approximates \textbf{averaging} over all subnetworks
\end{itemize}

\vskip 1.5em

\structure{Ensembles generalise better} than single models --- averaging reduces variance

\vskip 1em

\begin{center}
Dropout achieves ensemble-like generalisation at the cost of a single model
\end{center}

\end{frame}

% -------------------------------------------------------

\begin{frame}{\texttt{model.train()} and \texttt{model.eval()}}

\vskip 2em

\structure{Some layers behave differently during training and evaluation:}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Dropout}
\begin{itemize}
    \item \textbf{Train:} randomly zero activations with prob $p$
    \item \textbf{Eval:} all neurons active, outputs scaled by $(1-p)$
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Batch Normalisation}
\begin{itemize}
    \item \textbf{Train:} normalise using mini-batch statistics
    \item \textbf{Eval:} normalise using running statistics accumulated during training
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1.5em

\structure{All other layers} --- \texttt{nn.Linear}, \texttt{nn.ReLU}, etc.\ --- are unaffected by the mode

\vskip 1em

\begin{center}
\alert{Always call \texttt{model.eval()} before validation and inference}\\[0.5em]
\alert{Always call \texttt{model.train()} before resuming training}
\end{center}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Randomness as Regularisation}

\vskip 2em

\structure{A recurring theme:} many effective regularisation techniques inject randomness

\vskip 1em

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Technique} & \textbf{Source of randomness} & \textbf{Type} \\
\midrule
Weight decay & --- & explicit, deterministic \\
Mini-batch SGD & random batch selection & implicit \\
Dropout & random neuron masking & explicit \\
Data augmentation & random transformations & explicit \\
\bottomrule
\end{tabular}
\end{center}

\vskip 1.5em

\structure{Mini-batch SGD as implicit regularisation:}
\begin{itemize}
    \item No change to the loss function
    \item Gradient noise prevents convergence to sharp minima
    \item Flat minima found for free --- at no extra cost
\end{itemize}

\end{frame}
