\begin{frame}{Where We Are}

\vskip 2em

\structure{So far we have:}
\begin{itemize}
    \item \alert{Network} $f_{\pvec} : \inspace \to \outspace$ - parameterized function
    \item \alert{Loss} $\loss(\pvec;\, \dataset)$ - measures fit on training data
    \item \alert{Optimization} - algorithms to minimize $\loss(\pvec;\, \dataset)$
\end{itemize}

\vskip 1.5em

\structure{Open question:}

\vskip 0.5em

\begin{center}
{\large Does minimizing training loss give us a \textbf{good model}?}
\end{center}

\vskip 1.5em

\structure{This lecture:} what good actually means, and how to get there

\end{frame}

% -

\begin{frame}{True Goal}

\vskip 2em

\structure{What we optimize:}
\[
    \pvec^* = \arg\min_{\pvec}\; \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
\]

\vskip 0.5em

\structure{What we actually want:} low loss on \textbf{new, unseen examples from the population}
\[
    \loss_{\text{pop}}(\pvec) = \mathbb{E}_{(\vx, y) \sim p_{\text{data}}}\bigl[\ell(f_{\pvec}(\vx),\, y)\bigr]
\]

\vskip 0.5em

\structure{Generalization gap:}
\[
    \underbrace{\loss_{\text{pop}}(\pvec)}_{\text{true goal}} - \underbrace{\loss_{\text{train}}(\pvec)}_{\text{what we minimize}} \geq 0
\]

\vskip 0.5em

\alert{\textbf{Test set:}} finite sample we create to \textit{estimate} $\loss_{\text{pop}}$ - \textbf{measurement tool, not the goal}!

\end{frame}

% -

\begin{frame}{Train / Validation / Test Split}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Training set}\\[0.5em]
Fit model parameters
\[
    \min_{\pvec}\; \loss(\pvec;\, \dataset_{\text{train}})
\]
\end{column}
\begin{column}{0.32\textwidth}
\structure{Validation set}\\[0.5em]
Select model, tune hyperparameters
\[
    \loss(\pvec;\, \dataset_{\text{val}})
\]
Touched many times during development
\end{column}
\begin{column}{0.32\textwidth}
\structure{Test set}\\[0.5em]
Final honest evaluation
\[
    \loss(\pvec;\, \dataset_{\text{test}})
\]
\alert{\textbf{Touched once, at the end}}
\end{column}
\end{columns}

\vskip 2em

\structure{Common mistake:} tuning on test set - reported performance is optimistic

\vskip 0.5em

\structure{Typical splits:} 80/10/10 or 70/15/15 - depends on dataset size

\end{frame}

\begin{frame}{Overfitting and Underfitting}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.55\textwidth}

\structure{Underfitting:} model too simple
\begin{itemize}
    \item High train loss, high test loss
    \item Small generalisation gap
\end{itemize}

\vskip 1em

\structure{Overfitting:} model memorises training data
\begin{itemize}
    \item Low train loss, high test loss
    \item \alert{Large generalisation gap}
\end{itemize}

\begin{frame}{What To Do}

\vskip 3em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Underfitting} {\small (high train loss, small gap)}
\begin{itemize}
    \item Increase model capacity
    \item Train longer
    \item Tune learning rate and optimiser
    \item Reduce regularisation if applied
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Overfitting} {\small (low train loss, large gap)}
\begin{itemize}
    \item More data - always most effective
    \item \alert{Regularisation} - this lecture
    \item Reduce model capacity
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}


\end{frame}


\begin{frame}{Bias-Variance Tradeoff}

\vskip 2em

\structure{Classical view:} test error decomposes into two competing terms

\vskip 1.5em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Bias} - error from wrong assumptions
\begin{itemize}
    \item Model too simple to capture the data
    \item High train loss, high test loss
    \item Underfitting
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Variance} - sensitivity to training set
\begin{itemize}
    \item Model fits noise in training data
    \item Low train loss, high test loss
    \item Overfitting
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Classical prescription:} choose model complexity to balance the two

\vskip 1em

\begin{center}
\structure{More parameters} $\Rightarrow$ lower bias, higher variance $\Rightarrow$ \alert{overfitting}
\end{center}

\end{frame}

% -

\begin{frame}{Limits of Classical View}

\vskip 2em

\structure{Classical prediction:} more parameters than training examples $\Rightarrow$ \alert{catastrophic overfitting}

\vskip 1em

\structure{Reality in deep learning:}
\begin{itemize}
    \item Deep NNs: huge numbers of params - can memorise training data
    \item Deep NNs: routinely drive train loss close to zero
    \item Classical theory: model that nearly memorises train data \emph{\textbf{cannot generalise}}
    \item Deep NNs: \emph{\textbf{generalise well in practice}} ... hmm ... 
\end{itemize}

\vskip 1.5em

\begin{center}
{\large Why do deep NNs generalise despite being massively overparameterised?}
\end{center}

\end{frame}

\begin{frame}{Many Good Minima}

\vskip 2em

\structure{Classical concern:} gradient descent finds a local minimum - but which one?

\vskip 1em

\structure{Empirical observation in deep networks:}
\begin{itemize}
    \item Overparameterised networks - vast number of local minima
    \item Most local minima have very similar loss values
    \item Different intialisation -> different minima ->  \emph{\textbf{yet generalise similarly}}
\end{itemize}

\centering
\vskip 2em

\structure{Why:} NNs far more params than needed to fit the data.

\vskip 1em

Many equivalent solutions exist - which one we find not so critical

\end{frame}

% -

\begin{frame}{Flat vs Sharp Minima}

\vskip 2em

\structure{Not all minima generalise equally well}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.28\textwidth}
\structure{Sharp minimum:}
\begin{itemize}
    \item Loss changes rapidly
    \item Small distribution shift $\Rightarrow$ loss jumps
    \item Poor generalisation
\end{itemize}
\end{column}
\begin{column}{0.28\textwidth}
\structure{Flat minimum:}
\begin{itemize}
    \item Loss changes slowly
    \item Small distribution shift $\Rightarrow$ loss stays low
    \item Good generalisation
\end{itemize}
\end{column}
\begin{column}{0.38\textwidth}
\centering
% \input{\CommonPath/tikz/flat_sharp_minima.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1.5em

\structure{SGD:} mini-batching stochastic noise - favours flat minima -> \alert{implicit regularisation}

\end{frame}

\begin{frame}{Double Descent}

\vskip 2em

\structure{Classical bias-variance curve:} test error has one optimal complexity - U-shaped curve

\vskip 0.5em

\structure{Double descent:} beyond the interpolation threshold, test error decreases again

\vskip 1em

\begin{columns}[c]
\begin{column}{0.5\textwidth}
\centering
% \input{\CommonPath/tikz/double_descent.tex}
\end{column}
\begin{column}{0.46\textwidth}
\begin{itemize}
    \item \structure{Underfitting:} too little capacity
    \item \structure{Interpolation threshold:} maximally rigid - worst generalisation
    \item \structure{Overparameterised:} smoother interpolation - generalisation recovers
\end{itemize}

\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\centering
\vskip 2em

\alert{Deep networks operate in overparameterised regime}

\end{frame}
