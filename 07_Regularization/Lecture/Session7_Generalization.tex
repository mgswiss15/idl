\section{Generalization}
% -

\begin{frame}{Where We Are}

\vskip 2em

\structure{So far we have:}
\begin{itemize}
    \item \alert{Network} $f_{\pvec} : \inspace \to \outspace$ - parameterized function
    \item \alert{Loss} $\loss(\pvec;\, \dataset)$ - measures fit on training data
    \item \alert{Optimization} - algorithms to minimize $\loss(\pvec;\, \dataset)$
\end{itemize}

\vskip 1.5em

\structure{Open question:}

\vskip 0.5em

\begin{center}
{\large Does minimizing training loss give us a \textbf{good model}?}
\end{center}

\vskip 1.5em

\structure{This lecture:} what good actually means, and how to get there

\end{frame}

% -

\begin{frame}{True Goal}

\vskip 2em

\structure{What we optimize:}
\[
    \pvec^* = \arg\min_{\pvec}\; \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell(f_{\pvec}(\vx^{(i)}),\, y^{(i)})
\]

\vskip 0.5em

\structure{What we actually want:} low loss on \textbf{new, unseen examples from the population}
\[
    \loss_{\text{pop}}(\pvec) = \mathbb{E}_{(\vx, y) \sim p_{\text{data}}}\bigl[\ell(f_{\pvec}(\vx),\, y)\bigr]
\]

\vskip 0.5em

\structure{Generalization gap:}
\[
    \underbrace{\loss_{\text{pop}}(\pvec)}_{\text{true goal}} - \underbrace{\loss_{\text{train}}(\pvec)}_{\text{what we minimize}} \geq 0
\]

\vskip 0.5em

\alert{\textbf{Test set:}} finite sample we create to \textit{estimate} $\loss_{\text{pop}}$ - \textbf{measurement tool, not the goal}!

\end{frame}

% -

\begin{frame}{Train / Validation / Test Split}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Training set}\\[0.5em]
Fit model parameters
\[
    \min_{\pvec}\; \loss(\pvec;\, \dataset_{\text{train}})
\]
\end{column}
\begin{column}{0.32\textwidth}
\structure{Validation set}\\[0.5em]
Select model, tune hyperparameters
\[
    \loss(\pvec;\, \dataset_{\text{val}})
\]
Touched many times during development
\end{column}
\begin{column}{0.32\textwidth}
\structure{Test set}\\[0.5em]
Final honest evaluation
\[
    \loss(\pvec;\, \dataset_{\text{test}})
\]
\alert{\textbf{Touched once, at the end}}
\end{column}
\end{columns}

\vskip 2em

\structure{Common mistake:} tuning on test set - reported performance is optimistic

\vskip 0.5em

\structure{Typical splits:} 80/10/10 or 70/15/15 - depends on dataset size

\end{frame}

\begin{frame}{What To Do}

\vskip 3em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Underfitting} {\small (high train loss, small gap)}
\begin{itemize}
    \item Increase model capacity
    \item Train longer
    \item Tune learning rate and optimiser
    \item Reduce regularisation if applied
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Overfitting} {\small (low train loss, large gap)}
\begin{itemize}
    \item More data - always most effective
    \item \alert{Regularisation} - this lecture
    \item Reduce model capacity
\end{itemize}

\begin{frame}{Bias-Variance Tradeoff}

\vskip 2em

\structure{Classical view:} test error decomposes into two competing terms

\vskip 1.5em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Bias} - error from wrong assumptions
\begin{itemize}
    \item Model too simple to capture the data
    \item High train loss, high test loss
    \item Underfitting
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Variance} - sensitivity to training set
\begin{itemize}
    \item Model fits noise in training data
    \item Low train loss, high test loss
    \item Overfitting
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Classical prescription:} choose model complexity to balance the two

\vskip 1em

\begin{center}
\structure{More parameters} $\Rightarrow$ lower bias, higher variance $\Rightarrow$ \alert{overfitting}
\end{center}

\end{frame}

% -

\begin{frame}{Limits of Classical View}

\vskip 2em

\structure{Classical prediction:} more parameters than training examples $\Rightarrow$ \alert{catastrophic overfitting}

\vskip 1em

\structure{Reality in deep learning:}
\begin{itemize}
    \item Deep NNs: huge numbers of params - can memorise training data
    \item Deep NNs: routinely drive train loss close to zero
    \item Classical theory: model that nearly memorises train data \emph{\textbf{cannot generalise}}
    \item Deep NNs: \emph{\textbf{generalise well in practice}} ... hmm ... 
\end{itemize}

\vskip 1.5em

\begin{center}
{\large Why do deep NNs generalise despite being massively overparameterised?}
\end{center}

\end{frame}

\begin{frame}{Many Good Minima}

\vskip 2em

\structure{Classical concern:} gradient descent finds a local minimum - but which one?

\vskip 1em

\structure{Empirical observation in deep networks:}
\begin{itemize}
    \item Overparameterised networks - vast number of local minima
    \item Most local minima have very similar loss values
    \item Different intialisation -> different minima ->  \emph{\textbf{yet generalise similarly}}
\end{itemize}

\centering
\vskip 2em

\structure{Why:} NNs far more params than needed to fit the data.

\vskip 1em

Many equivalent solutions exist - which one we find not so critical

\end{frame}

% -

\begin{frame}{Flat vs Sharp Minima}

\vskip 2em

\structure{Not all minima generalise equally well}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.28\textwidth}
\structure{Sharp minimum:}
\begin{itemize}
    \item Loss changes rapidly
    \item Small distribution shift $\Rightarrow$ loss jumps
    \item Poor generalisation
\end{itemize}
\end{column}
\begin{column}{0.28\textwidth}
\structure{Flat minimum:}
\begin{itemize}
    \item Loss changes slowly
    \item Small distribution shift $\Rightarrow$ loss stays low
    \item Good generalisation
\end{itemize}
\end{column}

\vskip 1.5em

\structure{SGD:} mini-batching stochastic noise - favours flat minima -> \alert{implicit regularisation}

\end{frame}

\begin{frame}{Weight Decay}

\vskip 2em

\structure{Explicit regularisation:} add penalty term to the loss

\[
    \tilde{\loss}(\pvec) = \loss(\pvec) + \frac{\lambda}{2}\|\pvec\|^2
\]

Large weights $\Rightarrow$ model relies heavily on specific features $\Rightarrow$ high variance

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{L2 regularisation} (weight decay)
\begin{itemize}
    \item Penalises $\|\pvec\|^2$ - shrinks all weights
    \item Smooth, differentiable
    \item \alert{\texttt{weight\_decay} in \texttt{torch.optim}}
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{L1 regularisation}
\begin{itemize}
    \item Penalises $\|\pvec\|_1$ - promotes sparsity
    \item Some weights go exactly to zero
    \item Useful for feature selection
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1em

\structure{Hyperparameter:} $\lambda > 0$ controls strength - tuned on validation set

\vskip 0.5em

\structure{Deep NNs:} baseline, combined with other techniques - less central than in classical ML

\end{frame}

% -

\begin{frame}{Dropout - Training}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.5\textwidth}

\structure{During training:} randomly zero each activation with probability $p$

\vskip 1em

\begin{itemize}
    \item Each forward pass uses a different subnetwork
    \item Neurons cannot rely on specific others always being present
    \item Prevents co-adaptation of neurons
\end{itemize}


\end{column}
\begin{column}{0.45\textwidth}
\centering
\input{\CommonPath/tikz/dropout.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Typical values:} $p = 0.1$--$0.5$ - tuned on validation set

\end{frame}

\begin{frame}{Dropout - Evaluation}

\vskip 2em

\structure{During evaluation:} all neurons active - outputs scaled by $(1-p)$ to compensate

\vskip 1em

\structure{\texttt{model.train()} vs \texttt{model.eval()}:}
\begin{itemize}
    \item \texttt{nn.Dropout} - zeroing active in train mode, disabled in eval mode
    \item Most layers (\texttt{nn.Linear}, \texttt{nn.ReLU}, \ldots) - unaffected
    \item Some other layers also have dual behaviour - covered later
\end{itemize}

\vskip 1em

\begin{columns}
\begin{column}{0.1\textwidth}

\end{column}  
\begin{column}{0.4\textwidth}
\structure{Correct pattern:}\\[0.3em]
{\scriptsize\ttfamily
model.train()\\
for X\_batch, y\_batch in train\_loader:\\
\hspace{1.5em}...\ \# training step\\[0.5em]
model.eval()\\
with torch.no\_grad():\\
\hspace{1.5em}...\ \# validation
}
\end{column}  
\begin{column}{0.3\textwidth}

\vskip 1em

\alert{Forgetting \texttt{model.eval()} is a silent bug - no error, wrong predictions}
\end{column}  
\begin{column}{0.1\textwidth}

\end{column}  

\end{columns}
\end{frame}

\begin{frame}{Dropout - Ensemble Interpretation}

\vskip 2em

\structure{Why does dropout work?}

\vskip 1em

With $n$ neurons, dropout implicitly trains $2^n$ different subnetworks - one per possible dropout mask

\vskip 1em

\begin{itemize}
    \item Each mini-batch trains different subnetwork
    \item Subnetworks share weights - efficient training
    \item At test time: full network approximates \textbf{averaging} over all subnetworks
\end{itemize}

\vskip 1.5em

\structure{Ensembles generalise better} than single models - averaging reduces variance

\vskip 1em

\begin{center}
Dropout achieves ensemble-like generalisation at the cost of a single model
\end{center}

\end{frame}

\begin{frame}{Data Augmentation}

\vskip 2em

\structure{Idea:} apply label-preserving transformations to training examples to create new ones

\vskip 1em

\structure{Key principle - invariance:} the transformation must not change the label
\begin{itemize}
    \item Rotate image of cat $\Rightarrow$ still cat
    \item Add small noise to patient measurements $\Rightarrow$ same diagnosis
    \item Model should be invariant to these transformations - augmentation enforces it
\end{itemize}

\vskip 1em

\structure{Examples:}
\begin{itemize}
    \item \structure{Images:} random flips, rotations, crops, colour jitter
    \item \structure{Tabular data:} Gaussian noise on features, random feature masking
\end{itemize}

\vskip 1em

\alert{\textbf{Domain knowledge required - only you know which transformations preserve the label}}

\end{frame}


\begin{frame}{Early Stopping}

\vskip 2em

Training longer always reduces train loss - point beyond which generalisation gap grows

\vskip 2em

    \item Monitor validation loss each epoch
    \item Save model weights at best validation loss
    \item Stop when validation loss has not improved for $k$ epochs - \textbf{patience}
    \item Deploy the saved model, not the final one
\end{itemize}

\begin{frame}{Training Stability}

\vskip 2em

\structure{Regularisation:} can use large, overparameterised networks without overfitting :)

\vskip 1em

\structure{New problem:} deep NNs training instabilities - more layers amplify small issues :(

\vskip 1em

\begin{itemize}
    \item \structure{Internal covariate shift} - activation distributions shift as parameters update
    \item \structure{Vanishing/exploding grads} - signals decay or amplify exponentially through layers
    \item \structure{Dying ReLU} - neurons get stuck outputting zero permanently
\end{itemize}

\vskip 2em

\alert{Not overfitting problems - regularisation does not help here}

\end{frame}

\begin{frame}{Internal Covariate Shift}

\vskip 2em

\structure{Problem:} as weights update, the input distribution to each layer changes continuously

\vskip 1em

\begin{itemize}
    \item Layer $l$ learns to process inputs with a certain distribution
    \item Layer $l-1$ updates its weights - distribution shifts
    \item Layer $l$ must constantly re-adapt - a moving target
    \item Effect amplifies with depth - later layers more instability
\end{itemize}

\vskip 1em

\structure{Consequences:}
\begin{itemize}
    \item Slower convergence
    \item Need for lower learning rates
    \item High sensitivity to initialisation
\end{itemize}

\vskip 1em

\structure{Solution:} explicitly normalise activations at each layer - \structure{Batch Normalisation}

\end{frame}

\begin{frame}{Batch Normalisation}

\vskip 2em

\structure{Idea:} normalise activations within each mini-batch -> distributions stable across layers

\vskip 0.5em

\[
    \hat{h}_j^{(i)} = \frac{h_j^{(i)} - \mu_j}{\sqrt{\sigma^2_j + \epsilon}},
    \qquad
    \tilde{h}_j^{(i)} = \gamma_j \hat{h}_j^{(i)} + \beta_j, 
    \qquad
    \mu_j = \frac{1}{|\mathcal{B}|}\sum_{i \in \mathcal{B}} h_j^{(i)}, \quad 
    \sigma^2_j = \frac{1}{|\mathcal{B}|}\sum_{i \in \mathcal{B}} (h_j^{(i)} - \mu_j)^2
\]

$\mu_j, \sigma^2_j$ - mean \& var of feature $j$ across batch $\mathcal{B}$; \, $\gamma_j, \beta_j$ - learned per-feature scale \& shift

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{\texttt{model.train():}}
\begin{itemize}
    \item Normalise using mini-batch stats
    \item Accumulate running mean and var
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{\texttt{model.eval():}}
\begin{itemize}
    \item Normalise using running stats
    \item Deterministic - no batch dependence
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 1em

\structure{Benefits:} stable training, allows higher learning rates, less sensitive to initialisation

\vskip 0.5em

\structure{Side effect:} mini-batch noise introduces mild regularisation - but not main motivation

\end{frame}

\begin{frame}{Batch Norm and Dropout}

\vskip 2em

\structure{Problem:} batch norm and dropout interact badly - \textbf{variance shift}

\vskip 0.5em

During training, dropout reduces activation variance - running stats reflect this. At test time, dropout disabled - variance higher, running stats wrong $\Rightarrow$ wrong normalisation.

\vskip 1em

\structure{Practical options:}
\begin{itemize}
    \item \structure{Batch norm only} - sufficient in many cases, standard in CNNs
    \item \structure{Dropout only} - effective in MLPs, no interaction problems
    \item \structure{Both} - dropout \emph{after} batch norm; use with care
\end{itemize}


\end{frame}

\begin{frame}{Layer Normalisation}

\vskip 2em

\structure{Idea:} normalise across features per example - not across batch

\vskip 0.5em

\begin{itemize}
    \item Statistics computed per example independently - no running statistics needed
    \item Identical behaviour in \texttt{model.train()} and \texttt{model.eval()}
    \item Combines cleanly with dropout - no variance shift problem
\end{itemize}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Standard in:}
\begin{itemize}
    \item Transformers and sequence models
    \item MLPs when combined with dropout
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Other normalisations:}
\begin{itemize}
    \item \textbf{Group norm} - groups of channels
    \item \textbf{RMS norm} - no mean subtraction
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Practical guidance:} batch norm for CNNs; \, layer norm + dropout for MLPs and sequence

\end{frame}

\begin{frame}{Vanishing and Exploding Gradients}

\vskip 2em

\structure{Backprop multiplies Jacobians backwards} - effect compounds, earliest layers affected most

\[
    \frac{\partial \loss}{\partial \pvec^{(1)}} =
    \frac{\partial \loss}{\partial \vh^{(L)}}
    \cdot \frac{\partial \vh^{(L)}}{\partial \vh^{(L-1)}}
    \cdots
    \frac{\partial \vh^{(2)}}{\partial \vh^{(1)}}
    \cdot \frac{\partial \vh^{(1)}}{\partial \pvec^{(1)}}
\]


\vskip 2em

\begin{columns}[t]
\begin{column}{0.46\textwidth}
\structure{Vanishing} (factors $< 1$)
\begin{itemize}
    \item Gradient $\to 0$ at earliest layers
    \item Early layers stop learning
    \item Deep NNs effectively shallow
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\structure{Exploding} (factors $> 1$)
\begin{itemize}
    \item Gradient $\to \infty$ at earliest layers
    \item Single large step destroys weights
    \item Training diverges
\end{itemize}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\vskip 2em

\structure{Solutions:} Xavier/He init (Session 5); batch norm; gradient clipping (RNN, Transformers)

\end{frame}

\begin{frame}{Dying ReLU}

\vskip 2em

\alert{\textbf{Alarm:}} ReLU has zero gradient for all negative pre-activations - \alert{\textbf{half the input space}}

\[
    \frac{\partial \, \text{ReLU}(z)}{\partial z} = \begin{cases} 1 & z > 0 \\ 0 & z \leq 0 \end{cases}
\]


\vskip 2em

\structure{Relief:} neuron dies if pre-activation negative for \alert{\textbf{all}} training examples \alert{\textbf{simultaneously}}
\begin{itemize}
    \item Pre-activations vary across examples - some positive, some negative
    \item Batchnorm helps by balancing around zero
    \item Average gradient is nonzero - diversity in data saves us
    \item \textbf{ReLU remains solid baseline}
\end{itemize}

\end{frame}

\begin{frame}{Embracing Randomness}

\vskip 1em

\begin{center}
\begin{tabular}{lll}

\end{center}

\end{itemize}

\end{itemize}
