\section{Computation Graphs}
% -------------------------------------------------------

\begin{frame}{Computation Graphs --- Motivation}

\vskip 1em

\structure{Idea:} decompose complex functions into simple atomic operations

\vskip 1em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Why?}
\begin{itemize}
\item Make computation of gradients easier
\item Each node only needs to know its \textbf{local} derivative
\item \textbf{Reuse} intermediate values from forward pass in backward pass
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{How?}
\begin{itemize}
\item \alert{Input nodes} --- data and parameters
\item \alert{Compute nodes} --- elementary operations
\item Edges --- data flow between nodes
\end{itemize}
\end{column}
\end{columns}

\vskip 1.5em

\structure{Example:} linear regression $\ell(x, y, \pvec) = (\theta_0 + \theta_1 x - y)^2$

\begin{frame}{Computation Graphs --- Scalar vs Vector}

\vskip 0.5em

\structure{Scalar notation} (one neuron at a time) vs \structure{vector notation} (whole layer at once):

\vskip 1em
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_7.pdf}

\vskip 1em

\begin{columns}[t]

\section{Backpropagation}
% -------------------------------------------------------

\begin{frame}{Forward and Backward Pass}

\vskip 2em

\structure{Forward pass} --- compute prediction and loss:

\vskip 0.5em
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_1.pdf}

\vskip 1.5em

\begin{columns}[c]
\begin{column}{0.48\textwidth}
\centering
\structure{Forward pass}\\[0.3em]
$\vx \longrightarrow \vh^{(1)} \longrightarrow \cdots \longrightarrow \vh^{(L)} \longrightarrow \ell$
\end{column}
\begin{column}{0.48\textwidth}
\centering
\structure{Backward pass}\\[0.3em]
$\dfrac{\partial \ell}{\partial \pvec} \longleftarrow \cdots \longleftarrow \dfrac{\partial \ell}{\partial \vh^{(L)}} \longleftarrow \ell$
\end{column}
\end{columns}

\vskip 1em
\centering
\alert{\textbf{Backpropagation} = computing the backward pass efficiently}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Two Key Rules}

\vskip 1em

\structure{Goal:} compute $\nabla_{\pvec}\, \loss(\pvec)$ for the full dataset

\vskip 1.5em

\structure{1) Sum rule} --- gradient of average = average of gradients:
\[
\nabla_{\pvec}\, \loss(\pvec)
= \nabla_{\pvec} \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \ell(\vx^{(i)}, y^{(i)}, \pvec)
= \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \nabla_{\pvec}\, \ell(\vx^{(i)}, y^{(i)}, \pvec)
\]

\vskip 0.5em
\alert{$\Rightarrow$ compute gradient for a single sample, then average}

\vskip 1.5em

\structure{2) Chain rule} --- THE rule for backprop:
\[
\frac{d}{dx} f(g(x)) = \frac{df}{dg}\frac{dg}{dx}
\qquad
\frac{d}{dx} f(g_1(x), \ldots, g_m(x)) = \sum_{i=1}^{m} \frac{\partial f}{\partial g_i}\frac{dg_i}{dx}
\]

\end{frame}

% -------------------------------------------------------

\begin{frame}{Local vs Global Gradients}

\vskip 0.5em

\centering
\includegraphics[width=0.65\textwidth]{\PicsPath/CompGraph_locglob_1.pdf}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Local gradient} (easy --- per node):
\[ \frac{\partial\, \text{out}}{\partial\, \text{in}} \]
Each node computes its own derivative
\end{column}
\begin{column}{0.48\textwidth}
\structure{Global gradient} (chain rule):
\[ \frac{\partial \ell}{\partial\, \text{in}} = \frac{\partial \ell}{\partial\, \text{out}} \cdot \frac{\partial\, \text{out}}{\partial\, \text{in}} \]
Multiply incoming global gradient by local gradient
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Backprop Example: Linear Regression (Symbolic)}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass}\\[0.3em]
{\small
(1) $h = \theta_1 x$\\
(2) $\yhat = \theta_0 + h$\\
(3) $z = \yhat - y$\\
(4) $\ell = z^2$
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color{thwsblue}Local gradients}\\[0.3em]
{\small
(1) $\partial h/\partial \theta_1 = x$\\
(2) $\partial \yhat/\partial \theta_0 = 1$\\
\phantom{(2)} $\partial \yhat/\partial h = 1$\\
(3) $\partial z/\partial \yhat = 1$\\
(4) $\partial \ell/\partial z = 2z$
}
\end{column}
\begin{column}{0.36\textwidth}
\structure{\color{carnelian}Global gradients}\\[0.3em]
{\small
(1) $\partial \ell/\partial \theta_1 = \partial \ell/\partial h \cdot x = 2zx$\\
(3) $\partial \ell/\partial \theta_0 = \partial \ell/\partial \yhat \cdot 1 = 2z$\\
\phantom{(3)} $\partial \ell/\partial h = \partial \ell/\partial \yhat \cdot 1 = 2z$\\
(2) $\partial \ell/\partial \yhat = \partial \ell/\partial z \cdot 1 = 2z$\\
(1) $\partial \ell/\partial z = 1 \cdot 2z = 2z$
}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Backprop Example: Linear Regression (Numerical)}

\vskip 0.5em

\structure{Concrete values:} $x=2,\ y=5,\ \theta_0=1,\ \theta_1=3$

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass}\\[0.3em]
{\small
(1) $h = 3 \cdot 2 = 6$\\
(2) $\yhat = 1 + 6 = 7$\\
(3) $z = 7 - 5 = 2$\\
(4) $\ell = 2^2 = 4$
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color{thwsblue}Local gradients}\\[0.3em]
{\small
(1) $\partial h/\partial \theta_1 = 2$\\
(2) $\partial \yhat/\partial \theta_0 = 1$\\
\phantom{(2)} $\partial \yhat/\partial h = 1$\\
(3) $\partial z/\partial \yhat = 1$\\
(4) $\partial \ell/\partial z = 4$
}
\end{column}
\begin{column}{0.36\textwidth}
\structure{\color{carnelian}Global gradients}\\[0.3em]
{\small
(1) $\partial \ell/\partial \theta_1 = 4 \cdot 2 = 8$\\
(3) $\partial \ell/\partial \theta_0 = 4 \cdot 1 = 4$\\
\phantom{(3)} $\partial \ell/\partial h = 4 \cdot 1 = 4$\\
(2) $\partial \ell/\partial \yhat = 4 \cdot 1 = 4$\\
(1) $\partial \ell/\partial z = 1 \cdot 4 = 4$
}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Backprop Example: Logistic Regression}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass}\\[0.3em]
{\small
(1) $h = \theta_0 + \theta_1 x$\\
(2) $\yhat = \sigma(h)$\\
(3) $\ell = -y\log\yhat$\\
\phantom{(3)} $\quad - (1-y)\log(1-\yhat)$
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color{thwsblue}Local gradients}\\[0.3em]
{\small
(1) $\partial h/\partial \theta_0 = 1$\\
\phantom{(1)} $\partial h/\partial \theta_1 = x$\\
(2) $\partial \yhat/\partial h = \sigma(h)(1-\sigma(h))$\\
(3) $\partial \ell/\partial \yhat = \frac{\yhat - y}{\yhat(1-\yhat)}$
}
\end{column}
\begin{column}{0.36\textwidth}
\structure{\color{carnelian}Global gradients}\\[0.3em]
{\small
(3) $\partial \ell/\partial \theta_0 = \partial \ell/\partial h \cdot 1$\\
\phantom{(3)} $\partial \ell/\partial \theta_1 = \partial \ell/\partial h \cdot x$\\
(2) $\partial \ell/\partial h = \partial \ell/\partial \yhat \cdot \partial \yhat/\partial h$\\
(1) $\partial \ell/\partial \yhat = 1 \cdot \frac{\yhat-y}{\yhat(1-\yhat)}$
}
\end{column}
\end{columns}

\vskip 0.5em
\structure{Key insight:} same graph structure, only the node functions change

\end{frame}

\begin{frame}{Backprop: Multiple Neurons in a Layer}

\vskip 0.5em

\centering
\includegraphics[width=0.5\textwidth]{\PicsPath/CompGraph_6.pdf}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Forward pass}\\[0.3em]
{\small
(1) $h = h(x, \theta_0)$\\
(2) $u = u(h, \theta_1)$, $\quad v = v(h, \theta_2)$\\
(3) $\yhat = \yhat(u, v)$\\
(4) $\ell = \ell(\yhat, y)$
}
\end{column}
\begin{column}{0.48\textwidth}
\structure{\color{carnelian}Global gradients}\\[0.3em]
{\small
(3) $\partial \ell/\partial \theta_1 = \partial \ell/\partial u \cdot \partial u/\partial \theta_1$\\
\phantom{(3)} $\partial \ell/\partial \theta_2 = \partial \ell/\partial v \cdot \partial v/\partial \theta_2$\\
(2) $\partial \ell/\partial h = \partial \ell/\partial u \cdot \partial u/\partial h$\\
\phantom{(2)} $\qquad\qquad + \partial \ell/\partial v \cdot \partial v/\partial h$\\
}
\end{column}
\end{columns}

\vskip 0.3em
\alert{When a node feeds multiple downstream nodes, gradients \textbf{accumulate} (sum).}

\end{frame}

\begin{frame}{Technical Implementation}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.35\textwidth}
Each node stores:
\begin{itemize}
\item \texttt{\color{thwspetrol}.val} --- forward value
\item \texttt{\color{thwsorange}.grad} --- accumulated gradient
\end{itemize}
\vskip 0.5em
Each compute node implements:
\begin{itemize}
\item Forward function
\item Local gradient function
\end{itemize}
\end{column}
\begin{column}{0.30\textwidth}
\structure{Forward pass}\\[0.3em]
{\scriptsize\ttfamily
x.val, y.val = data\\
t0.val, t1.val = init\\
h.val = t1.val * x.val\\
yp.val = t0.val + h.val\\
z.val = yp.val - y.val\\
l.val = z.val ** 2
}
\end{column}
\begin{column}{0.35\textwidth}
\structure{Backward pass}\\[0.3em]
{\scriptsize\ttfamily
l.grad = 1\\
z.grad = l.grad * 2*z.val\\
yp.grad = z.grad * 1\\
t0.grad = yp.grad * 1\\
h.grad = yp.grad * 1\\
t1.grad = h.grad * x.val
}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{From For Loops to Vectorization}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Elementwise for loops}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
for i: z.val[i] = b.val[i]\\
for i,j: z.val[j] += W.val[i,j] * x.val[i]\\
for i: a.val[i] = act(z.val[i])\\[0.5em]
\# Backward\\
for i: z.grad[i] = a.grad[i] * dact(z.val[i])\\
for i,j: W.grad[i,j] = z.grad[j] * x.val[i]\\
for i: b.grad[i] = z.grad[i]
}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Vectorized}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
z.val = W.val @ x.val + b.val\\
a.val = act(z.val)\\[0.5em]
\# Backward\\
z.grad = a.grad * dact(z.val)\\
W.grad = z.grad[:,None] * x.val[None,:]\\
b.grad = z.grad
}
\vskip 1em
\alert{Same computation, $\sim$100$\times$ faster on GPU}
\end{column}
\end{columns}

\end{frame}

\section*{Summary}
% -------------------------------------------------------

\begin{frame}{Summary}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Computation graphs}
\begin{itemize}
\item Decompose complex functions into atomic operations
\item Forward pass --- function evaluation
\item Backward pass --- gradient computation
\item Flexible level of detail and definition of operations
\end{itemize}

\vskip 1em

\structure{Backpropagation}
\begin{itemize}
\item Computes gradients --- does \textbf{not} optimize
\item Two rules: sum rule + chain rule
\item {\color{thwsblue}Local gradients} easy per node
\item {\color{carnelian}Global gradients} via chain rule
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Implementation}
\begin{itemize}
\item Each node stores \texttt{.val} and \texttt{.grad}
\item Compute nodes only need to be local-aware:
    \begin{itemize}
    \item inputs and outputs
    \item forward function
    \item local gradient function
    \end{itemize}
\item Vectorize for efficiency ($\sim$100$\times$ faster on GPU)
\end{itemize}

\vskip 1em

\structure{Next}
\begin{itemize}
\item PyTorch autograd does all of this automatically
\item \texttt{loss.backward()} = one backward pass
\end{itemize}
\end{column}
\end{columns}

\end{frame}
