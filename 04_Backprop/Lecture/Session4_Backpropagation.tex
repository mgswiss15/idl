\begin{frame}{Forward and Backward Pass}

\vskip 1.5em

\structure{Forward pass} - evaluate network $\to$ get predictions $\yhat \, \to$ compute loss $\loss$

\vskip 0.5em

\[
\vx = \underbrace{\vh^{(0)} \xrightarrow{\thetamat{1}} \vh^{(1)} \xrightarrow{\thetamat{2}} \cdots \xrightarrow{\thetamat{L}} \vh^{(L)}}_{f_{\pvec}} = \yhat \xrightarrow{\ell} \loss
\]

\vskip 0.5em

\structure{Backward pass} - propagate gradients from loss back to parameters: \alert{\textbf{Backpropagation}}

\vskip 0.5em

\[
\nabla_{\pvec}\,\loss(\pvec) = \underbrace{\frac{\partial \loss}{\partial \pvec} \xleftarrow{\thetamat{1}} \frac{\partial \loss}{\partial \vh^{(1)}} \xleftarrow{\thetamat{2}} \cdots \xleftarrow{\thetamat{L}} \frac{\partial \loss}{\partial \vh^{(L)}}}_{\nabla_{\pvec}} \xleftarrow{\ell} \loss
\]

\vskip 0.8em
\structure{Gradient:} $\nabla_{\pvec}\,\loss(\pvec) = \dfrac{\partial \loss}{\partial \pvec}$ - one partial derivative per parameter

\end{frame}

\section{Computation Graphs}
% -------------------------------------------------------
 
\begin{frame}{Computation Graphs - Motivation}
 
\vskip 2em
 
\structure{Idea:} decompose complex functions into simple atomic operations

\vskip 1.5em

\structure{Why?}
\begin{itemize}
\item Gradients of complex functions are hard - gradients of atomic operations are easy
\item Intermediate values from forward pass can be reused in backward pass
\end{itemize}

\vskip 1em

\structure{Graph structure:}
\begin{itemize}
\item \alert{Input nodes} - fixed data ($x$, $y$)
\item \alert{Parameter nodes} - learnable parameters ($\pvec$)
\item \alert{Compute nodes} - elementary operations, each with a simple local derivative
\end{itemize}
 
\end{frame}

\section{Backpropagation}

\begin{frame}{Forward and Backward Pass (Backpropagation)}

\vskip 2em

\structure{Forward pass} - prediction and loss; \hskip 1cm \structure{Backward pass} - parameter gradients

\vskip 2em
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_1.pdf}

\vskip 1.5em

\begin{columns}[c]
\begin{column}{0.48\textwidth}
\centering
\structure{Forward pass}\\[0.3em]
$\vx \longrightarrow \vh^{(1)} \longrightarrow \cdots \longrightarrow \vh^{(L)} \longrightarrow \ell$
\end{column}
\begin{column}{0.48\textwidth}
\centering
\structure{Backpropagation}\\[0.3em]
$\dfrac{\partial \ell}{\partial \pvec} \longleftarrow \cdots \longleftarrow \dfrac{\partial \ell}{\partial \vh^{(L)}} \longleftarrow \ell$
\end{column}
\end{columns}

\vskip 1em

\end{frame}

% -------------------------------------------------------

\begin{frame}{Two Key Rules}

\vskip 2em

\structure{Goal:} compute $\nabla_{\pvec}\, \loss(\pvec)$ for the full dataset

\vskip 1.5em

\structure{1) Sum rule} - gradient of average = average of gradients:
\[
\nabla_{\pvec}\, \loss(\pvec)
= \nabla_{\pvec} \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \ell(\vx^{(i)}, y^{(i)}, \pvec)
= \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \nabla_{\pvec}\, \ell(\vx^{(i)}, y^{(i)}, \pvec)
\]

\vskip 0.5em
\alert{$\Rightarrow$ compute gradient for a single sample, then average}

\vskip 1.5em

\structure{2) Chain rule} - THE rule for backprop:
\[
\frac{d}{dx} f(g(x)) = \frac{df}{dg}\frac{dg}{dx}
\qquad
\frac{d}{dx} f(g_1(x), \ldots, g_m(x)) = \sum_{i=1}^{m} \frac{\partial f}{\partial g_i}\frac{dg_i}{dx}
\]

\end{frame}

\begin{frame}{Local vs Global Gradients}
 
\vskip 2em
 
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_locglob_1.pdf}
 
\vskip 1em
 
\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{\color[HTML]{0066FF}Local gradient} (easy --- per node):
\[ {\color[HTML]{0066FF}\frac{\partial\, \text{out}}{\partial\, \text{in}}} \]
Can compute during forward pass
\end{column}
\begin{column}{0.48\textwidth}
\structure{\color[HTML]{FF0000}Global gradient} (chain rule):
\[ {\color[HTML]{FF0000}\frac{\partial \ell}{\partial\, \text{in}}} = {\color[HTML]{FF0000}\frac{\partial \ell}{\partial\, \text{out}}} \cdot {\color[HTML]{0066FF}\frac{\partial\, \text{out}}{\partial\, \text{in}}} \]
Goes backward: incoming {\color[HTML]{FF0000}global} $\times$ {\color[HTML]{0066FF}local} $=$ new {\color[HTML]{FF0000}global} for the left
\end{column}
\end{columns}


\end{frame}

\begin{frame}{Backprop Example: Linear Regression (Symbolic)}
 
\vskip 1em
 
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_locglob_1.pdf}
 
\vskip 0.5em
 
\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass $\downarrow$}\\[-1em]
{\small
\begin{align*}
(1)\ & h = \theta_1 x\\
(2)\ & \yhat = \theta_0 + h\\
(3)\ & z = \yhat - y\\
(4)\ & \ell = z^2
\end{align*}
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color[HTML]{0066FF}Local gradients $\downarrow$}\\[-1em]
{\small\color[HTML]{0066FF}
\begin{align*}
(1)\ & \partial h/\partial \theta_1 = x\\
(2)\ & \partial \yhat/\partial \theta_0 = 1\\
     & \partial \yhat/\partial h = 1\\
(3)\ & \partial z/\partial \yhat = 1\\
(4)\ & \partial \ell/\partial z = 2z
\end{align*}
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color[HTML]{FF0000}Global gradients $\uparrow$}\\[-1em]
{\small\color[HTML]{FF0000}
\begin{align*}
(4)\ & \partial \ell/\partial \theta_1 = 2z \cdot {\color[HTML]{0066FF}x} = 2zx\\
     & \partial \ell/\partial \theta_0 = 2z \cdot {\color[HTML]{0066FF}1} = 2z\\
(3)\ & \partial \ell/\partial h = 2z \cdot {\color[HTML]{0066FF}1} = 2z\\
(2)\ & \partial \ell/\partial \yhat = 2z \cdot {\color[HTML]{0066FF}1} = 2z\\
(1)\ & \partial \ell/\partial z = 1 \cdot {\color[HTML]{0066FF}2z} = 2z
\end{align*}
}
\end{column}
\end{columns}
 
\end{frame}

\begin{frame}{Backprop Example: Linear Regression (Numerical)}
 
\vskip 1em
 
\centering
\includegraphics[width=0.55\textwidth]{\PicsPath/CompGraph_locglob_2.pdf}
 
\vskip 0.3em
 
\structure{Concrete values:} $x=2,\ y=5,\ \theta_0=1,\ \theta_1=3$
 
\vskip 0.3em
 
\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass $\downarrow$}\\[-1em]
{\small
\begin{align*}
(1)\ & h = \theta_1 x = 3 \cdot 2 = 6\\
(2)\ & \yhat = \theta_0 + h = 1 + 6 = 7\\
(3)\ & z = \yhat - y = 7 - 5 = 2\\
(4)\ & \ell = z^2 = 2^2 = 4
\end{align*}
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color[HTML]{0066FF}Local gradients $\downarrow$}\\[-1em]
{\small\color[HTML]{0066FF}
\begin{align*}
(1)\ & \partial h/\partial \theta_1 = x = 2\\
(2)\ & \partial \yhat/\partial \theta_0 = 1\\
     & \partial \yhat/\partial h = 1\\
(3)\ & \partial z/\partial \yhat = 1\\
(4)\ & \partial \ell/\partial z = 2z = 4
\end{align*}
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color[HTML]{FF0000}Global gradients $\uparrow$}\\[-1em]
{\small\color[HTML]{FF0000}
\begin{align*}
(4)\ & \partial \ell/\partial \theta_1 = \partial\ell/\partial h \cdot {\color[HTML]{0066FF}x} = 4 \cdot {\color[HTML]{0066FF}2} = 8\\
     & \partial \ell/\partial \theta_0 = \partial\ell/\partial h \cdot {\color[HTML]{0066FF}1} = 4 \cdot {\color[HTML]{0066FF}1} = 4\\
(3)\ & \partial \ell/\partial h = \partial\ell/\partial \yhat \cdot {\color[HTML]{0066FF}1} = 4 \cdot {\color[HTML]{0066FF}1} = 4\\
(2)\ & \partial \ell/\partial \yhat = \partial\ell/\partial z \cdot {\color[HTML]{0066FF}1} = 4 \cdot {\color[HTML]{0066FF}1} = 4\\
(1)\ & \partial \ell/\partial z = 1 \cdot {\color[HTML]{0066FF}2z} = 1 \cdot {\color[HTML]{0066FF}4} = 4
\end{align*}
}
\end{column}
\end{columns}
 
\end{frame}

\begin{columns}[t]

\structure{Forward pass $\downarrow$}\\[-1em]
{\small
\begin{align*}
(1)\ & h = \theta_0 + \theta_1 x\\
(2)\ & \yhat = \sigma(h)\\

\end{column}
\begin{column}{0.3\textwidth}
\structure{\color[HTML]{0066FF}Local gradients $\downarrow$}\\[-1em]
{\small\color[HTML]{0066FF}

(1)\ & \partial h/\partial \theta_0 = 1\\

\begin{column}{0.3\textwidth}
\structure{\color[HTML]{FF0000}Global gradients $\uparrow$}\\[-1em]
{\small\color[HTML]{FF0000}

(3)\ & \partial \ell/\partial \theta_0 = \partial \ell/\partial h \cdot {\color[HTML]{0066FF}1}\\
     & \partial \ell/\partial \theta_1 = \partial \ell/\partial h \cdot {\color[HTML]{0066FF}x}\\
(2)\ & \partial \ell/\partial h = \partial \ell/\partial \yhat \cdot {\color[HTML]{0066FF}\partial \yhat/\partial h}\\
(1)\ & \partial \ell/\partial \yhat = 1 \cdot {\color[HTML]{0066FF}\frac{\yhat-y}{\yhat(1-\yhat)}}
\end{align*}

\end{column}
\end{columns}

\begin{frame}{Backprop: Multiple Neurons in a Layer}
 
\vskip 1em
 
\centering
\includegraphics[width=0.5\textwidth]{\PicsPath/CompGraph_6.pdf}
 
\vskip 0.5em
 
\begin{columns}[t]
\begin{column}{0.32\textwidth}
\structure{Forward pass $\downarrow$}\\[-1em]
{\small
\begin{align*}
(1)\ & h = h(x, \theta_0)\\
(2)\ & u = u(h, \theta_1)\\
     & v = v(h, \theta_2)\\
(3)\ & \yhat = \yhat(u, v)\\
(4)\ & \ell = \ell(\yhat, y)
\end{align*}
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color[HTML]{0066FF}Local gradients $\downarrow$}\\[-1em]
{\small\color[HTML]{0066FF}
\begin{align*}
(1)\ & \partial h/\partial \theta_0\\
(2)\ & \partial u/\partial \theta_1, \quad \partial u/\partial h\\
     & \partial v/\partial \theta_2, \quad \partial v/\partial h\\
(3)\ & \partial \yhat/\partial u, \quad \partial \yhat/\partial v\\
(4)\ & \partial \ell/\partial \yhat
\end{align*}
}
\end{column}
\begin{column}{0.4\textwidth}
\structure{\color[HTML]{FF0000}Global gradients $\uparrow$}\\[-1em]
{\small\color[HTML]{FF0000}
\begin{align*}
(4)\ & \partial \ell/\partial \theta_0 = \partial \ell/\partial h \cdot {\color[HTML]{0066FF}\partial h/\partial \theta_0}\\
(3)\ & \partial \ell/\partial h = \partial \ell/\partial u \cdot {\color[HTML]{0066FF}\partial u/\partial h}
     + \partial \ell/\partial v \cdot {\color[HTML]{0066FF}\partial v/\partial h}\\
     & \partial \ell/\partial \theta_1 = \partial \ell/\partial u \cdot {\color[HTML]{0066FF}\partial u/\partial \theta_1}\\
     & \partial \ell/\partial \theta_2 = \partial \ell/\partial v \cdot {\color[HTML]{0066FF}\partial v/\partial \theta_2}\\
(2)\ & \partial \ell/\partial u = \partial \ell/\partial \yhat \cdot {\color[HTML]{0066FF}\partial \yhat/\partial u}\\
     & \partial \ell/\partial v = \partial \ell/\partial \yhat \cdot {\color[HTML]{0066FF}\partial \yhat/\partial v}\\
(1)\ & \partial \ell/\partial \yhat = 1 \cdot {\color[HTML]{0066FF}\partial \ell/\partial \yhat}
\end{align*}
}
\end{column}
\end{columns}
\vskip -1.5em
\alert{When a node feeds multiple downstream nodes, gradients \textbf{accumulate} (sum).}
 
\end{frame}

\begin{frame}{Technical Implementation}
 
\vskip 2em
 
\begin{itemize}
\item {\color[HTML]{71C837}Input nodes} --- store \texttt{*.val}
\item {\color[HTML]{00CCCC}Parameter nodes} --- store \texttt{*.val} and {\color[HTML]{FF0000}\texttt{*.grad}}
\item {\color[HTML]{FF7F2A}Compute nodes} --- store \texttt{*.val} and {\color[HTML]{FF0000}\texttt{*.grad}}, compute forward func and {\color[HTML]{0066FF}local grads}
\end{itemize}
 
\vskip 2em
 
\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Forward pass}\\[0.3em]
{\scriptsize\ttfamily
x.val = data\\
t0.val, t1.val = init\\
\ldots\\
h.val = func\_h(x.val, t0.val, t1.val)\\
\ldots
}
\end{column}
\begin{column}{0.48\textwidth}
\structure{\color[HTML]{FF0000}Backward pass}\\[0.3em]
{\scriptsize\ttfamily
t0.grad = t1.grad = h.grad = 0\\
l.grad = 1\\
\ldots\\
h.grad += yp.grad * {\color[HTML]{0066FF}df\_yp/h(h.val)}\\
t1.grad += h.grad * {\color[HTML]{0066FF}df\_h/t1(x.val,...)}\\
\ldots
}
\end{column}
\end{columns}
 
\end{frame}

\begin{frame}{From For Loops to Vectorization}

\vskip 2em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Elementwise for loops}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
for i: z.val[i] = b.val[i]\\
for i,j: z.val[j] += W.val[i,j] * x.val[i]\\
for i: a.val[i] = act(z.val[i])\\[0.5em]
\# Backward\\
for i: z.grad[i] = a.grad[i] * dact(z.val[i])\\
for i,j: W.grad[i,j] = z.grad[j] * x.val[i]\\
for i: b.grad[i] = z.grad[i]
}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Vectorized}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
z.val = W.val @ x.val + b.val\\
a.val = act(z.val)\\[0.5em]
\# Backward\\
z.grad = a.grad * dact(z.val)\\
W.grad = z.grad[:,None] * x.val[None,:]\\
b.grad = z.grad
}
\end{column}
\end{columns}

\centering
\vskip 2em
\alert{Same computation, $\sim$100$\times$ faster on GPU}

\end{frame}

\section*{Summary}
% -------------------------------------------------------
 
\begin{frame}{Summary}
 
\vskip 2em
 
\begin{itemize}
\item Neural networks are function compositions --- gradients must flow through all layers
\item \textbf{Computation graphs} decompose complex functions into atomic operations with easy local derivatives
\item \textbf{Forward pass} --- evaluate the function, store intermediate values
\item \textbf{Backward pass} --- propagate gradients right to left using the chain rule
\item \textbf{Backpropagation} computes gradients --- it does \textbf{not} optimize
\item {\color[HTML]{0066FF}Local gradients} computed per node from stored forward values
\item {\color[HTML]{FF0000}Global gradients} assembled via: incoming {\color[HTML]{FF0000}global} $\times$ {\color[HTML]{0066FF}local} $=$ new {\color[HTML]{FF0000}global} for the left
\item Vectorize for efficiency
\end{itemize}
 
\end{frame}
