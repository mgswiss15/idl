\section{What is Deep Learning?}

\begin{frame}{What is Deep Learning?}

\vskip 1em

\structure{Family of Techniques}
\begin{itemize}
    \item \textbf{Artificial Intelligence} - machines that exhibit intelligent behaviour
    \item \textbf{Machine Learning} - machines that learn from data rather than explicit rules
    \item \textbf{Deep Learning} - machine learning using deep neural networks
    \item \textbf{GenAI} - LLMs \& image/ video/ music/ and other generation
