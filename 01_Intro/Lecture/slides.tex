\begin{frame}{Stuff - Image Classification}

\structure{Task: Image Classification}
\begin{itemize}
    \item Given an image, assign it one label from a fixed set of categories
    \item Input: raw pixels \quad $\rightarrow$ \quad Output: a class label
\end{itemize}

\vskip 1em

\structure{CIFAR-10: A Benchmark Dataset}
\begin{itemize}
    \item 60,000 colour images of $32 \times 32$ pixels
    \item 10 classes, 6,000 images per class
\end{itemize}

\section{What is Deep Learning?}

\begin{frame}{What is Deep Learning?}

\vskip 1em

\structure{Family of Techniques}
\begin{itemize}
    \item \textbf{Artificial Intelligence} - machines that exhibit intelligent behaviour
    \item \textbf{Machine Learning} - machines that learn from data rather than explicit rules
    \item \textbf{Deep Learning} - machine learning using deep neural networks
    \item \textbf{GenAI} - LLMs \& image/ video/ music/ and other generation
\end{itemize}
