\begin{frame}{Stuff - Image Classification}

\begin{itemize}
    \item Given an image, assign it one label from a fixed set of categories
    \item Input: raw pixels \quad $\rightarrow$ \quad Output: a class label

\structure{CIFAR-10: A Benchmark Dataset}
\begin{itemize}
    \item 60,000 colour images of $32 \times 32$ pixels
    \item 10 classes, 6,000 images per class

\section{What is Deep Learning?}

\begin{frame}{What is Deep Learning?}

\vskip 1em

\structure{Family of Techniques}
\begin{itemize}
    \item \textbf{Artificial Intelligence} - machines that exhibit intelligent behaviour
    \item \textbf{Machine Learning} - machines that learn from data rather than explicit rules
    \item \textbf{Deep Learning} - machine learning using deep neural networks
    \item \textbf{GenAI} - LLMs \& image/ video/ music/ and other generation
\end{itemize}

\section{Applications}

\begin{frame}{What Can Deep Learning Do?}

\begin{columns}
    \begin{column}{0.48\textwidth}
        \structure{Vision}
        \begin{itemize}
            \item Image classification \& object detection
            \item Medical imaging - tumour detection, X-ray analysis
            \item Autonomous driving
            \item Face recognition
        \end{itemize}

        \vskip 1em

        \structure{Language}
        \begin{itemize}
            \item Machine translation
            \item Question answering - ChatGPT, Claude
            \item Code generation
            \item Document summarisation
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \structure{Generative Models}
        \begin{itemize}
            \item Image generation - Stable Diffusion, DALL-E
            \item Video generation
            \item Music and speech synthesis
            \item Drug and protein design
        \end{itemize}

        \vskip 1em

        \structure{Beyond}
        \begin{itemize}
            \item Game playing - AlphaGo, AlphaFold
            \item Robotics and control
            \item Scientific discovery
            \item Recommendation systems
        \end{itemize}
    \end{column}
\end{columns}
