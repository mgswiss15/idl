% Early stopping: train/val loss curves with best model marker
\begin{tikzpicture}
\begin{axis}[
    width=7cm, height=5.5cm,
    xlabel={Epoch},
    ylabel={Loss},
    xmin=0, xmax=10,
    ymin=0, ymax=5,
    xtick=\empty,
    ytick=\empty,
    legend pos=north east,
    legend style={font=\small, draw=none},
    tick label style={font=\small},
    clip=false,
]

% train loss
\addplot[thick, thwspetrol, solid, domain=0:10, samples=100]
    {0.3 + 4.2*exp(-0.5*x)};
\addlegendentry{Train}

% val loss
\addplot[thick, thwsorange, dashed, domain=0:10, samples=100]
    {1.0 + 3.0*exp(-0.6*x) + 0.15*(x-4)^2 - 0.15*16};
\addlegendentry{Val}

% vertical line at best val loss (minimum around x=4)
\addplot[thick, carnelian, dashed] coordinates {(4, 0) (4, 5)};

% dot at minimum of val loss
\addplot[only marks, mark=*, carnelian, mark size=3pt]
    coordinates {(4, 1.0 + 3.0*exp(-2.4) + 0)};

% annotation
\node[carnelian, font=\small, anchor=west] at (axis cs:4.2, 4.2)
    {best model};

\end{axis}
\end{tikzpicture}
