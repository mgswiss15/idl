% Figure A: Unrolled recurrence — hidden states only, weight sharing made explicit
\begin{tikzpicture}[
  hnode/.style  = {draw, rounded corners=4pt, minimum width=1.0cm, minimum height=0.62cm,
                   fill=thwsblue!15, draw=thwsblue, thick, font=\small},
  xnode/.style  = {draw, circle, minimum size=0.62cm,
                   fill=thwsorange!20, draw=thwsorange!80!black, thick, font=\small},
  ynode/.style  = {draw, circle, minimum size=0.62cm,
                   fill=thwsgrey!40, draw=thwsgrey!80, thick, font=\small},
  arr/.style    = {->, thick, >=stealth},
  warr/.style   = {->, thick, >=stealth, color=carnelian},
  lbl/.style    = {font=\scriptsize, midway},
]

% Hidden state nodes
\node[hnode] (h0) {$\mathbf{h}^{(0)}$};
\node[hnode, right=1.1cm of h0] (h1) {$\mathbf{h}^{(1)}$};
\node[hnode, right=1.1cm of h1] (h2) {$\mathbf{h}^{(2)}$};
\node[hnode, right=1.1cm of h2] (h3) {$\mathbf{h}^{(3)}$};

% Input nodes (below hidden states, skip h0)
\node[xnode, below=0.9cm of h1] (x1) {$\mathbf{x}^{(1)}$};
\node[xnode, below=0.9cm of h2] (x2) {$\mathbf{x}^{(2)}$};
\node[xnode, below=0.9cm of h3] (x3) {$\mathbf{x}^{(3)}$};

% Output nodes (above hidden states, skip h0)
\node[ynode, above=0.9cm of h1] (y1) {$\mathbf{y}^{(1)}$};
\node[ynode, above=0.9cm of h2] (y2) {$\mathbf{y}^{(2)}$};
\node[ynode, above=0.9cm of h3] (y3) {$\mathbf{y}^{(3)}$};

% Horizontal recurrent arrows
\draw[arr] (h0) -- node[above, lbl, color=carnelian] {$\mathbf{W}_h$} (h1);
\draw[arr] (h1) -- node[above, lbl, color=carnelian] {$\mathbf{W}_h$} (h2);
\draw[arr] (h2) -- node[above, lbl, color=carnelian] {$\mathbf{W}_h$} (h3);

% Input arrows
\draw[arr] (x1) -- node[right, lbl, color=thwsorange!80!black] {$\mathbf{W}_x$} (h1);
\draw[arr] (x2) -- node[right, lbl, color=thwsorange!80!black] {$\mathbf{W}_x$} (h2);
\draw[arr] (x3) -- node[right, lbl, color=thwsorange!80!black] {$\mathbf{W}_x$} (h3);

% Output arrows
\draw[arr] (h1) -- (y1);
\draw[arr] (h2) -- (y2);
\draw[arr] (h3) -- (y3);

% h0 initialized to zero
\node[below=0.12cm of h0, font=\scriptsize, text=thwsblue] {$= \mathbf{0}$};

\end{tikzpicture}
