% conv2d_stride.tex
% Top row: stride=1, three consecutive positions
% Bottom row: stride=2, two positions (jumps over one)
\begin{tikzpicture}[scale=0.72]

% ── TOP: stride=1 ────────────────────────────────────────────────
\node[font=\small, thwspetrol] at (-0.4, 2.8) {$s{=}1$};

\foreach \step/\xsh/\pcol in {0/0/0, 1/5/1, 2/10/2} {
    \begin{scope}[xshift=\xsh cm]
        \foreach \col in {0,...,4} {
            \foreach \row in {0,...,4} {
                \draw[thwspetrol!25, fill=thwspetrol!7]
                    (\col*0.52, \row*0.52) rectangle (\col*0.52+0.48, \row*0.52+0.48);
            }
        }
        % patch starts at col \pcol
        \foreach \col in {0,1,2} {
            \foreach \row in {2,3,4} {
                \pgfmathtruncatemacro\dc{\col+\pcol}
                \draw[thwsorange, fill=thwsorange!30, thick]
                    (\dc*0.52, \row*0.52) rectangle (\dc*0.52+0.48, \row*0.52+0.48);
            }
        }
        % output grid
        \foreach \col in {0,1,2} {
            \foreach \row in {0,1,2} {
                \draw[carnelian!30, fill=carnelian!8]
                    (\col*0.52+3.0, \row*0.52+0.52) rectangle
                    (\col*0.52+3.48, \row*0.52+1.0);
            }
        }
        % highlight active output cell
        \pgfmathsetmacro\ox{3.0+\pcol*0.52}
        \draw[carnelian, fill=carnelian!40, thick]
            (\ox, 1.56) rectangle (\ox+0.48, 2.04);
        \draw[->, thwsgrey, thin] (2.7,1.3) -- (2.95,1.3);
        % step label
        \node[font=\scriptsize, thwsorange] at (1.2,-0.3) {step \pgfmathparse{int(\step+1)}\pgfmathresult};
    \end{scope}
    % arrow between steps
    \ifnum\step<2
        \pgfmathsetmacro\ax{\xsh+4.5}
        \draw[->, thick, thwsgrey] (\ax+0.1,1.3) -- (\ax+0.4,1.3);
    \fi
}

% ── BOTTOM: stride=2 ─────────────────────────────────────────────
\node[font=\small, thwspetrol] at (-0.4, -1.0) {$s{=}2$};

\foreach \step/\xsh/\pcol in {0/0/0, 1/5/2} {
    \begin{scope}[xshift=\xsh cm, yshift=-4.0cm]
        \foreach \col in {0,...,4} {
            \foreach \row in {0,...,4} {
                \draw[thwspetrol!25, fill=thwspetrol!7]
                    (\col*0.52, \row*0.52) rectangle (\col*0.52+0.48, \row*0.52+0.48);
            }
        }
        \foreach \col in {0,1,2} {
            \foreach \row in {2,3,4} {
                \pgfmathtruncatemacro\dc{\col+\pcol}
                \draw[thwsorange, fill=thwsorange!30, thick]
                    (\dc*0.52, \row*0.52) rectangle (\dc*0.52+0.48, \row*0.52+0.48);
            }
        }
        % output grid (stride=2: 2x2 output for 5x5 input, k=3)
        \foreach \col in {0,1} {
            \foreach \row in {0,1} {
                \draw[carnelian!30, fill=carnelian!8]
                    (\col*0.52+3.0, \row*0.52+0.78) rectangle
                    (\col*0.52+3.48, \row*0.52+1.26);
            }
        }
        % highlight active output
        \pgfmathtruncatemacro\oc{int(\pcol/2)}
        \pgfmathsetmacro\ox{3.0+\oc*0.52}
        \draw[carnelian, fill=carnelian!40, thick]
            (\ox, 1.3) rectangle (\ox+0.48, 1.78);
        \draw[->, thwsgrey, thin] (2.7,1.3) -- (2.95,1.3);
        \node[font=\scriptsize, thwsorange] at (1.2,-0.3) {step \pgfmathparse{int(\step+1)}\pgfmathresult};
    \end{scope}
    \ifnum\step<1
        \pgfmathsetmacro\ax{\xsh+3.6}
        \draw[->, thick, thwsgrey] (\ax+1.0,-2.7) -- (\ax+1.3,-2.7);
        \node[font=\scriptsize, thwsorange] at (\ax+2.3,-1.2) {skip 1 pixel};
    \fi
}

% output size annotations
\node[font=\scriptsize, carnelian] at (13.9, 2.3) {output $3\times3$};
\node[font=\scriptsize, carnelian] at (8.7, -1.9) {output $2\times2$};

\end{tikzpicture}
