% Figure E: BPTT — hidden states only, backward gradient arrows
\begin{tikzpicture}[
  hnode/.style = {draw, rounded corners=4pt, minimum width=1.0cm, minimum height=0.65cm,
                  fill=thwsblue!15, draw=thwsblue, thick, font=\small},
  xnode/.style = {draw, circle, minimum size=0.62cm,
                  fill=thwsorange!25, draw=thwsorange!80!black, thick, font=\scriptsize},
  lossnode/.style = {draw, rounded corners=4pt, minimum width=0.9cm, minimum height=0.65cm,
                     fill=carnelian!15, draw=carnelian, thick, font=\small},
  farr/.style  = {->, >=stealth, thick, color=thwsblue!60},
  barr/.style  = {->, >=stealth, thick, color=carnelian, dashed},
  lbl/.style   = {font=\scriptsize, midway},
]

\def\step{1.8}

% Hidden state nodes
\node[hnode] (h0) at (0, 0)          {$\mathbf{h}^{(0)}$};
\node[hnode] (h1) at (1*\step, 0)    {$\mathbf{h}^{(1)}$};
\node[hnode] (h2) at (2*\step, 0)    {$\mathbf{h}^{(2)}$};
\node[hnode] (h3) at (3*\step, 0)    {$\mathbf{h}^{(3)}$};

\node[font=\tiny, below=0.1cm of h0, color=thwsblue] {$=\mathbf{0}$};

% Input nodes
\node[xnode] (x1) at (1*\step, -1.1) {$\mathbf{x}^{(1)}$};
\node[xnode] (x2) at (2*\step, -1.1) {$\mathbf{x}^{(2)}$};
\node[xnode] (x3) at (3*\step, -1.1) {$\mathbf{x}^{(3)}$};

% Forward arrows
\draw[farr] (h0) -- node[above, lbl, color=thwsblue!80] {$\mathbf{W}_h$} (h1);
\draw[farr] (h1) -- node[above, lbl, color=thwsblue!80] {$\mathbf{W}_h$} (h2);
\draw[farr] (h2) -- node[above, lbl, color=thwsblue!80] {$\mathbf{W}_h$} (h3);
\draw[farr] (x1) -- (h1);
\draw[farr] (x2) -- (h2);
\draw[farr] (x3) -- (h3);

% Loss node
\node[lossnode] (L) at (3*\step+1.5, 0) {$\mathcal{L}$};
\draw[farr] (h3) -- (L);

% Backward arrows (above the forward path)
\draw[barr] (L.north)  to[out=120, in=60]
            node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(3)}}{\partial \mathbf{h}^{(2)}}$} (h3.north);
\draw[barr] (h3.north) to[out=150, in=30]
            node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(3)}}{\partial \mathbf{h}^{(2)}}$} (h2.north);
\draw[barr] (h2.north) to[out=150, in=30]
            node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(2)}}{\partial \mathbf{h}^{(1)}}$} (h1.north);
\draw[barr] (h1.north) to[out=150, in=30]
            node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(1)}}{\partial \mathbf{h}^{(0)}}$} (h0.north);

\end{tikzpicture}
