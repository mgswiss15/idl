% Figure E: BPTT — unrolled graph with backward gradient arrows
\begin{tikzpicture}[
  hnode/.style = {draw, rounded corners=4pt, minimum width=1.0cm, minimum height=0.65cm,
                  fill=thwsblue!15, draw=thwsblue, thick, font=\small},
  cell/.style  = {draw, rounded corners=5pt, minimum width=0.95cm, minimum height=0.65cm,
                  fill=thwspetrol!15, draw=thwspetrol, thick, font=\scriptsize},
  xnode/.style = {draw, circle, minimum size=0.62cm,
                  fill=thwsorange!25, draw=thwsorange!80!black, thick, font=\scriptsize},
  lossnode/.style = {draw, rounded corners=4pt, minimum width=0.9cm, minimum height=0.65cm,
                     fill=carnelian!15, draw=carnelian, thick, font=\small},
  farr/.style  = {->, >=stealth, thick, color=thwsblue!60},
  barr/.style  = {->, >=stealth, thick, color=carnelian, dashed},
]

\def\step{2.6}

\node[hnode] (h0) at (0,0) {$\mathbf{h}^{(0)}$};
\node[font=\tiny, below=0.1cm of h0, color=thwsblue] {$=\mathbf{0}$};

\foreach \t in {1,2,3} {
  \pgfmathsetmacro\xpos{\t*\step}
  \node[cell]  (f\t) at (\xpos, 0)    {$\tanh$};
  \node[hnode] (h\t) at (\xpos+1.1, 0) {$\mathbf{h}^{(\t)}$};
  \node[xnode] (x\t) at (\xpos, -1.3) {$\mathbf{x}^{(\t)}$};
  \draw[farr] (x\t) -- (f\t);
  \draw[farr] (f\t) -- (h\t);
}
\draw[farr] (h0) -- (f1);
\draw[farr] (h1) -- (f2);
\draw[farr] (h2) -- (f3);

% Loss node after h3
\node[lossnode] (L) at (3*\step+1.1+1.8, 0) {$\mathcal{L}$};
\draw[farr] (h3) -- (L);

% Backward arrows (above the forward path)
\draw[barr] (L.north)  to[out=120, in=60] node[above, font=\tiny, color=carnelian] {} (h3.north);
\draw[barr] (h3.north) to[out=150, in=30] node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(3)}}{\partial \mathbf{h}^{(2)}}$} (h2.north);
\draw[barr] (h2.north) to[out=150, in=30] node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(2)}}{\partial \mathbf{h}^{(1)}}$} (h1.north);
\draw[barr] (h1.north) to[out=150, in=30] node[above, font=\tiny, color=carnelian]
            {$\frac{\partial \mathbf{h}^{(1)}}{\partial \mathbf{h}^{(0)}}$} (h0.north);

% Product annotation
\node[align=center, font=\scriptsize, color=carnelian] at (5.2, -2.2)
  {$\dfrac{\partial \mathcal{L}}{\partial \mathbf{W}_h} \;\propto\;
    \displaystyle\prod_{t=1}^{T} \frac{\partial \mathbf{h}^{(t)}}{\partial \mathbf{h}^{(t-1)}}
    \quad \longleftarrow$ product of $T$ Jacobians};

\end{tikzpicture}
