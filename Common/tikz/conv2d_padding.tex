% conv2d_padding.tex
% Left: no padding, 3x3 input, 3x3 filter -> 1x1 output
% Right: same padding (1), 3x3 input + zeros -> 5x5, 3x3 filter -> 3x3 output
\begin{tikzpicture}[scale=0.72]

% ── LEFT: no padding ──────────────────────────────────────────────
\begin{scope}[xshift=0cm, yshift=0.5cm]
    % 3x3 input
    \foreach \col in {0,1,2} {
        \foreach \row in {0,1,2} {
            \draw[thwspetrol, fill=thwspetrol!35, thick]
                (\col*0.52, \row*0.52) rectangle (\col*0.52+0.48, \row*0.52+0.48);
        }
    }
    % filter covers entire input
    \foreach \col in {0,1,2} {
        \foreach \row in {0,1,2} {
            \draw[thwsorange, fill=thwsorange!30, thick]
                (\col*0.52+0.06, \row*0.52+0.06) rectangle
                (\col*0.52+0.44, \row*0.52+0.44);
        }
    }
    % 1x1 output
    \draw[carnelian, fill=carnelian!40, thick]
        (2.3, 0.52) rectangle (2.78, 0.52+0.48);
    % labels
    \node[font=\scriptsize, thwspetrol] at (0.76, -0.3) {input $3\times3$};
    \node[font=\scriptsize, carnelian]  at (2.6, 0.3) {out $1\times1$};
    \node[font=\small, thwspetrol] at (0.76, 2.1) {no padding};
    % output size annotation
    \draw[->, thwsgrey, thin] (1.85, 0.76) -- (2.15, 0.76);
\end{scope}

% ── RIGHT: same padding (p=1) ────────────────────────────────────
\begin{scope}[xshift=4.5cm]
    % zero padding ring (5x5)
    \foreach \col in {0,...,4} {
        \foreach \row in {0,...,4} {
            \draw[thwsgrey!30, fill=thwsgrey!50]
                (\col*0.52, \row*0.52) rectangle (\col*0.52+0.48, \row*0.52+0.48);
            \node[font=\tiny, white] at (\col*0.52+0.24, \row*0.52+0.24) {0};
        }
    }
    % original 3x3 input (center)
    \foreach \col in {1,2,3} {
        \foreach \row in {1,2,3} {
            \draw[thwspetrol, fill=thwspetrol!35, thick]
                (\col*0.52, \row*0.52) rectangle (\col*0.52+0.48, \row*0.52+0.48);
        }
    }
    % filter at top-left position (cols 0-2, rows 2-4)
    \foreach \col in {0,1,2} {
        \foreach \row in {2,3,4} {
            \draw[thwsorange, fill=thwsorange!30, thick]
                (\col*0.52+0.04, \row*0.52+0.04) rectangle
                (\col*0.52+0.46, \row*0.52+0.46);
        }
    }
    % 3x3 output
    \foreach \col in {0,1,2} {
        \foreach \row in {0,1,2} {
            \draw[carnelian!30, fill=carnelian!8]
                (\col*0.52+3.2, \row*0.52+0.52) rectangle
                (\col*0.52+3.68, \row*0.52+1.0);
        }
    }
    \draw[carnelian, fill=carnelian!40, thick]
        (3.2, 2*0.52+0.52) rectangle (3.68, 2*0.52+1.00);
    \draw[->, thwsgrey, thin] (2.7, 1.8) -- (3.15, 1.8);
    % labels
    \node[font=\scriptsize, thwspetrol]   at (1.2, -0.3) {padded $5\times5$};
    \node[font=\scriptsize, carnelian]  at (4., 0.3) {out $3\times3$};
    \node[font=\small, thwspetrol] at (1.2, 2.8) {same padding ($p=1$)};
\end{scope}

\end{tikzpicture}
