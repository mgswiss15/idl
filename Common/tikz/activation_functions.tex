% Activation functions: ReLU, LeakyReLU, ELU
\begin{tikzpicture}
\begin{axis}[
    width=6.5cm, height=5cm,
    xlabel={$z$},
    ylabel={$a$},
    xmin=-3, xmax=3,
    ymin=-3, ymax=2,
    xtick={-2, -1, 0, 1, 2},
    ytick={-1, 0, 1, 2},
    grid=major,
    grid style={thwsgrey!40},
    legend pos=south east,
    legend style={font=\scriptsize, draw=none},
    axis lines=center,
    tick label style={font=\small},
]

% ReLU
\addplot[thick, thwspetrol, solid, domain=-3:0, samples=2] {0};
\addlegendentry{ReLU}
\addplot[thick, thwspetrol, solid, domain=0:3, samples=2, forget plot] {x};

% LeakyReLU (alpha = 0.1)
\addplot[thick, thwsorange, dashed, domain=-3:0, samples=2] {0.1*x};
\addlegendentry{LeakyReLU}
\addplot[thick, thwsorange, dashed, domain=0:3, samples=2, forget plot] {x};

% ELU (alpha = 1)
\addplot[thick, carnelian, dotted, domain=-3:0, samples=60] {exp(x)-1};
\addlegendentry{ELU}
\addplot[thick, carnelian, dotted, domain=0:3, samples=2, forget plot] {x};

\end{axis}
\end{tikzpicture}
