\begin{tikzpicture}
\begin{axis}[
    width=7cm, height=6.5cm,
    view={40}{35},
    axis lines=none,
    ztick=\empty, xtick=\empty, ytick=\empty,
    colormap={thwsmap}{
        color(0)=(thwsorange)
        color(0.5)=(lightblue)
        color(1)=(thwspetrol)
    },
    /pgf/number format/1000 sep={},
    samples=30,
    domain=-2.2:2.2,
    y domain=-2.2:2.2,
    zmin=-0.3,
]

% Light base plane
\addplot3[surf, shader=flat, fill=thwsgrey!40, draw=none,
    samples=2, z filter/.code={\def\pgfmathresult{-0.25}}] {0};

% Loss surface
\addplot3[surf, opacity=0.85, shader=interp, z buffer=sort] {
    0.5*(x^2 + y^2)
    - 1.1*exp(-((x-0.9)^2 + (y-0.9)^2)*2.5)
    - 0.7*exp(-((x+0.8)^2 + (y+0.6)^2)*3.0)
    + 0.3
};

% Contours projected on base plane - global min region
\foreach \r in {0.28, 0.55, 0.90, 1.35}{
    \addplot3[thin, thwspetrol!70, domain=0:360, samples=36]
        ({0.9 + \r*cos(x)}, {0.9 + 0.88*\r*sin(x)}, -0.25);
}
% Contours - local min region
\foreach \r in {0.22, 0.48, 0.78}{
    \addplot3[thin, thwspetrol!50, domain=0:360, samples=36]
        ({-0.8 + \r*cos(x)}, {-0.6 + 0.88*\r*sin(x)}, -0.25);
}

\end{axis}
\end{tikzpicture}
