\[ \vh = \sigma\!\left(\thetamat{}\vx + \boldsymbol{\theta}_0\right) \]

\begin{frame}{Layer Notation}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Index layers} $l = 0, 1, \ldots, L$

\vskip 0.8em

\structure{Layer 0:} input
\[ \va^{(0)} = \vx \]

\structure{Hidden layers} $l = 1, \ldots, L-1$:
\begin{align*}
\vz^{(l)} &= \thetamat{l} \va^{(l-1)} + \thetavec{l}_0 & &\text{pre-activation}\\
\va^{(l)} &= \sigma\!\left(\vz^{(l)}\right) & &\text{activation}
\end{align*}

\structure{Output layer} $l = L$:
\begin{align*}
\vz^{(L)} &= \thetamat{L} \va^{(L-1)} + \thetavec{L}_0 \\
\yhat &= g\!\left(\vz^{(L)}\right)
\end{align*}
$g$: output activation (task-dependent)

\end{column}

\centering
\input{\CommonPath/tikz/mlp_layer_notation.tex}
\vskip 1em
\structure{Forward pass:} evaluate layer by layer, $l = 1 \to L$
\end{column}

\end{column}

\begin{frame}{Batch Formulation}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Single sample} $\vx \in \R^d$:
\[ \vz^{(l)} = \thetamat{l} \va^{(l-1)} + \thetavec{l}_0 \]

\vskip 1em

\structure{Batch of $N$ samples} --- stack as rows:
\[ \mathbf{X} \in \R^{N \times d}, \quad \mathbf{A}^{(l)} \in \R^{N \times k_l} \]

\[ \mathbf{Z}^{(l)} = \mathbf{A}^{(l-1)} \left(\thetamat{l}\right)\T + \mathbf{1}\thetavec{l}_0{}\T \]

\vskip 0.5em

\begin{frame}{The Learning Problem}

\vskip 2em

\begin{columns}[c]
\begin{column}{0.48\textwidth}
\structure{What we have:}
\begin{itemize}
\item Network $f_\theta$ with forward pass \checkmark

\end{itemize}

\structure{Two open questions:}
\begin{enumerate}
\item How do we \textbf{measure} how wrong $f_\theta$ is? \\ \alert{$\Rightarrow$ loss function}
\item How do we \textbf{find} good $\pvec$ systematically? \\ \alert{$\Rightarrow$ optimization}
\end{enumerate}
\end{column}

\section{Loss Functions}
% -------------------------------------------------------

\begin{frame}{What is a Loss Function?}

\vskip 1em

\structure{Goal:} measure how wrong $f_\theta$ is with a single scalar

\vskip 1.5em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Per-sample loss} $\ell(\yhat, y)$:
\begin{itemize}
\item $\ell(\yhat, y) = 0$ when $\yhat = y$ (perfect)
\item $\ell(\yhat, y) > 0$ otherwise
\item Larger when more wrong
\end{itemize}

\vskip 1em

\structure{Dataset loss} $\loss(\pvec)$:
\[ \loss(\pvec) = \frac{1}{\nsamples} \sum_{i=1}^{\nsamples} \ell\!\left(f_\theta(\vx^{(i)}), y^{(i)}\right) \]
Average over all training samples
\end{column}
\begin{column}{0.48\textwidth}
\structure{Choice of $\ell$ depends on the task:}

\vskip 0.8em

\begin{tabular}{ll}
\structure{Task} & \structure{Loss} \\
\hline\\[-0.8em]
Regression & Mean Squared Error \\[0.3em]
Binary classification & Binary Cross-Entropy \\[0.3em]
Multiclass & Cross-Entropy \\
\end{tabular}

\vskip 1.5em
\alert{The loss is the compass --- it tells the model which direction is ``better''.}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Regression: Mean Squared Error}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Setting:} $y \in \R$, $\yhat = \vz^{(L)} \in \R$ (no output activation)

\vskip 1em

\structure{Mean Squared Error (MSE):}
\[ \ell(\yhat, y) = (\yhat - y)^2 \]
\[ \loss(\pvec) = \frac{1}{\nsamples}\sum_{i=1}^{\nsamples}\!\left(\yhat^{(i)} - y^{(i)}\right)^2 \]

\vskip 1em

\structure{Intuition:}
\begin{itemize}
\item Penalizes large errors quadratically
\item Errors of opposite sign cancel in average --- use absolute value or square
\item Squaring also makes $\ell$ differentiable
\end{itemize}
\end{column}
\begin{column}{0.44\textwidth}
\centering
\input{\CommonPath/tikz/mse_loss.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Classification: Probability Output}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Setting:} $y \in \{0, 1\}$, we want $f_\theta(\vx) \in (0,1)$

\vskip 0.8em

\structure{Problem:} raw network output $z^{(L)} \in \R$ --- not a probability

\vskip 0.8em

\structure{Solution:} apply \textbf{sigmoid} as output activation
\[ \yhat = \sigma(z^{(L)}) = \frac{1}{1 + e^{-z^{(L)}}} \in (0,1) \]

\vskip 0.8em

\structure{Interpretation:} $\yhat = P(y = 1 \mid \vx)$

\vskip 0.8em

\structure{Decision:} predict class 1 if $\yhat \geq 0.5$, else class 0
\end{column}

\end{columns}

\end{frame}

\begin{frame}{Binary Classification: Binary Cross-Entropy}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Binary Cross-Entropy (BCE):}
\[
\ell(\yhat, y) = -y \log \yhat - (1-y)\log(1-\yhat)
\]
\[ \loss(\pvec) = -\frac{1}{\nsamples}\sum_{i=1}^{\nsamples}\!\left[y^{(i)}\log\yhat^{(i)} + (1-y^{(i)})\log(1-\yhat^{(i)})\right] \]

\vskip 0.8em

\structure{Intuition:}
\begin{itemize}
\item When $y=1$: loss $= -\log\yhat$ --- penalizes low confidence in correct class
\item When $y=0$: loss $= -\log(1-\yhat)$ --- penalizes high confidence in wrong class
\item \alert{Confident wrong predictions are punished most severely}
\end{itemize}
\end{column}
\begin{column}{0.44\textwidth}
\centering
\input{\CommonPath/tikz/bce_loss.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Multiclass Classification: Softmax + Cross-Entropy}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Setting:} $y \in \{1,\ldots,C\}$, output $\vz^{(L)} \in \R^C$

\vskip 0.8em

\structure{Softmax} converts logits to probabilities:
\[ \yhat_c = \frac{e^{z^{(L)}_c}}{\sum_{c'=1}^{C} e^{z^{(L)}_{c'}}}, \quad c = 1,\ldots,C \]
\begin{itemize}
\item $\yhat_c \in (0,1)$, \quad $\sum_c \yhat_c = 1$
\item Amplifies the largest logit
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Cross-Entropy loss:}
\[ \ell(\yhat, y) = -\sum_{c=1}^{C} \mathds{1}[y=c]\, \log \yhat_c \]
\vskip 0.3em
With one-hot label: only the true class contributes
\[ \ell(\yhat, y) = -\log \yhat_y \]

\vskip 0.8em

\structure{Intuition:} maximize the predicted probability of the correct class
\end{column}
\end{columns}

\vskip 1em
\centering
\input{\CommonPath/tikz/softmax.tex}

\end{frame}

\section{Gradient Descent}
% -------------------------------------------------------

\begin{frame}{The Optimization Problem}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Goal:} find parameters that minimize the loss
\[ \pvec^* = \arg\min_{\pvec}\ \loss(\pvec) \]

\vskip 0.8em

\structure{Why not solve analytically?}
\begin{itemize}
\item $\loss(\pvec)$ is \textbf{nonlinear} in $\pvec$ --- no closed form
\item $\pvec$ has millions of dimensions in practice
\item $\loss(\pvec)$ is \textbf{non-convex} --- many local minima
\end{itemize}

\vskip 0.8em

\structure{Strategy:} iterative, gradient-based optimization
\end{column}
\begin{column}{0.44\textwidth}
\centering
\input{\CommonPath/tikz/loss_surface.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Gradient Descent}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Gradient} $\nabla_{\pvec} \loss(\pvec)$:
\begin{itemize}
\item Points in direction of \textbf{steepest ascent}
\item One partial derivative per parameter: $\frac{\partial \loss}{\partial \theta_j}$
\end{itemize}

\vskip 1em

\structure{Gradient descent update:}
\[
\pvec \leftarrow \pvec - \eta\, \nabla_{\pvec} \loss(\pvec)
\]
\begin{itemize}
\item Step \textbf{opposite} to gradient $\Rightarrow$ go downhill
\item $\eta > 0$: \textbf{learning rate} --- controls step size
\item Repeat until convergence
\end{itemize}

\vskip 0.8em

\structure{Key question:} how do we compute $\nabla_{\pvec} \loss(\pvec)$?
\end{column}
\begin{column}{0.44\textwidth}
\centering
\input{\CommonPath/tikz/gradient_descent_2d.tex}
\end{column}
\begin{column}{0.04\textwidth}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Walking Down a Hill}

\vskip 1em

\begin{columns}[c]
\begin{column}{0.52\textwidth}
\structure{Intuition:} a hiker descending a mountain in fog
\begin{itemize}
\item Cannot see the whole landscape --- only local slope
\item Step in the direction that goes downhill
\item Repeat until no more descent
\end{itemize}

\vskip 1em

\structure{Effect of learning rate $\eta$:}
\begin{itemize}
\item $\eta$ too large: \alert{overshoots}, may diverge
\item $\eta$ too small: very slow convergence
\item $\eta$ just right: steady descent
\end{itemize}

\vskip 1em

\structure{Complication:} loss surface has many \textbf{valleys} (local minima) --- the hiker may get stuck
\end{column}

\section{Backpropagation}
% -------------------------------------------------------

\begin{frame}{Forward and Backward Pass}

\vskip 2em

\structure{Forward pass} --- compute prediction and loss:

\vskip 0.5em
\centering
\input{\CommonPath/tikz/compgraph\_forward\_backward.tex}

\vskip 1.5em

\begin{columns}[c]
\begin{column}{0.48\textwidth}
\centering
\structure{Forward pass}\\[0.3em]
$\vx \longrightarrow \va^{(1)} \longrightarrow \cdots \longrightarrow \va^{(L)} \longrightarrow \ell$
\end{column}
\begin{column}{0.48\textwidth}
\centering
\structure{Backward pass}\\[0.3em]
$\dfrac{\partial \ell}{\partial \pvec} \longleftarrow \cdots \longleftarrow \dfrac{\partial \ell}{\partial \va^{(L)}} \longleftarrow \ell$
\end{column}
\end{columns}

\vskip 1em
\centering
\alert{\textbf{Backpropagation} = computing the backward pass efficiently}

\end{frame}

% -------------------------------------------------------

\begin{frame}{Two Key Rules}

\vskip 1em

\structure{Goal:} compute $\nabla_{\pvec}\, \loss(\pvec)$ for the full dataset

\vskip 1.5em

\structure{1) Sum rule} --- gradient of average = average of gradients:
\[
\nabla_{\pvec}\, \loss(\pvec)
= \nabla_{\pvec} \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \ell(\vx^{(i)}, y^{(i)}, \pvec)
= \frac{1}{\nsamples}\sum_{i=1}^{\nsamples} \nabla_{\pvec}\, \ell(\vx^{(i)}, y^{(i)}, \pvec)
\]

\vskip 0.5em
\alert{$\Rightarrow$ compute gradient for a single sample, then average}

\vskip 1.5em

\structure{2) Chain rule} --- THE rule for backprop:
\[
\frac{d}{dx} f(g(x)) = \frac{df}{dg}\frac{dg}{dx}
\qquad
\frac{d}{dx} f(g_1(x), \ldots, g_m(x)) = \sum_{i=1}^{m} \frac{\partial f}{\partial g_i}\frac{dg_i}{dx}
\]

\end{frame}

% -------------------------------------------------------

\begin{frame}{Local vs Global Gradients}

\vskip 0.5em

\centering
\input{\CommonPath/tikz/compgraph\_backprop\_local.tex}

\vskip 1em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Local gradient} (easy --- per node):
\[ \frac{\partial\, \text{out}}{\partial\, \text{in}} \]
Each node computes its own derivative
\end{column}
\begin{column}{0.48\textwidth}
\structure{Global gradient} (chain rule):
\[ \frac{\partial \ell}{\partial\, \text{in}} = \frac{\partial \ell}{\partial\, \text{out}} \cdot \frac{\partial\, \text{out}}{\partial\, \text{in}} \]
Multiply incoming global gradient by local gradient
\end{column}
\end{columns}

\end{frame}

{\small
(1) $h = \theta_1 x$\\

(4) $\ell = z^2$
}
\end{column}
\begin{column}{0.32\textwidth}
\structure{\color{thwspetrol}Local gradients}\\[0.3em]
{\small
(1) $\partial h/\partial \theta_1 = x$\\
(2) $\partial \yhat/\partial \theta_0 = 1$\\
\phantom{(2)} $\partial \yhat/\partial h = 1$\\
(3) $\partial z/\partial \yhat = 1$\\
(4) $\partial \ell/\partial z = 2z$
}
\end{column}
\begin{column}{0.36\textwidth}
\structure{\color{thwsorange}Global gradients}\\[0.3em]
{\small
(1) $\partial \ell/\partial \theta_1 = \partial \ell/\partial h \cdot x = 2zx$\\
(3) $\partial \ell/\partial \theta_0 = \partial \ell/\partial \yhat \cdot 1 = 2z$\\
\phantom{(3)} $\partial \ell/\partial h = \partial \ell/\partial \yhat \cdot 1 = 2z$\\
(2) $\partial \ell/\partial \yhat = \partial \ell/\partial z \cdot 1 = 2z$\\
(1) $\partial \ell/\partial z = 1 \cdot 2z = 2z$
}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Technical Implementation}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.35\textwidth}
Each node stores:
\begin{itemize}
\item \texttt{\color{thwspetrol}.val} --- forward value
\item \texttt{\color{thwsorange}.grad} --- accumulated gradient
\end{itemize}
\vskip 0.5em
Each compute node implements:
\begin{itemize}
\item Forward function
\item Local gradient function
\end{itemize}
\end{column}
\begin{column}{0.30\textwidth}
\structure{Forward pass}\\[0.3em]
{\scriptsize\ttfamily
x.val, y.val = data\\
t0.val, t1.val = init\\
h.val = t1.val * x.val\\
yp.val = t0.val + h.val\\
z.val = yp.val - y.val\\
l.val = z.val ** 2
}
\end{column}
\begin{column}{0.35\textwidth}
\structure{Backward pass}\\[0.3em]
{\scriptsize\ttfamily
l.grad = 1\\
z.grad = l.grad * 2*z.val\\
yp.grad = z.grad * 1\\
t0.grad = yp.grad * 1\\
h.grad = yp.grad * 1\\
t1.grad = h.grad * x.val
}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{From For Loops to Vectorization}

\vskip 0.5em

\begin{columns}[t]
\begin{column}{0.48\textwidth}
\structure{Elementwise for loops}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
for i: z.val[i] = b.val[i]\\
for i,j: z.val[j] += W.val[i,j] * x.val[i]\\
for i: a.val[i] = act(z.val[i])\\[0.5em]
\# Backward\\
for i: z.grad[i] = a.grad[i] * dact(z.val[i])\\
for i,j: W.grad[i,j] = z.grad[j] * x.val[i]\\
for i: b.grad[i] = z.grad[i]
}
\end{column}
\begin{column}{0.48\textwidth}
\structure{Vectorized}\\[0.3em]
{\scriptsize\ttfamily
\# Forward\\
z.val = W.val @ x.val + b.val\\
a.val = act(z.val)\\[0.5em]
\# Backward\\
z.grad = a.grad * dact(z.val)\\
W.grad = z.grad[:,None] * x.val[None,:]\\
b.grad = z.grad
}
\vskip 1em
\alert{Same computation, $\sim$100$\times$ faster on GPU}
\end{column}
\end{columns}

\end{frame}
