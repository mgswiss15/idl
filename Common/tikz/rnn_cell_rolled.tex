\begin{tikzpicture}[
  hnode/.style = {draw, rounded corners=4pt, minimum width=1.1cm, minimum height=0.65cm,

  xnode/.style = {draw, circle, minimum size=0.65cm,
                  fill=thwsorange!25, draw=thwsorange!80!black, thick, font=\small},
  ynode/.style = {draw, circle, minimum size=0.65cm,
                  fill=thwsgrey!40, draw=thwsgrey!80, thick, font=\small},

  lbl/.style   = {font=\scriptsize, midway},
]

\node[hnode] (hprev) {$\mathbf{h}^{(t-1)}$};

\draw[arr] (hprev) -- node[above, lbl, color=carnelian] {$\mathbf{W}_h$} (hnext);

% Continuation arrow to the right
