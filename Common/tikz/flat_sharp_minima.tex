% Flat vs sharp minima
\begin{tikzpicture}
\begin{axis}[
    width=6.5cm, height=5cm,
    xlabel={$\theta$},
    ylabel={$\mathcal{L}$},
    xmin=-5, xmax=8,
    ymin=0, ymax=4,
    xtick=\empty,
    ytick=\empty,
    label style={font=\small},
    axis lines=left,
    clip=true,
]

% sharp minimum at x=-1.5 — narrow deep parabola
\addplot[thick, carnelian, solid, domain=-5:1.5, samples=150]
    {0.15 + 3.0*(x+1.5)^2};

% flat minimum at x=5 — very gentle wide parabola
\addplot[thick, thwspetrol, solid, domain=0:8, samples=150]
    {0.15 + 0.08*(x-5)^2};

% sharp label — above minimum
\node[font=\scriptsize, carnelian, anchor=south] at (axis cs:-1.5, 3.6)
    {sharp};
\draw[->, carnelian, thin] (axis cs:-1.5, 3.5) -- (axis cs:-1.5, 0.4);

% flat label — above minimum
\node[font=\scriptsize, thwspetrol, anchor=south] at (axis cs:5.0, 3.6)
    {flat};
\draw[->, thwspetrol, thin] (axis cs:5.0, 3.5) -- (axis cs:5.0, 0.4);

\end{axis}
\end{tikzpicture}
