\begin{tikzpicture}

    width=7cm, height=5.5cm,
    xlabel={$\theta$},
    ylabel={$\mathcal{L}$},
    xmin=-4, xmax=6,
    ymin=-0.5, ymax=4,
    xtick=\empty,
    ytick=\empty,
    tick label style={font=\small},
    clip=false,

% single curve with sharp minimum (left) and flat minimum (right)

    {0.15 + 0.5*(x+1.5)^2/(1 + 5*exp(-2*(x+1.5)^2))
     + 2.5*exp(-8*(x+1.5)^2)
     + 0.12*(x-3.5)^2};

% sharp minimum label
\node[carnelian, font=\small, anchor=north] at (axis cs:-1.5, -0.3)
    {sharp};
\draw[carnelian, <->] (axis cs:-2.1, 0.15) -- (axis cs:-0.9, 0.15);

% flat minimum label
\node[thwspetrol, font=\small, anchor=north] at (axis cs:3.5, -0.3)
    {flat};
\draw[thwspetrol, <->] (axis cs:1.5, 0.15) -- (axis cs:5.5, 0.15);

\end{axis}
