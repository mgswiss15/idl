% Full CNN architecture: input -> conv blocks -> GAP/flatten -> FC head
% Feature maps shown as 3D volumes, shrinking spatially, growing in channels

\begin{tikzpicture}[scale=0.85, every node/.style={font=\scriptsize}]

\def\ox{0.18}  % 3D x offset
\def\oy{0.18}  % 3D y offset

% ── helper: draw 3D feature map volume ───────────────────────────
% \fmvol{x}{y}{w}{h}{d}{fillcolor}{bordercolor}
% w=width (channels shown as depth), h=height, d=depth offset scale

% ── INPUT: 3x32x32 ───────────────────────────────────────────────
% front face
\draw[thwspetrol, fill=thwspetrol!15]
    (0, 0) rectangle (0.45, 2.0);
% top face
\draw[thwspetrol, fill=thwspetrol!25]
    (0, 2.0) -- (0+\ox, 2.0+\oy) -- (0.45+\ox, 2.0+\oy) -- (0.45, 2.0) -- cycle;
% right face
\draw[thwspetrol, fill=thwspetrol!20]
    (0.45, 0) -- (0.45+\ox, \oy) -- (0.45+\ox, 2.0+\oy) -- (0.45, 2.0) -- cycle;
\node[thwspetrol, anchor=north] at (0.35, -0.1) {$3{\times}32{\times}32$};

% ── ARROW ────────────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (0.45+\ox+0.1, 1.0) -- (1.4, 1.0);

% ── CONV BLOCK 1: 32x16x16 ───────────────────────────────────────
\draw[thwspetrol, fill=thwspetrol!25]
    (1.5, 0.2) rectangle (1.85, 1.8);
\draw[thwspetrol, fill=thwspetrol!35]

\draw[thwspetrol, fill=thwspetrol!30]
    (1.85, 0.2) -- (1.85+\ox, 0.2+\oy) -- (1.85+\ox, 1.8+\oy) -- (1.85, 1.8) -- cycle;
\node[thwspetrol, anchor=north] at (1.85, -0.1) {$32{\times}16{\times}16$};
\node[thwsgrey, anchor=south] at (1.85, 2.05) {\tiny Conv-BN-ReLU-Pool};

% ── ARROW ────────────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (1.85+\ox+0.1, 1.0) -- (2.8, 1.0);

% ── CONV BLOCK 2: 64x8x8 ─────────────────────────────────────────
\draw[thwspetrol, fill=thwspetrol!35]
    (2.9, 0.4) rectangle (3.35, 1.6);
\draw[thwspetrol, fill=thwspetrol!45]
    (2.9, 1.6) -- (2.9+\ox, 1.6+\oy) -- (3.35+\ox, 1.6+\oy) -- (3.35, 1.6) -- cycle;
\draw[thwspetrol, fill=thwspetrol!40]
    (3.35, 0.4) -- (3.35+\ox, 0.4+\oy) -- (3.35+\ox, 1.6+\oy) -- (3.35, 1.6) -- cycle;
\node[thwspetrol, anchor=north] at (3.25, -0.1) {$64{\times}8{\times}8$};
\node[thwsgrey, anchor=south] at (3.25, 1.85) {\tiny Conv-BN-ReLU-Pool};

% ── ARROW ────────────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (3.35+\ox+0.1, 1.0) -- (4.3, 1.0);

% ── CONV BLOCK 3: 128x4x4 ────────────────────────────────────────
\draw[thwspetrol, fill=thwspetrol!50]
    (4.4, 0.55) rectangle (4.95, 1.45);
\draw[thwspetrol, fill=thwspetrol!60]

\draw[->, thick, thwsgrey] (4.95+\ox+0.1, 1.0) -- (5.8, 1.0);
\node[thwsgrey, anchor=south] at (5.4, 1.05) {\tiny GAP};

% ── VECTOR 128x1x1 ───────────────────────────────────────────────
\draw[thwspetrol, fill=thwspetrol!55, thick]

\node[thwspetrol, anchor=north] at (6.0, -0.1) {$128$};

\draw[->, thick, thwsgrey] (6.1, 1.0) -- (6.5, 1.0);

% ── FC 256 ───────────────────────────────────────────────────────

    (6.55, 0.45) rectangle (6.75, 1.55);
\node[carnelian, anchor=north] at (6.65, -0.1) {FC 256};

% ── ARROW ────────────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (6.75, 1.0) -- (7.15, 1.0);

% ── FC 10 ────────────────────────────────────────────────────────
\draw[carnelian, fill=carnelian!35, thick]
    (7.2, 0.6) rectangle (7.4, 1.4);
\node[carnelian, anchor=north] at (7.3, -0.1) {FC 10};

% ── OUTPUT ARROW ─────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (7.4, 1.0) -- (7.8, 1.0);
\node[carnelian, anchor=west] at (7.82, 1.0) {softmax};

\draw[decorate, decoration={brace, amplitude=4pt}, thwspetrol]

\node[thwspetrol, anchor=north] at (2.6, -0.75) {conv blocks --- feature extraction};

\draw[decorate, decoration={brace, amplitude=4pt}, carnelian]
    (6.5, -0.55) -- (7.45, -0.55);
\node[carnelian, anchor=north] at (6.97, -0.75) {FC head};
