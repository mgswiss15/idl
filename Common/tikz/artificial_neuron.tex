\begin{tikzpicture}[scale=0.85]
\tikzstyle{every node}=[font=\scriptsize]
\tikzstyle{neuron} = [circle, draw, fill=thwsorange!20, minimum size=0.8cm, outer sep=3pt]
\tikzstyle{input}  = [circle, draw, fill=thwsblue!15, minimum size=0.5cm, outer sep=3pt]
\tikzstyle{conn}   = [thick, -Stealth, color=thwsblue]
\tikzstyle{outarr} = [thick, -Stealth]
\tikzstyle{annot} = [color=carnelian, font=\small]

% Input nodes
\node[input] (x1) at (-2.5,  0.9) {$x_1$};
\node[input] (x2) at (-2.5,  0.0) {$x_2$};
\node[input] (xd) at (-2.5, -1.4) {$x_d$};

% Neuron
\node[neuron] (n) at (0, 0) {$z$};
% Bias
\node[input, fill=thwsblue!5] (b1) at (0, 1.7) {$1$};

% Connections
\draw[conn] (x1) -- (n) node[midway, above] {$w_1$};
\draw[conn] (x2) -- (n) node[midway, above] {$w_2$};
\draw[conn] (xd) -- (n) node[midway, above] {$w_d$};
\draw[dotted, thick] (x2) -- (xd);
\draw[conn] (b1) -- (n) node[midway, right] {$b$};

% Output
\draw[outarr] (n) -- ++(1.0, 0.0) node[right] (z) {$z$};

% Labels
\node[annot] [below=0.1 of xd] {Inputs};
\node[annot] [below=0.1 of n] {Node};
\node[annot] [below=0.1 of z.south east] {Output};

\end{tikzpicture}
