\begin{tikzpicture}[scale=0.85]
\tikzstyle{every node}=[font=\scriptsize]
\tikzstyle{neuron} = [circle, draw, fill=thwsorange!20, minimum size=0.6cm, outer sep=3pt]
\tikzstyle{input}  = [circle, draw, fill=thwsblue!15, minimum size=0.6cm, outer sep=3pt]
\tikzstyle{conn}   = [thick, -Stealth, color=thwsblue]
\tikzstyle{outarr} = [thick, -Stealth]
\tikzstyle{annot} = [color=carnelian, font=\small]

% Input nodes
\node[input] (x1) at (-2.5,  0.9) {$x_1$};
\node[input] (x2) at (-2.5,  0.0) {$x_2$};
\node[input] (xd) at (-2.5, -1.4) {$x_d$};

% Neuron
\node[neuron] (n) at (0, 0) {$\sigma$};

% Connections
\draw[conn] (x1) -- (n) node[midway, above] {$\theta_1$};
\draw[conn] (x2) -- (n) node[midway, above] {$\theta_2$};
\draw[conn] (xd) -- (n) node[midway, above] {$\theta_d$};
\draw[dotted, thick] (x2) -- (xd);

% Output
\draw[outarr] (n) -- ++(1.0, 0.0) node[right] (z) {$y$};

% Labels
\node[annot] [below=0.1 of xd] {Inputs};
\node[annot] [below=0.1 of n] {Node};
\node[annot] [below=0.1 of z.south east] {Output};

\end{tikzpicture}
