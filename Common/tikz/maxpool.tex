% maxpool.tex
% Shows 2x2 max pooling with stride 2 on a 4x4 input
% Four non-overlapping 2x2 regions, each highlighted differently
% Output: 2x2 with max values

\begin{tikzpicture}[scale=0.72]

\def\cs{0.55}  % cell size

% ── INPUT 4x4 with values ────────────────────────────────────────
% row 0 (bottom): 2 5 1 4
% row 1:          4 1 3 2
% row 2:          5 2 1 6
% row 3 (top):    1 3 2 4

% region colors (4 non-overlapping 2x2 regions):
% region A (cols 0-1, rows 2-3): thwspetrol!40  -> max=5
% region B (cols 2-3, rows 2-3): thwsorange!35  -> max=6
% region C (cols 0-1, rows 0-1): thwspetrol!20  -> max=5
% region D (cols 2-3, rows 0-1): thwsorange!20  -> max=4

% region A: cols 0-1, rows 2-3
\foreach \col in {0,1} {
    \foreach \row in {2,3} {
        \draw[thwspetrol, fill=thwspetrol!40, thick]
            (\col*\cs, \row*\cs) rectangle (\col*\cs+\cs-0.02, \row*\cs+\cs-0.02);
    }
}
% region B: cols 2-3, rows 2-3
\foreach \col in {2,3} {
    \foreach \row in {2,3} {
        \draw[thwsorange, fill=thwsorange!35, thick]
            (\col*\cs, \row*\cs) rectangle (\col*\cs+\cs-0.02, \row*\cs+\cs-0.02);
    }
}
% region C: cols 0-1, rows 0-1
\foreach \col in {0,1} {
    \foreach \row in {0,1} {
        \draw[thwspetrol, fill=thwspetrol!20, thick]
            (\col*\cs, \row*\cs) rectangle (\col*\cs+\cs-0.02, \row*\cs+\cs-0.02);
    }
}
% region D: cols 2-3, rows 0-1
\foreach \col in {2,3} {
    \foreach \row in {0,1} {
        \draw[thwsorange, fill=thwsorange!20, thick]
            (\col*\cs, \row*\cs) rectangle (\col*\cs+\cs-0.02, \row*\cs+\cs-0.02);
    }
}

% values
\node[font=\small] at (0*\cs+\cs/2, 3*\cs+\cs/2) {1};
\node[font=\small] at (1*\cs+\cs/2, 3*\cs+\cs/2) {\textbf{3}};
\node[font=\small] at (2*\cs+\cs/2, 3*\cs+\cs/2) {2};
\node[font=\small] at (3*\cs+\cs/2, 3*\cs+\cs/2) {\textbf{4}};
\node[font=\small] at (0*\cs+\cs/2, 2*\cs+\cs/2) {\textbf{5}};
\node[font=\small] at (1*\cs+\cs/2, 2*\cs+\cs/2) {2};
\node[font=\small] at (2*\cs+\cs/2, 2*\cs+\cs/2) {1};
\node[font=\small] at (3*\cs+\cs/2, 2*\cs+\cs/2) {\textbf{6}};
\node[font=\small] at (0*\cs+\cs/2, 1*\cs+\cs/2) {4};
\node[font=\small] at (1*\cs+\cs/2, 1*\cs+\cs/2) {1};
\node[font=\small] at (2*\cs+\cs/2, 1*\cs+\cs/2) {3};
\node[font=\small] at (3*\cs+\cs/2, 1*\cs+\cs/2) {2};
\node[font=\small] at (0*\cs+\cs/2, 0*\cs+\cs/2) {2};
\node[font=\small] at (1*\cs+\cs/2, 0*\cs+\cs/2) {\textbf{5}};
\node[font=\small] at (2*\cs+\cs/2, 0*\cs+\cs/2) {1};
\node[font=\small] at (3*\cs+\cs/2, 0*\cs+\cs/2) {\textbf{4}};

% dashed dividing lines
\draw[thwsgrey, dashed, thick] (2*\cs, 0) -- (2*\cs, 4*\cs);
\draw[thwsgrey, dashed, thick] (0, 2*\cs) -- (4*\cs, 2*\cs);

\node[font=\scriptsize, thwspetrol] at (2*\cs, -0.35) {input $4{\times}4$};

% ── ARROW ────────────────────────────────────────────────────────
\draw[->, thick, thwsgrey] (4*\cs+0.15, 2*\cs) -- (4*\cs+0.6, 2*\cs);
\node[font=\scriptsize, thwspetrol, align=center] at (6.5*\cs+0.38, 3*\cs+0.3)
    {$2{\times}2$\\[-2pt]stride 2};

% ── OUTPUT 2x2 ───────────────────────────────────────────────────
\def\xout{4*\cs+0.75}
\draw[thwspetrol, fill=thwspetrol!40, thick]
    (\xout, \cs) rectangle (\xout+\cs-0.02, 2*\cs-0.02);
\draw[thwsorange, fill=thwsorange!35, thick]
    (\xout+\cs, \cs) rectangle (\xout+2*\cs-0.02, 2*\cs-0.02);
\draw[thwspetrol, fill=thwspetrol!20, thick]
    (\xout, 0) rectangle (\xout+\cs-0.02, \cs-0.02);
\draw[thwsorange, fill=thwsorange!20, thick]
    (\xout+\cs, 0) rectangle (\xout+2*\cs-0.02, \cs-0.02);

\node[font=\small] at (\xout+\cs/2,   \cs+\cs/2) {\textbf{5}};
\node[font=\small] at (\xout+\cs+\cs/2, \cs+\cs/2) {\textbf{6}};
\node[font=\small] at (\xout+\cs/2,   \cs/2) {\textbf{5}};
\node[font=\small] at (\xout+\cs+\cs/2, \cs/2) {\textbf{4}};

\node[font=\scriptsize, thwspetrol] at (\xout+\cs, -0.35) {output $2{\times}2$};

\end{tikzpicture}
