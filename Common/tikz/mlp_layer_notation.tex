% Common/tikz/mlp_layer_notation.tex
\begin{tikzpicture}[scale=0.85]
\tikzstyle{every node}=[font=\scriptsize]
\tikzstyle{neuron} = [circle, draw, fill=thwsorange!20, minimum size=0.6cm, outer sep=3pt]
\tikzstyle{input}  = [circle, draw, fill=thwsblue!15, minimum size=0.6cm, outer sep=3pt]
\tikzstyle{conn}   = [thick, -Stealth, color=thwsblue]

% Layer l-1
\node[neuron] (a11) at (0,  0.5) {};
\node[neuron] (a12) at (0, -0.5) {};
\node[neuron] (a13) at (0, -2.0) {};
\draw[dotted, thick] (a12) -- (a13);

% Layer l
\node[neuron] (a21) at (2,  0.5) {};
\node[neuron] (a22) at (2, -0.5) {};
\node[neuron] (a23) at (2, -2.0) {};
\draw[dotted, thick] (a22) -- (a23);

% Layer l+1
\node[neuron] (a31) at (4,  0.5) {};
\node[neuron] (a32) at (4, -0.5) {};
\node[neuron] (a33) at (4, -2.0) {};
\draw[dotted, thick] (a32) -- (a33);

% Connections l-1 -> l
\foreach \a in {a11, a12, a13} {
    \draw[conn] (\a) -- (a21);
    \draw[conn] (\a) -- (a22);
    \draw[conn] (\a) -- (a23);
}

% Connections l -> l+1
\foreach \a in {a21, a22, a23} {
    \draw[conn] (\a) -- (a31);
    \draw[conn] (\a) -- (a32);
    \draw[conn] (\a) -- (a33);
}

% Incoming arrow to layer l-1
\draw[conn] (-1.2, -0.75) -- (a11);
\draw[conn] (-1.2, -0.75) -- (a12);
\draw[conn] (-1.2, -0.75) -- (a13);
\node at (-1.5, -0.75) {$\cdots$};

% Outgoing arrow from layer l+1
\draw[conn] (a31) -- (5.2, -0.75);
\draw[conn] (a32) -- (5.2, -0.75);
\draw[conn] (a33) -- (5.2, -0.75);
\node at (5.5, -0.75) {$\cdots$};

% Layer labels
\node[thwsorange] at (0,  1.3) {$\vh^{(l-1)}$};
\node[thwsorange] at (2,  1.3) {$\vh^{(l)}$};
\node[thwsorange] at (4,  1.3) {$\vh^{(l+1)}$};

% Theta labels on arrows
\node[thwspetrol, font=\tiny] at (1, 0.8) {$\thetamat{l}$};
\node[thwspetrol, font=\tiny] at (3, 0.8) {$\thetamat{l+1}$};

\end{tikzpicture}
