\section{The Sequence Problem}

\begin{frame}{The Sequence Problem}
  \vskip 2em

  Many real-world inputs are \structure{ordered and variable-length} — their
  meaning depends on position and context.

  \vskip 1.2em
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Sequence inputs}
      \begin{itemize}
        \item Natural language: \textit{``The food was not good''}
        \item Speech: 16\,000 samples/second
        \item Time series: stock prices, sensor readings
        \item Video: ordered frames
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Key properties}
      \begin{itemize}
        \item \structure{Variable length} — no fixed input dimension
        \item \structure{Order matters} — permuting changes meaning
        \item \structure{Long-range dependencies} — word at step 1 may determine meaning at step 50
      \end{itemize}
    \end{column}
  \end{columns}

  \vskip 1.2em
  \begin{center}
    {\setlength{\fboxsep}{8pt}\colorbox{thwsorange!10}{%
      \structure{We need a model that processes sequences
        step-by-step and accumulates context.}}}
  \end{center}
\end{frame}

% ------------------------------------------------------------------

\begin{frame}{Three Approaches to Sequence Modeling}
  \vskip 2em

  \begin{center}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{lccc}
    \toprule
    & \textbf{MLP} & \textbf{1D CNN} & \textbf{RNN} \\
    \midrule
    Handles variable length    & \alert{no}        & \alert{no*}  & \structure{yes} \\
    Long-range dependencies    & \alert{no}        & limited      & \structure{yes} \\
    Parallelizable training    & \structure{yes}   & \structure{yes} & \alert{no} \\
    Parameter sharing in time  & \alert{no}        & \structure{yes (local)} & \structure{yes (global)} \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vskip 0.8em
  \small{*requires padding or global pooling}

  \vskip 1.0em
  \textbf{1D CNNs} — fast, parallelizable, strong for \structure{local patterns}
  (n-grams, short-term trends); receptive field bounded by depth.
  Useful when long-range context is not needed.

  \vskip 0.6em
  \textbf{RNNs} — process step-by-step; \structure{unbounded context}
  via hidden state; sequential by design.
  \alert{This session focuses on RNNs.}
\end{frame}

% ------------------------------------------------------------------

\begin{frame}{The Recurrence Idea}
  \vskip 2em

  At each timestep $t$, compute \structure{hidden state}
  $\mathbf{h}^{(t)}$ from current input and previous state:
  \[
    \mathbf{h}^{(t)} = f\!\left(\mathbf{x}^{(t)},\, \mathbf{h}^{(t-1)}\right),
    \qquad \mathbf{h}^{(0)} = \mathbf{0}
  \]

  \vskip 0.8em
  \begin{center}\resizebox{0.8\textwidth}{!}{%
    \tikzsetnextfilename{rnn_unrolled_recurrence}
    \input{\CommonPath/tikz/rnn_unrolled_recurrence.tex}
  }
  \end{center}

  \vskip 0.8em
  \begin{center}
    {\setlength{\fboxsep}{8pt}\colorbox{thwsorange!10}{%
      \structure{Same $f$, same weights $\mathbf{W}_h$, $\mathbf{W}_x$
        at every timestep — this \emph{is} parameter sharing across time.}}}
  \end{center}
\end{frame}

\begin{frame}{The RNN Cell: Forward Pass}
  \vskip 2em

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \tikzsetnextfilename{rnn_cell_rolled}
      \input{\CommonPath/tikz/rnn_cell_rolled.tex}
    \end{column}
    \begin{column}{0.45\textwidth}
      \vskip 0.5em
      \[
        \mathbf{h}^{(t)} = \tanh\!\bigl(
          \underbrace{\mathbf{W}_h}_{\text{recurrent}} \mathbf{h}^{(t-1)}
          + \underbrace{\mathbf{W}_x}_{\text{input}} \mathbf{x}^{(t)}
          + \mathbf{b}
        \bigr)
      \]
      \[
        \mathbf{y}^{(t)} = \mathbf{W}_y\, \mathbf{h}^{(t)}
      \]
      \vskip 0.8em
      \begin{itemize}
        \item $\mathbf{W}_h \in \mathbb{R}^{d_h \times d_h}$ — mixes
              previous hidden state
        \item $\mathbf{W}_x \in \mathbb{R}^{d_h \times d_x}$ — projects
              current input
        \item $\tanh$ squashes to $(-1, 1)$; keeps hidden state bounded
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
